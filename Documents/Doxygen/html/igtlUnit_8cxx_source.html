62int Unit::Append(int unit, int exp)
63{
64 // Check number of units already appended
65 if (this->m_NUnits >= 6)
66 {
67 return 0;
68 }
69
70 // Check range
71 if (unit < SI_BASE_NONE || unit >= SI_NUM_UNIT_TYPES)
72 {
73 return 0;
74 }
75 if (exp < -7 || exp > 7)
76 {
77 return 0;
78 }
79
80 // Append
81 this->m_Unit[this->m_NUnits] = (igtlUint8)unit;
82 this->m_Exp[this->m_NUnits] = (igtlInt8)exp;
83 this->m_NUnits++;
84 return 1;
85}
