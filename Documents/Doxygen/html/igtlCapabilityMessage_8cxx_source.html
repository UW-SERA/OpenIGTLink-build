95int CapabilityMessage::PackContent()
96{
97 AllocateBuffer();
98
99 if (this->m_TypeNames.size() == 0)
100 {
101 return 0;
102 }
103
104 igtl_capability_info info;
105 int nTypes = this->m_TypeNames.size();
106
107 igtl_capability_init_info(&info);
108 info.ntypes = nTypes;
109 igtl_capability_alloc_info(&info, nTypes);
110
111 for(int i = 0; i < nTypes; i++ )
112 {
113 memcpy(info.typenames[i], this->m_TypeNames[i].c_str(), IGTL_HEADER_TYPE_SIZE);
114 }
115
116 igtl_capability_pack(&info, this->m_Content);
117
118 return 1;
119}

122int CapabilityMessage::UnpackContent()
123{
124
125 igtl_capability_info info;
126
127 igtl_capability_init_info(&info);
128 bool isUnpacked(true);
129 igtl_capability_unpack(this->m_Content, &info, this->CalculateReceiveContentSize(isUnpacked));
130
131 int ntypes = info.ntypes;
132
133 if(ntypes == 0)
134 {
135 return 0;
136 }
137
138 this->m_TypeNames.clear();
139 for(int i = 0; i < ntypes; i++)
140 {
141 std::string buf;
142 if (igtl::Strnlen((const char*)info.typenames[i], IGTL_HEADER_TYPE_SIZE) < IGTL_HEADER_TYPE_SIZE)
143 {
144 buf.append((const char*)info.typenames[i]);
145 }
146 else
147 {
148 buf.append((const char*)info.typenames[i], IGTL_HEADER_TYPE_SIZE);
149 }
150 this->m_TypeNames.push_back(buf);
151 }
152
153
154 return 1;
155}
