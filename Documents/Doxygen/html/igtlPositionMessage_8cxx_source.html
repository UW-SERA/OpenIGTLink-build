168int PositionMessage::PackContent()
169{
170 // Allocate buffer
171 AllocateBuffer();
172 igtl_position* p = NULL;
173#if OpenIGTLink_HEADER_VERSION >= 2
174 p = (igtl_position*)(this->m_Content);
175#elif OpenIGTLink_PROTOCOL_VERSION <=2
176 p = (igtl_position*)(this->m_Body);
177#endif
178
179 p->position[0] = this->m_Position[0];
180 p->position[1] = this->m_Position[1];
181 p->position[2] = this->m_Position[2];
182
183 if (this->m_PackType == WITH_QUATERNION3)
184 {
185 p->quaternion[0] = this->m_Quaternion[0];
186 p->quaternion[1] = this->m_Quaternion[1];
187 p->quaternion[2] = this->m_Quaternion[2];
188 }
189 else if (this->m_PackType == ALL)
190 {
191 p->quaternion[0] = this->m_Quaternion[0];
192 p->quaternion[1] = this->m_Quaternion[1];
193 p->quaternion[2] = this->m_Quaternion[2];
194 p->quaternion[3] = this->m_Quaternion[3];
195 }
196
197 switch (this->m_PackType)
198 {
199 case POSITION_ONLY:
200 igtl_position_convert_byte_order_position_only(p);
201 break;
202 case WITH_QUATERNION3:
203 igtl_position_convert_byte_order_quaternion3(p);
204 break;
205 default: //IGTL_POSITION_MESSAGE_DEFAULT_SIZE
206 igtl_position_convert_byte_order(p);
207 break;
208 }
209
210 return 1;
211}

213int PositionMessage::UnpackContent()
214{
215 igtl_position* p = NULL;
216#if OpenIGTLink_HEADER_VERSION >= 2
217 p = (igtl_position*)(this->m_Content);
218#elif OpenIGTLink_PROTOCOL_VERSION <=2
219 p = (igtl_position*)(this->m_Body);
220#endif
221
222 bool isUnpacked(true);
223 igtl_uint64 contentSize = CalculateReceiveContentSize(isUnpacked);
224 if( !isUnpacked)
225 {
226 return 0;
227 }
228 switch (contentSize)
229 {
230 case IGTL_POSITION_MESSAGE_POSITON_ONLY_SIZE:
231 this->m_PackType = POSITION_ONLY;
232 igtl_position_convert_byte_order_position_only(p);
233 break;
234 case IGTL_POSITION_MESSAGE_WITH_QUATERNION3_SIZE:
235 this->m_PackType = WITH_QUATERNION3;
236 igtl_position_convert_byte_order_quaternion3(p);
237 break;
238 default: //IGTL_POSITION_MESSAGE_DEFAULT_SIZE
239 this->m_PackType = ALL;
240 igtl_position_convert_byte_order(p);
241 break;
242 }
243
244 this->m_Position[0] = p->position[0];
245 this->m_Position[1] = p->position[1];
246 this->m_Position[2] = p->position[2];
247
248 if (this->m_PackType == POSITION_ONLY)
249 {
250 this->m_Quaternion[0] = 0.0;
251 this->m_Quaternion[1] = 0.0;
252 this->m_Quaternion[2] = 0.0;
253 this->m_Quaternion[3] = 1.0;
254 }
255
256 if (this->m_PackType == WITH_QUATERNION3)
257 {
258 float ox = p->quaternion[0];
259 float oy = p->quaternion[1];
260 float oz = p->quaternion[2];
261 float sq = 1.0 - (ox*ox + oy*oy + oz*oz);
262 if (sq < 0.0)
263 {
264 // TODO: what should we do with invalid quaternion?
265 // Tentatively we set (0, 0, 0, 1);
266 this->m_Quaternion[0] = 0.0;
267 this->m_Quaternion[1] = 0.0;
268 this->m_Quaternion[2] = 0.0;
269 this->m_Quaternion[3] = 1.0;
270 }
271 else
272 {
273 float w = sqrt(sq);
274 this->m_Quaternion[0] = ox;
275 this->m_Quaternion[1] = oy;
276 this->m_Quaternion[2] = oz;
277 this->m_Quaternion[3] = w;
278 }
279 }
280 else if (this->m_PackType == ALL)
281 {
282 this->m_Quaternion[0] = p->quaternion[0];
283 this->m_Quaternion[1] = p->quaternion[1];
284 this->m_Quaternion[2] = p->quaternion[2];
285 this->m_Quaternion[3] = p->quaternion[3];
286 }
287
288 return 1;
289}
