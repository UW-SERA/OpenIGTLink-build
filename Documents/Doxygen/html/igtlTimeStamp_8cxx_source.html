164void TimeStamp::SetTime(double tm)
165{
166 double second = floor(tm);
167 this->m_Second = static_cast<igtlInt32>(second);
168 this->m_Nanosecond = static_cast<igtlInt32>((tm - second)*1e9);
169}

181void TimeStamp::SetTime(igtlUint64 tm)
182{
183 // Export from 64-bit fixed-point expression used in OpenIGTLink
184 igtlInt32 sec = static_cast<igtlInt32>((tm >> 32 ) & 0xFFFFFFFF);
185 igtlInt32 fraction = static_cast<igtlInt32>(tm & 0xFFFFFFFF);
186 this->m_Second = sec;
187 this->m_Nanosecond = igtl_frac_to_nanosec(static_cast<igtlUint32>(fraction));
188}

191//-----------------------------------------------------------------------------
192void TimeStamp::SetTimeInNanoseconds(igtlUint64 tm)
193{
194 igtlUint64 sec = tm / 1e9; // integer rounding
195 igtlUint64 nano = sec * 1e9; // round it back up to get whole number of seconds expressed in nanoseconds.
196 this->m_Second = static_cast<igtlInt32>(sec);
197 this->m_Nanosecond = static_cast<igtlInt32>(tm - nano);
198}

201double TimeStamp::GetTimeStamp()
202{
203 double tm;
204 tm = static_cast<double>(this->m_Second) +
205 static_cast<double>(this->m_Nanosecond) / 1e9;
206
207 return tm;
208}

217igtlUint64 TimeStamp::GetTimeStampUint64()
218{
219 // Export as 64-bit fixed-point expression used in OpenIGTLink
220 igtlInt32 sec = this->m_Second;
221 igtlInt32 fraction = igtl_nanosec_to_frac(this->m_Nanosecond);
222
223 igtlUint64 ts = sec & 0xFFFFFFFF;
224 ts = (ts << 32) | (fraction & 0xFFFFFFFF);
225
226 return ts;
227}
