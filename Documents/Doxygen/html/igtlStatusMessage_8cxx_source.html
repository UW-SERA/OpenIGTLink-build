96igtlUint64 StatusMessage::CalculateContentBufferSize()
97{
98 // The body size sum of the header size and status message size.

103int StatusMessage::PackContent()
104{
105 // Allocate buffer
106 AllocateBuffer();
107 m_StatusHeader = this->m_Content;
108
109 m_StatusMessage = (char*)&m_StatusHeader[IGTL_STATUS_HEADER_SIZE];
110 igtl_status_header* status_header = (igtl_status_header*)this->m_StatusHeader;
111
112 status_header->code = static_cast<igtlUint16>(this->m_Code);
113 status_header->subcode = this->m_SubCode;
114 strncpy(status_header->error_name, this->m_ErrorName, IGTL_STATUS_ERROR_NAME_LENGTH);
115 strcpy(this->m_StatusMessage, this->m_StatusMessageString.c_str());
116
117 igtl_status_convert_byte_order(status_header);
118
119 return 1;
120}

122int StatusMessage::UnpackContent()
123{
124 m_StatusHeader = this->m_Content;
125 m_StatusMessage = (char*)&m_StatusHeader[IGTL_STATUS_HEADER_SIZE];
126
127 igtl_status_header* status_header = (igtl_status_header*)this->m_StatusHeader;
128 igtl_status_convert_byte_order(status_header);
129
130 this->m_Code = status_header->code;
131 this->m_SubCode = status_header->subcode;
132 this->m_ErrorName[IGTL_STATUS_ERROR_NAME_LENGTH-1] = '\0';
133 strncpy(this->m_ErrorName, status_header->error_name, IGTL_STATUS_ERROR_NAME_LENGTH);
134
135 // make sure that the status message in the pack ends with '\0'
136 bool isUnpacked(true);
137 if (m_StatusMessage[CalculateReceiveContentSize(isUnpacked)-IGTL_STATUS_HEADER_SIZE-1] == '\0')
138 {
139 this->m_StatusMessageString = m_StatusMessage;
140 }
141 else
142 {
143 //std::cerr << "status message in the pack does not end with '\0'" << std::endl;
144 }
145
146 return 1;
147}
