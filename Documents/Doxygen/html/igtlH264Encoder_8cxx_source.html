138int H264Encoder::FillSpecificParameters() {
139 /* Test for temporal, spatial, SNR scalability */
140 sSvcParam.iUsageType = CAMERA_VIDEO_REAL_TIME;
141 sSvcParam.fMaxFrameRate = 200.0f; // input frame rate
142 sSvcParam.iPicWidth = 256; // width of picture in samples
143 sSvcParam.iPicHeight = 256; // height of picture in samples
144 sSvcParam.iTargetBitrate = 2500000; // target bitrate desired
145 sSvcParam.iMaxBitrate = UNSPECIFIED_BIT_RATE;
146 sSvcParam.iRCMode = RC_OFF_MODE; // rc mode control
147 sSvcParam.bIsLosslessLink = true;
148 sSvcParam.iTemporalLayerNum = 1; // layer number at temporal level
149 sSvcParam.iSpatialLayerNum = 1; // layer number at spatial level
150 sSvcParam.iEntropyCodingModeFlag = 1;
151 sSvcParam.bEnableDenoise = 0; // denoise control
152 sSvcParam.bEnableBackgroundDetection = 1; // background detection control
153 sSvcParam.bEnableAdaptiveQuant = 1; // adaptive quantization control
154 sSvcParam.bEnableFrameSkip = 0; // frame skipping
155 sSvcParam.bEnableLongTermReference = 0; // long term reference control
156 sSvcParam.iLtrMarkPeriod = 30;
157 sSvcParam.uiIntraPeriod = 16; // period of Intra frame
158 sSvcParam.bUseLoadBalancing = false;
159 sSvcParam.eSpsPpsIdStrategy = INCREASING_ID;
160 sSvcParam.bPrefixNalAddingCtrl = 0;
161 sSvcParam.iComplexityMode = HIGH_COMPLEXITY;
162 sSvcParam.bSimulcastAVC = false;
163 sSvcParam.iMaxQp = 51;
164 sSvcParam.iMinQp = 0;
165 int iIndexLayer = 0;
166 sSvcParam.sSpatialLayers[iIndexLayer].uiProfileIdc = PRO_BASELINE;
167 sSvcParam.sSpatialLayers[iIndexLayer].uiLevelIdc = LEVEL_5_2;
168 sSvcParam.sSpatialLayers[iIndexLayer].iVideoWidth = 256;
169 sSvcParam.sSpatialLayers[iIndexLayer].iVideoHeight = 256;
170 sSvcParam.sSpatialLayers[iIndexLayer].iDLayerQp = 1;
171 sSvcParam.sSpatialLayers[iIndexLayer].fFrameRate = 60.0;
172 sSvcParam.sSpatialLayers[iIndexLayer].iSpatialBitrate = 1500000;
173 sSvcParam.sSpatialLayers[iIndexLayer].iMaxSpatialBitrate = UNSPECIFIED_BIT_RATE;
174 sSvcParam.sSpatialLayers[iIndexLayer].sSliceArgument.uiSliceMode = SM_SINGLE_SLICE;
175 sSvcParam.sSpatialLayers[iIndexLayer].sSliceArgument.uiSliceNum = 1;
176
177 float fMaxFr = sSvcParam.sSpatialLayers[sSvcParam.iSpatialLayerNum - 1].fFrameRate;
178 for (int32_t i = sSvcParam.iSpatialLayerNum - 2; i >= 0; --i) {
179 if (sSvcParam.sSpatialLayers[i].fFrameRate > (fMaxFr+EPSN))
180 fMaxFr = sSvcParam.sSpatialLayers[i].fFrameRate;
181 }
182 sSvcParam.fMaxFrameRate = fMaxFr;
183
184 return 0;

234int H264Encoder::ParseLayerConfig (string strTag[], const int iLayer, SEncParamExt& pSvcParam) {
235
236 SSpatialLayerConfig* pDLayer = &pSvcParam.sSpatialLayers[iLayer];
237 int iLeftTargetBitrate = (pSvcParam.iRCMode != RC_OFF_MODE) ? pSvcParam.iTargetBitrate : 0;
238 SLayerPEncCtx sLayerCtx;
239 memset (&sLayerCtx, 0, sizeof (SLayerPEncCtx));
240
241 string str_ ("SlicesAssign");
242 const int kiSize = (int)str_.size();
243
244 if (!strTag[0].empty())
245 {
246 if (strTag[0].compare ("FrameWidth") == 0)
247 {
248 pDLayer->iVideoWidth = atoi (strTag[1].c_str());
249 }
250 else if (strTag[0].compare ("FrameHeight") == 0)
251 {
252 pDLayer->iVideoHeight = atoi (strTag[1].c_str());
253 }
254 else if (strTag[0].compare ("FrameRateOut") == 0)
255 {
256 pDLayer->fFrameRate = (float)atof (strTag[1].c_str());
257 }
258 else if (strTag[0].compare ("ProfileIdc") == 0)
259 {
260 pDLayer->uiProfileIdc = (EProfileIdc)atoi (strTag[1].c_str());
261 }
262 else if (strTag[0].compare ("SpatialBitrate") == 0)
263 {
264 pDLayer->iSpatialBitrate = 1000 * atoi (strTag[1].c_str());
265 if (pSvcParam.iRCMode != RC_OFF_MODE)
266 {
267 if (pDLayer->iSpatialBitrate <= 0)
268 {
269 fprintf (stderr, "Invalid spatial bitrate(%d) in dependency layer #%d.\n", pDLayer->iSpatialBitrate, iLayer);
270 return -1;
271 }
272 if (pDLayer->iSpatialBitrate > iLeftTargetBitrate)
273 {
274 fprintf (stderr, "Invalid spatial(#%d) bitrate(%d) setting due to unavailable left(%d)!\n", iLayer,
275 pDLayer->iSpatialBitrate, iLeftTargetBitrate);
276 return -1;
277 }
278 iLeftTargetBitrate -= pDLayer->iSpatialBitrate;
279 }
280 }
281 else if (strTag[0].compare ("MaxSpatialBitrate") == 0)
282 {
283 pDLayer->iMaxSpatialBitrate = 1000 * atoi (strTag[1].c_str());
284 if (pSvcParam.iRCMode != RC_OFF_MODE)
285 {
286 if (pDLayer->iMaxSpatialBitrate < 0)
287 {
288 fprintf (stderr, "Invalid max spatial bitrate(%d) in dependency layer #%d.\n", pDLayer->iMaxSpatialBitrate, iLayer);
289 return -1;
290 }
291 if (pDLayer->iMaxSpatialBitrate > 0 && pDLayer->iMaxSpatialBitrate < pDLayer->iSpatialBitrate)
292 {
293 fprintf (stderr, "Invalid max spatial(#%d) bitrate(%d) setting::: < layerBitrate(%d)!\n", iLayer,
294 pDLayer->iMaxSpatialBitrate, pDLayer->iSpatialBitrate);
295 return -1;
296 }
297 }
298 }
299 else if (strTag[0].compare ("InitialQP") == 0)
300 {
301 sLayerCtx.iDLayerQp = atoi (strTag[1].c_str());
302 }
303 else if (strTag[0].compare ("SliceMode") == 0)
304 {
305 sLayerCtx.sSliceArgument.uiSliceMode = (SliceModeEnum)atoi (strTag[1].c_str());
306 }
307 else if (strTag[0].compare ("SliceSize") == 0)
308 { //SM_SIZELIMITED_SLICE
309 sLayerCtx.sSliceArgument.uiSliceSizeConstraint = atoi (strTag[1].c_str());
310 }
311 else if (strTag[0].compare ("SliceNum") == 0)
312 {
313 sLayerCtx.sSliceArgument.uiSliceNum = atoi (strTag[1].c_str());
314 }
315 else if (strTag[0].compare (0, kiSize, str_) == 0)
316 {
317 const char* kpString = strTag[0].c_str();
318 int uiSliceIdx = atoi (&kpString[kiSize]);
319 sLayerCtx.sSliceArgument.uiSliceMbNum[uiSliceIdx] = atoi (strTag[1].c_str());
320 }
321 }
322 pDLayer->iDLayerQp = sLayerCtx.iDLayerQp;
323 pDLayer->sSliceArgument.uiSliceMode = sLayerCtx.sSliceArgument.uiSliceMode;
324
325 memcpy (&pDLayer->sSliceArgument, &sLayerCtx.sSliceArgument, sizeof (SSliceArgument)); // confirmed_safe_unsafe_usage
326 memcpy (&pDLayer->sSliceArgument.uiSliceMbNum[0], &sLayerCtx.sSliceArgument.uiSliceMbNum[0],
327 sizeof (sLayerCtx.sSliceArgument.uiSliceMbNum)); // confirmed_safe_unsafe_usage
328
329 return 0;
330}

332int H264Encoder::ParseConfig() {
333 string strTag[4];
334
335 while (!cRdCfg.EndOfFile())
336 {
337 long iRd = cRdCfg.ReadLine (&strTag[0]);
338 if (iRd > 0)
339 {
340 if (strTag[0].empty())
341 continue;
342 if (strTag[0].compare ("PicWidth") == 0)
343 {
344 sSvcParam.iPicWidth = atoi(strTag[1].c_str());
345 }
346 if (strTag[0].compare ("PicHeight") == 0)
347 {
348 sSvcParam.iPicHeight = atoi(strTag[1].c_str());
349 }
350 if (strTag[0].compare ("UsageType") == 0)
351 {
352 sSvcParam.iUsageType = (EUsageType)atoi (strTag[1].c_str());
353 }
354 else if (strTag[0].compare ("SimulcastAVC") == 0)
355 {
356 sSvcParam.bSimulcastAVC = atoi (strTag[1].c_str()) ? true : false;
357 }
358 else if (strTag[0].compare ("TemporalLayerNum") == 0)
359 {
360 sSvcParam.iTemporalLayerNum = atoi (strTag[1].c_str());
361 }
362 else if (strTag[0].compare ("IntraPeriod") == 0)
363 {
364 sSvcParam.uiIntraPeriod = atoi (strTag[1].c_str());
365 }
366 else if (strTag[0].compare ("MaxNalSize") == 0)
367 {
368 sSvcParam.uiMaxNalSize = atoi (strTag[1].c_str());
369 }
370 else if (strTag[0].compare ("SpsPpsIDStrategy") == 0)
371 {
372 int32_t iValue = atoi (strTag[1].c_str());
373 switch (iValue)
374 {
375 case 0:
376 sSvcParam.eSpsPpsIdStrategy = CONSTANT_ID;
377 break;
378 case 0x01:
379 sSvcParam.eSpsPpsIdStrategy = INCREASING_ID;

383 break;
384 case 0x03:
385 sSvcParam.eSpsPpsIdStrategy = SPS_LISTING_AND_PPS_INCREASING;
386 break;
387 case 0x06:
388 sSvcParam.eSpsPpsIdStrategy = SPS_PPS_LISTING;
389 break;
390 default:
391 sSvcParam.eSpsPpsIdStrategy = CONSTANT_ID;
392 break;
393 }
394 }
395 else if (strTag[0].compare ("EnableScalableSEI") == 0)
396 {
397 sSvcParam.bEnableSSEI = atoi (strTag[1].c_str()) ? true : false;
398 }
399 else if (strTag[0].compare ("EnableFrameCropping") == 0)
400 {
401 sSvcParam.bEnableFrameCroppingFlag = (atoi (strTag[1].c_str()) != 0);
402 }
403 else if (strTag[0].compare ("EntropyCodingModeFlag") == 0)

406 }

469 sSvcParam.iMaxQp = atoi (strTag[1].c_str());

475 else if (strTag[0].compare ("EnableDenoise") == 0)

478 }
479 else if (strTag[0].compare ("EnableSceneChangeDetection") == 0)

502 }

524 if (ctrl_flag > 1)

528 sSvcParam.bPrefixNalAddingCtrl = ctrl_flag ? true : false;

530 }
531 }
532 igtlUint8 actLayerConfigNum = 0;

534 while (!cRdCfg.EndOfFile())
535 {
536 long iRd = cRdCfg.ReadLine (&strTag[0]);

546 SSpatialLayerConfig* pDLayer = &sSvcParam.sSpatialLayers[layerIndex];

548 pDLayer->iVideoHeight = this->sSvcParam.iPicHeight;

550 }
551 }
552 if(layerIndex>=0)
553 {
554 this->ParseLayerConfig(strTag, layerIndex, sSvcParam);
555 }
556 }
557 }
558
559 const igtlUint8 kiActualLayerNum = sSvcParam.iSpatialLayerNum<actLayerConfigNum ? sSvcParam.iSpatialLayerNum:actLayerConfigNum;
560 if (sSvcParam.iSpatialLayerNum >
561 kiActualLayerNum) { // fixed number of dependency layer due to parameter error in settings
562 sSvcParam.iSpatialLayerNum = kiActualLayerNum;
563 }
564 return 0;
565}

594int H264Encoder::InitializeEncoder()
595{
596 //------------------------------------------------------------
597 int iRet = 0;
598
599 if (this->configFile=="" && this->pSVCEncoder)
600 {
601 fprintf (stderr, "No configuration file specified. Use Default Parameters\n");
602 iRet = this->pSVCEncoder->InitializeExt (&sSvcParam);
603 if (iRet) {
604 fprintf (stderr, "parse svc parameter config file failed.\n");
605 goto INSIDE_MEM_FREE;
606 }
607 this->initializationDone = true;
608 return iRet;
609 }
610 else
611 {
612 cRdCfg.OpenFile (this->configFile.c_str());// to do get the first augments from this->augments.
613 if (cRdCfg.ExistFile())
614 {
615 iRet = ParseConfig();
616 if (iRet) {
617 fprintf (stderr, "parse svc parameter config file failed.\n");
618 goto INSIDE_MEM_FREE;
619 }
620 }
621 else
622 {
623 fprintf (stderr, "Specified file: %s not exist, maybe invalid path or parameter settting.\n",
624 cRdCfg.GetFileName().c_str());
625 iRet = 1;
626 goto INSIDE_MEM_FREE;
627 }

666igtlInt32 H264Encoder::InitPic (const void* kpSrc, const int32_t colorspace, const int32_t width, const int32_t height) {
667 SSourcePicture* pSrcPic = (SSourcePicture*)kpSrc;
668
669 if (NULL == pSrcPic || width == 0 || height == 0)
670 return 1;
671
672 pSrcPic->iColorFormat = colorspace;
673 pSrcPic->iPicWidth = width;
674 pSrcPic->iPicHeight = height;
675
676 //currently encoder only supports videoFormatI420.
677 if ((colorspace & (~videoFormatVFlip)) != videoFormatI420)
678 return 2;
679 switch (colorspace & (~videoFormatVFlip)) {
680 case videoFormatI420:
681 case videoFormatYV12:
682 pSrcPic->pData[0] = NULL;
683 pSrcPic->pData[1] = NULL;
684 pSrcPic->pData[2] = NULL;
685 pSrcPic->pData[3] = NULL;
686 pSrcPic->iStride[0] = width;
687 pSrcPic->iStride[2] = pSrcPic->iStride[1] = width >> 1;
688 pSrcPic->iStride[3] = 0;
689 break;
690 case videoFormatYUY2:
691 case videoFormatYVYU:
692 case videoFormatUYVY:
693 pSrcPic->pData[0] = NULL;
694 pSrcPic->pData[1] = NULL;
695 pSrcPic->pData[2] = NULL;
696 pSrcPic->pData[3] = NULL;
697 pSrcPic->iStride[0] = CALC_BI_STRIDE (width, 16);
698 pSrcPic->iStride[3] = pSrcPic->iStride[2] = pSrcPic->iStride[1] = 0;
699 break;
700 case videoFormatRGB:
701 case videoFormatBGR:
702 pSrcPic->pData[0] = NULL;
703 pSrcPic->pData[1] = NULL;
704 pSrcPic->pData[2] = NULL;
705 pSrcPic->pData[3] = NULL;
706 pSrcPic->iStride[0] = CALC_BI_STRIDE (width, 24);
707 pSrcPic->iStride[3] = pSrcPic->iStride[2] = pSrcPic->iStride[1] = 0;
708 if (colorspace & videoFormatVFlip)
709 pSrcPic->iColorFormat = colorspace & (~videoFormatVFlip);
710 else
711 pSrcPic->iColorFormat = colorspace | videoFormatVFlip;
712 break;
713 case videoFormatBGRA:
714 case videoFormatRGBA:
715 case videoFormatARGB:
716 case videoFormatABGR:
717 pSrcPic->pData[0] = NULL;
718 pSrcPic->pData[1] = NULL;
719 pSrcPic->pData[2] = NULL;
720 pSrcPic->pData[3] = NULL;
721 pSrcPic->iStride[0] = width << 2;
722 pSrcPic->iStride[3] = pSrcPic->iStride[2] = pSrcPic->iStride[1] = 0;
723 if (colorspace & videoFormatVFlip)
724 pSrcPic->iColorFormat = colorspace & (~videoFormatVFlip);
725 else
726 pSrcPic->iColorFormat = colorspace | videoFormatVFlip;
727 break;
728 default:
729 return 2; // any else?
730 }
731
732 return 0;
733}

735int H264Encoder::ConvertToLocalImageFormat(SourcePicture* pSrcPic)
736{
737 if(pSrcPic->colorFormat==FormatI420)
738 {
739 if((pSrcPic->picWidth != this->h264SrcPicture.iPicWidth || pSrcPic->picHeight != this->h264SrcPicture.iPicHeight))
740 {
741 InitPic (&this->h264SrcPicture, videoFormatI420, pSrcPic->picWidth, pSrcPic->picHeight);
742 }
743 for(int i = 0; i < 4; i++)
744 {
745 h264SrcPicture.iStride[i] = pSrcPic->stride[i];
746 h264SrcPicture.pData[i] = pSrcPic->data[i];
747 }
748 h264SrcPicture.iPicWidth = pSrcPic->picWidth;
749 h264SrcPicture.iPicHeight = pSrcPic->picHeight;
750 return 1;
751 }
752 return -1;// image format not supported
753}

756int H264Encoder::EncodeSingleFrameIntoVideoMSG(SourcePicture* pSrcPic, igtl::VideoMessage* videoMessage, bool isGrayImage)
757{
758 int encodeRet = -1;
759 int iSourceWidth = pSrcPic->picWidth;
760 int iSourceHeight = pSrcPic->picHeight;
761 if (iSourceWidth != this->sSvcParam.iPicWidth || iSourceHeight != this->sSvcParam.iPicHeight)
762 {
763 this->SetPicWidthAndHeight(iSourceWidth, iSourceHeight);
764 this->InitializeEncoder();
765 }
766 if (this->initializationDone == true)
767 {
768 pSrcPic->stride[0] = iSourceWidth;
769 pSrcPic->stride[1] = pSrcPic->stride[2] = pSrcPic->stride[0] >> 1;
770 int frameSize = 0;
771 int iLayer = 0;
772 this->ConvertToLocalImageFormat(pSrcPic);
773 encodeRet = pSVCEncoder->EncodeFrame(&h264SrcPicture, &sFbi);
774 videoMessage->SetBitStreamSize(sFbi.iFrameSizeInBytes);
775 videoMessage->AllocateScalars();
776 videoMessage->SetCodecType(IGTL_VIDEO_CODEC_NAME_H264);
777 videoMessage->SetEndian(igtl_is_little_endian()==true?2:1); //little endian is 2 big endian is 1
778 videoMessage->SetWidth(pSrcPic->picWidth);
779 videoMessage->SetHeight(pSrcPic->picHeight);
780 if (sFbi.eFrameType == videoFrameTypeIDR)
781 {
782 encodedFrameType = FrameTypeKey;
783 }
784 else
785 {
786 // To do, assign other frame type too.
787 encodedFrameType = FrameTypeUnKnown;
788 }
789 if (isGrayImage)
790 {
791 encodedFrameType = sFbi.eFrameType<<8;
792 }
793 videoMessage->SetFrameType(encodedFrameType);
794 while (iLayer < sFbi.iLayerNum) {
795 SLayerBSInfo* pLayerBsInfo = &sFbi.sLayerInfo[iLayer];
796 if (pLayerBsInfo != NULL) {
797 int iLayerSize = 0;
798 int iNalIdx = pLayerBsInfo->iNalCount - 1;
799 do {
800 iLayerSize += pLayerBsInfo->pNalLengthInByte[iNalIdx];
801 -- iNalIdx;
802 } while (iNalIdx >= 0);
803 frameSize += iLayerSize;
804 for (int i = 0; i < iLayerSize ; i++)
805 {
806 videoMessage->GetPackFragmentPointer(2)[frameSize-iLayerSize+i] = pLayerBsInfo->pBsBuf[i];
807 }
808 }
809 ++ iLayer;
810 }
811 videoMessage->Pack();
812 }
813 return encodeRet;
814};
