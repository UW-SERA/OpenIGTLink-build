54//-----------------------------------------------------------------------------
55int ServerSocket::CreateServer(int port)
56{
57 if (this->m_SocketDescriptor != -1)
58 {
59 igtlWarningMacro("Server Socket already exists. Closing old socket.");
60 this->CloseSocket(this->m_SocketDescriptor);
61 this->m_SocketDescriptor = -1;
62 }
63 this->m_SocketDescriptor = this->CreateSocket();
64 if (this->m_SocketDescriptor < 0)
65 {
66 return -1;
67 }
68 if ( this->BindSocket(this->m_SocketDescriptor, port) != 0||
69 this->Listen(this->m_SocketDescriptor) != 0)
70 {
71 // failed to bind or listen.
72 this->CloseSocket(this->m_SocketDescriptor);
73 this->m_SocketDescriptor = -1;
74 return -1;
75 }
76 // Success.
77 return 0;
78}

81//ClientSocket* ServerSocket::WaitForConnection(unsigned long msec /*=0*/)
82ClientSocket::Pointer ServerSocket::WaitForConnection(unsigned long msec /*=0*/)
83{
84 if (this->m_SocketDescriptor < 0)
85 {
86 igtlErrorMacro("Server Socket not created yet!");
87 return NULL;
88 }
89
90 int ret = this->SelectSocket(this->m_SocketDescriptor, msec);
91 if (ret == 0)
92 {
93 // Timed out.
94 return NULL;
95 }
96 if (ret == -1)
97 {
98 igtlErrorMacro("Error selecting socket.");
99 return NULL;
100 }
101 int clientsock = this->Accept(this->m_SocketDescriptor);
102 if (clientsock == -1)
103 {
104 igtlErrorMacro("Failed to accept the socket.");
105 return NULL;
106 }
107 // Create a new ClientSocket and return it.
108 ClientSocket::Pointer cs = ClientSocket::New();
109 cs->m_SocketDescriptor = clientsock;
110 return cs;
111}
