168igtl_uint32 igtl_export igtl_nanosec_to_frac(igtl_uint32 nanosec)
169{
170
171 igtl_uint32 base = 1000000000; /*10^9*/
172
173 igtl_uint32 mask = 0x80000000;
174 igtl_uint32 r = 0x00000000;
175
176 do
177 {
178 base++;
179 base >>= 1;
180 if (nanosec >= base)
181 {
182 r |= mask;
183 nanosec -= base;
184 }
185 mask >>= 1;
186 }
187 while (mask);
188
189 return r;
190}

193igtl_uint32 igtl_export igtl_frac_to_nanosec(igtl_uint32 frac)
194{
195 igtl_uint32 base = 1000000000; /*10^9*/
196
197 igtl_uint32 mask = 0x80000000;
198 igtl_uint32 r = 0x00000000;
199
200 do
201 {
202 base++;
203 base >>= 1;
204 r += (frac & mask)? base : 0;
205 mask >>= 1;
206 } while (mask);
207
208 return r;
209}

212void igtl_export igtl_message_dump_hex(FILE* stream, const void* message, int bytes)
213{
214 /* Print first 256 bytes as HEX values in STDERR for debug */
215
216 unsigned char* p;
217 int row;
218 int cols_in_last_row;
219 int i;
220 int j;
221
222 p = (unsigned char*)message;
223 row = bytes / 16;
224 cols_in_last_row = bytes % 16;
225
226 for (i = 0; i < row; i ++)
227 {
228 for (j = 0; j < 16; j ++)
229 {
230 fprintf(stream, "%02x ", *p);
231 p++;
232 }
233 fprintf(stream, "\n");
234 }
235
236 for (j = 0; j < cols_in_last_row; j ++)
237 {
238 fprintf(stream, "%02x ", p[j]);
239 }
240 fprintf(stream, "\n");
241
242}

246igtl_uint32 igtl_export igtl_get_scalar_size(int type)
247{
248 igtl_uint32 bytes;
249
250 switch (type)
251 {
252 case IGTL_SCALAR_INT8:
253 case IGTL_SCALAR_UINT8:
254 bytes = 1;
255 break;
256 case IGTL_SCALAR_INT16:
257 case IGTL_SCALAR_UINT16:
258 bytes = 2;
259 break;
260 case IGTL_SCALAR_INT32:
261 case IGTL_SCALAR_UINT32:
262 case IGTL_SCALAR_FLOAT32:
263 bytes = 4;
264 break;
265 case IGTL_SCALAR_FLOAT64:
266 bytes = 8;
267 break;
268 case IGTL_SCALAR_COMPLEX:
269 bytes = 16;
270 break;
271 default:
272 bytes = 0;
273 break;
274 }
275
276 return bytes;
277}
