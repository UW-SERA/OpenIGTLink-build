234int igtl_polydata_convert_byteorder_topology(igtl_uint32 * dst, igtl_uint32 * src, igtl_uint32 size)
235{
236 igtl_uint32 * ptr32_src;
237 igtl_uint32 * ptr32_src_end;
238 igtl_uint32 * ptr32_dst;
239
240 if (size == 0)
241 {
242 return 1;
243 }
244
245 if (!src || !dst)
246 {
247 /* Return error, if either src or dst is NULL despite size>0 */
248 return 0;
249 }
250
251 if (!igtl_is_little_endian())
252 {
253 memcpy((void*)dst, (void*)src, size);
254 }
255 else
256 {
257 ptr32_src = (igtl_uint32 *) src;
258 ptr32_src_end = ptr32_src + (size/sizeof(igtl_uint32));
259 ptr32_dst = (igtl_uint32 *) dst;
260 while (ptr32_src < ptr32_src_end)
261 {
262 *ptr32_dst = BYTE_SWAP_INT32(*ptr32_src);
263 ptr32_dst ++;
264 ptr32_src ++;
265 }
266 }
267
268 return 1;
269}

272int igtl_export igtl_polydata_unpack(int type, void * byte_array, igtl_polydata_info * info, igtl_uint64 size)
273{
274 /* size = number of points (not number of bytes). In case of vertices, this is specified
275 by size_vertices in igtl_polydata_header. */
276 igtl_polydata_header * header;
277 char * ptr;
278
279 igtl_uint32 * ptr32_src;
280 igtl_uint32 * ptr32_src_end;
281 igtl_uint32 * ptr32_dst;
282 igtl_uint32 s;
283
284 igtl_polydata_attribute_header * att_header;
285 igtl_polydata_attribute * att;
286
287 int total_name_length;
288 int name_length;
289 char name_buf[IGTL_POLY_MAX_ATTR_NAME_LEN+1];
290
291 unsigned int i;
292 int n;
293
294 if (byte_array == NULL || info == NULL || size == 0)
295 {
296 return 0;
297 }
298
299 /* POLYDATA header */
300 header = (igtl_polydata_header *) byte_array;
301 if (igtl_is_little_endian())
302 {
303 info->header.npoints = BYTE_SWAP_INT32(header->npoints);
304 info->header.nvertices = BYTE_SWAP_INT32(header->nvertices);
305 info->header.size_vertices = BYTE_SWAP_INT32(header->size_vertices);
306 info->header.nlines = BYTE_SWAP_INT32(header->nlines);
307 info->header.size_lines = BYTE_SWAP_INT32(header->size_lines);
308 info->header.npolygons = BYTE_SWAP_INT32(header->npolygons);
309 info->header.size_polygons = BYTE_SWAP_INT32(header->size_polygons);
310 info->header.ntriangle_strips = BYTE_SWAP_INT32(header->ntriangle_strips);
311 info->header.size_triangle_strips = BYTE_SWAP_INT32(header->size_triangle_strips);
312 info->header.nattributes = BYTE_SWAP_INT32(header->nattributes);
313 }
314 else
315 {
316 memcpy(&(info->header), header, sizeof(igtl_polydata_header));
317 }
318
319 /* Allocate memory to read data */
320 /* TODO: compare the size of info before copying the header. */
321 /* If the size doesn't change, avoid reallocation of memory. */
322 if (igtl_polydata_alloc_info(info) == 0)
323 {
324 return 0;
325 }
326
327 /* POINT section */
328 ptr = (char*) byte_array + sizeof(igtl_polydata_header);
329 if (!igtl_is_little_endian())
330 {
331 memcpy(info->points, ptr, sizeof(igtl_float32)*info->header.npoints*3);
332 }
333 else
334 {
335 ptr32_src = (igtl_uint32 *) ptr;
336 ptr32_src_end = ptr32_src + info->header.npoints*3;
337 ptr32_dst = (igtl_uint32 *) info->points;
338 while (ptr32_src < ptr32_src_end)
339 {
340 *ptr32_dst = BYTE_SWAP_INT32(*ptr32_src);
341 ptr32_dst ++;
342 ptr32_src ++;
343 }
344 }
345
346 ptr += sizeof(igtl_float32)*info->header.npoints*3;
347
348 /* Check size parameters */
349 if (info->header.size_vertices%sizeof(igtl_uint32) != 0 ||
350 info->header.size_lines%sizeof(igtl_uint32) != 0 ||
351 info->header.size_polygons%sizeof(igtl_uint32) != 0 ||
352 info->header.size_triangle_strips%sizeof(igtl_uint32))
353 {
354 /* More than one of size parameters is multiples of 4 (size of 32-bit value) */
355 return 0;
356 }
357
358 /* VERTICES section */
359 igtl_polydata_convert_byteorder_topology(info->vertices, (igtl_uint32*)ptr, info->header.size_vertices);
360 ptr += info->header.size_vertices;
361
362 /* LINES section */
363 igtl_polydata_convert_byteorder_topology(info->lines, (igtl_uint32*)ptr, info->header.size_lines);
364 ptr += info->header.size_lines;
365
366 /* POLYGONS section */
367 igtl_polydata_convert_byteorder_topology(info->polygons, (igtl_uint32*)ptr, info->header.size_polygons);
368 ptr += info->header.size_polygons;
369
370 /* TRIANGLE_STRIPS section */
371 igtl_polydata_convert_byteorder_topology(info->triangle_strips, (igtl_uint32*)ptr, info->header.size_triangle_strips);
372 ptr += info->header.size_triangle_strips;
373
374 /* Attribute header */
375 for (i = 0; i < info->header.nattributes; i ++)
376 {
377 att = &(info->attributes[i]);
378 att_header = (igtl_polydata_attribute_header *) ptr;
379 att->type = att_header->type;
380 att->ncomponents = att_header->ncomponents;
381 if (igtl_is_little_endian())
382 {
383 att->n = BYTE_SWAP_INT32(att_header->n);
384 }
385 else
386 {
387 att->n = att_header->n;
388 }
389 ptr += sizeof(igtl_polydata_attribute_header);
390 }
391
392 /* Attribute names */
393 total_name_length = 0;
394 name_buf[IGTL_POLY_MAX_ATTR_NAME_LEN] = '\0';
395 for (i = 0; i < info->header.nattributes; i ++)
396 {
397 name_length = strlen(ptr);
398 if (name_length <= IGTL_POLY_MAX_ATTR_NAME_LEN)
399 {
400 info->attributes[i].name = malloc(name_length+1);
401 strcpy(info->attributes[i].name, ptr);
402 }
403 else
404 {
405 /* invalid name length */
406 return 0;
407 }
408 total_name_length += (name_length+1);
409 ptr += (name_length+1);
410 }
411
412 if (total_name_length % 2 > 0)
413 {
414 /* add padding */
415 ptr ++;
416 }
417
418 /* Attributes */
419 for (i = 0; i < info->header.nattributes; i ++)
420 {
421 if (info->attributes[i].type == IGTL_POLY_ATTR_TYPE_SCALAR)
422 {
423 n = info->attributes[i].ncomponents * info->attributes[i].n;
424 s = n * sizeof(igtl_float32);
425 }
426 else if (info->attributes[i].type == IGTL_POLY_ATTR_TYPE_NORMAL)
427 {
428 n = 3 * info->attributes[i].n;
429 s = n * sizeof(igtl_float32);
430 }
431 else if (info->attributes[i].type == IGTL_POLY_ATTR_TYPE_VECTOR)
432 {
433 n = 3 * info->attributes[i].n;
434 s = n * sizeof(igtl_float32);
435 }
436 else /* TENSOR */
437 {
438 n = 9 * info->attributes[i].n;
439 s = n * sizeof(igtl_float32);
440 }
441 info->attributes[i].data = (igtl_float32*)malloc((size_t)s);
442 ptr32_dst = (igtl_uint32*)info->attributes[i].data;
443 ptr32_src = (igtl_uint32*)ptr;
444 ptr32_src_end = ptr32_src + n;
445 if (igtl_is_little_endian())
446 {
447 while (ptr32_src < ptr32_src_end)
448 {
449 *ptr32_dst = BYTE_SWAP_INT32(*ptr32_src);
450 ptr32_dst ++;
451 ptr32_src ++;
452 }
453 }
454 else
455 {
456 memcpy(ptr32_dst, ptr32_src, s);
457 }
458 ptr += s;
459 }
460
461 return 1;
462}

535 info->header.size_polygons%sizeof(igtl_uint32) != 0 ||

668igtl_uint64 igtl_export igtl_polydata_get_size(igtl_polydata_info * info, int type)
669{
670 igtl_uint64 data_size;
671 int name_len;
672 unsigned int i;
673 int n;
674 int size;
675
676 /* Header */
677 data_size = sizeof(igtl_polydata_header);
678 /* POINT section */
679 data_size += sizeof(igtl_float32)*info->header.npoints*3;
680 /* VERTICES */
681 data_size += info->header.size_vertices;
682 /* LINES */
683 data_size += info->header.size_lines;
684 /* POLYGONS */
685 data_size += info->header.size_polygons;
686 /* TRIANGLE_STRIPS */
687 data_size += info->header.size_triangle_strips;
688 /* Attribute header */
689 data_size += sizeof(igtl_polydata_attribute_header) * info->header.nattributes;
690 /* Attribute names */
691 for (i = 0; i < info->header.nattributes; i ++)
692 {
693 name_len = strlen(info->attributes[i].name);
694 if (name_len > IGTL_POLY_MAX_ATTR_NAME_LEN)
695 {
696 /* Invaid name length */
697 return 0;
698 }
699 data_size += name_len + 1; /* Including terminator */
700 }
701
702 /* Padding */
703 if (data_size % 2 > 0)
704 {
705 data_size ++;
706 }
707
708 /* Attributes */
709 for (i = 0; i < info->header.nattributes; i ++)
710 {
711 if (info->attributes[i].type == IGTL_POLY_ATTR_TYPE_SCALAR)
712 {
713 n = info->attributes[i].ncomponents * info->attributes[i].n;
714 size = n * sizeof(igtl_float32);
715 }
716 else if (info->attributes[i].type == IGTL_POLY_ATTR_TYPE_NORMAL)
717 {
718 n = 3 * info->attributes[i].n;
719 size = n * sizeof(igtl_float32);
720 }
721 else if (info->attributes[i].type == IGTL_POLY_ATTR_TYPE_VECTOR)
722 {
723 n = 3 * info->attributes[i].n;
724 size = n * sizeof(igtl_float32);
725 }
726 else /* TENSOR */
727 {
728 n = 9 * info->attributes[i].n;
729 size = n * sizeof(igtl_float32);
730 }
731 data_size += size;
732 }
733
734 return data_size;
735}
