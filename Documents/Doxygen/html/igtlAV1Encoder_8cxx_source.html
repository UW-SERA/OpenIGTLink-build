168int igtlAV1Encoder::InitializeEncoder()
169{
170 cfg.g_lag_in_frames = 0;
171 cfg.g_w = this->GetPicWidth();
172 cfg.g_h = this->GetPicHeight();
173 aom_img_alloc(inputImage, AOM_IMG_FMT_I420, cfg.g_w,
174 cfg.g_h, 1);
175 if (aom_codec_enc_init(codec, encoder->codec_interface(), &cfg, 0))
176 {
177 error_output(codec, "Failed to initialize encoder");
178 return -1;
179 }
180
181 if (aom_codec_control_(codec, AV1E_SET_LOSSLESS, this->GetLosslessLink()))
182 {
183 error_output(codec, "Failed to set lossless mode");
184 return -1;
185 }
186
187 if (codecSpeed >= SlowestSpeed && codecSpeed <= FastestSpeed)
188 {
189 if (aom_codec_control(codec, AOME_SET_CPUUSED, codecSpeed))
190 {
191 error_output(codec, "Failed to set Speed");
192 return -1;
193 }
194 }
195 this->initializationDone = true;
196 return 0;
197}

214int igtlAV1Encoder::ConvertToLocalImageFormat(SourcePicture* pSrcPic)
215{
216 if (pSrcPic->picWidth != this->cfg.g_w || pSrcPic->picHeight != this->cfg.g_h)
217 {
218 this->SetPicWidthAndHeight(pSrcPic->picWidth,pSrcPic->picHeight);
219 this->InitializeEncoder();
220 }
221 int plane;
222 for (plane = 0; plane < 3; ++plane)
223 {
224 unsigned char *buf = inputImage->planes[plane];
225 const int w = aom_img_plane_width(inputImage, plane) *
226 ((inputImage->fmt & AOM_IMG_FMT_HIGHBITDEPTH) ? 2 : 1);
227 const int h = aom_img_plane_height(inputImage, plane);
228 memcpy(buf, pSrcPic->data[plane], w*h);
229 }
230
231 return 1;
232}

234int igtlAV1Encoder::EncodeSingleFrameIntoVideoMSG(SourcePicture* pSrcPic, igtl::VideoMessage* videoMessage, bool isGrayImage)
235{
236 int iSourceWidth = pSrcPic->picWidth;
237 int iSourceHeight = pSrcPic->picHeight;
238 if (iSourceWidth != this->cfg.g_w || iSourceHeight != this->cfg.g_h)
239 {
240 this->SetPicWidthAndHeight(iSourceWidth,iSourceHeight);
241 }
242 if (this->initializationDone == true)
243 {
244 static igtl_uint32 messageID = 6;
245 messageID ++;
246 this->ConvertToLocalImageFormat(pSrcPic);
247 const aom_codec_err_t res2 = aom_codec_encode(codec, inputImage, messageID, 1, 0);
248 if (res2 != AOM_CODEC_OK)
249 {
250 error_output(codec, "Failed to encode frame");
251 return -1;
252 }
253 iter = NULL;
254 while ((pkt = aom_codec_get_cx_data(codec, &iter)) != NULL)
255 {
256 if (pkt->kind == AOM_CODEC_CX_FRAME_PKT)
257 {
258 if((pkt->data.frame.flags & AOM_FRAME_IS_KEY) != 0)
259 {
260 encodedFrameType = FrameTypeKey;
261 }
262 else
263 {
264 // To do, assign other frame type too.
265 encodedFrameType = FrameTypeUnKnown;
266 }
267 videoMessage->SetBitStreamSize(pkt->data.frame.sz);
268 videoMessage->AllocateScalars();
269 videoMessage->SetCodecType(IGTL_VIDEO_CODEC_NAME_AV1);
270 videoMessage->SetEndian(igtl_is_little_endian() == true ? 2 : 1); //little endian is 2 big endian is 1
271 videoMessage->SetWidth(pSrcPic->picWidth);
272 videoMessage->SetHeight(pSrcPic->picHeight);
273 if (isGrayImage)
274 {
275 encodedFrameType = encodedFrameType << 8;
276 }
277 videoMessage->SetFrameType(encodedFrameType);
278 memcpy(videoMessage->GetPackFragmentPointer(2), pkt->data.frame.buf, pkt->data.frame.sz);
279 videoMessage->Pack();
280 }
281 }
282 if (videoMessage->GetBitStreamSize())
283 {
284 return 0;
285 }
286 return -1;
287 }
288 return -1;
289}
