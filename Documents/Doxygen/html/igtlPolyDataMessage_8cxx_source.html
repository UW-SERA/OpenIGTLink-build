267int PolyDataAttribute::SetType(int t, int n)
268{
269 int valid = 0;
270
271 switch(t)
272 {
273 case POINT_SCALAR:
274 case CELL_SCALAR:
275 if (n > 0 && n < 128)
276 {
277 valid = 1;
278 this->m_NComponents = n;
279 }
280 break;
281 case POINT_TCOORDS:
282 case CELL_TCOORDS:
283 case POINT_VECTOR:
284 case CELL_VECTOR:
285 case POINT_NORMAL:
286 case CELL_NORMAL:
287 valid = 1;
288 this->m_NComponents = 3;
289 break;
290 case POINT_TENSOR:
291 case CELL_TENSOR:
292 valid = 1;
293 this->m_NComponents = 9;
294 break;
295 case POINT_RGBA:
296 case CELL_RGBA:
297 valid = 1;
298 this->m_NComponents = 4;
299 break;
300 default:
301 break;
302 }
303 if (valid)
304 {
305 this->m_Type = t;
306 unsigned int n = this->m_Size * this->m_NComponents;
307 if (n != this->m_Data.size())
308 {
309 // TODO: this may cause unnecessary memory allocation,
310 // unless m_Size == 0.
311 // Memory should be reallocate just before use.
312 this->m_Data.resize(n);
313 }
314 return t;
315 }
316 else
317 {
318 return -1;
319 }
320}

590void IGTLCommon_EXPORT SetPolyDataInfoAttribute(igtl_polydata_info * info, PolyDataMessage * pdm)
591{
592
593 igtl_polydata_attribute * attr = info->attributes;
594 for (unsigned int i = 0; i < info->header.nattributes; i ++)
595 {
596 PolyDataAttribute * src = pdm->GetAttribute(static_cast<PolyDataMessage::AttributeList::size_type>(i));
597 if (src)
598 {
599 attr->type = src->GetType();
600 attr->ncomponents = src->GetNumberOfComponents();
601 attr->n = src->GetSize();
602 //attr->name = const_cast<char*>(src->GetName());
603 // TODO: aloways allocating memory isn't a good approach...
604 if (attr->name)
605 {
606 free(attr->name);
607 }
608 attr->name = (char *) malloc(strlen(src->GetName())+1);
609 if (attr->name)
610 {
611 strcpy(attr->name, src->GetName());
612 }
613 if (attr->data)
614 {
615 free(attr->data);
616 }
617 igtlUint32 size = attr->ncomponents * attr->n;
618 attr->data = (igtlFloat32*)malloc((size_t)size*sizeof(igtlFloat32));
619 if (attr->data)
620 {
621 src->GetData(attr->data);
622 }
623 attr ++;
624 }
625 }

653igtlUint64 PolyDataMessage::CalculateContentBufferSize()
654{
655 // TODO: The current implementation of GetBodyPackSize() allocates
656 // igtl_polydata_info and the array of igtl_polydata_attribute to calculate
657 // the size of pack. However, this approach is not efficient because
658 // it causes unnecessary memory allocation.
659
660 int dataSize;
661 igtl_polydata_info info;
662
663 SetPolyDataInfo(&info, this);
664
665 // Instead of calling igtl_polydata_alloc_info(), we only allocate
666 // memory for the attribute array, since igtl_polydata_alloc_info()
667 // allocates also allocates the memory area for actual points and
668 // cell data, which is not necessary to calculate polydata size.
669
670 info.attributes = new igtl_polydata_attribute[info.header.nattributes];
671
672 if (!info.attributes)
673 {
674 //ERROR
675 return 0;
676 }
677
678 // init attributes
679 igtl_polydata_attribute * attr = info.attributes;
680 for (unsigned int i = 0; i < info.header.nattributes; i ++)
681 {
682 attr->type = 0;
683 attr->ncomponents = 0;
684 attr->n = 0;
685 attr->name = NULL;
686 attr->data = NULL;
687 attr ++;
688 }
689
690 SetPolyDataInfoAttribute(&info, this);
691
692 dataSize = igtl_polydata_get_size(&info, IGTL_TYPE_PREFIX_NONE);
693
694 UnSetPolyDataInfoAttribute(&info);
695
696 //delete [] (info.attributes);
697 delete [] info.attributes;
698
699 return dataSize;
700}

703int PolyDataMessage::PackContent()
704{
705 // Allocate buffer
706 AllocateBuffer();
707
708 igtl_polydata_info info;
709
710 SetPolyDataInfo(&info, this);
711
712 if (igtl_polydata_alloc_info(&info) == 0)
713 {
714 return 0;
715 }
716
717 //SetPolyDataInfoAttribute(&info, this);
718
719 // Points
720 if (info.points)
721 {
722 igtlFloat32 * ptr_f = info.points;
723 for (int i = 0; i < this->m_Points->GetNumberOfPoints(); i ++)
724 {
725 igtlFloat32 points[3];
726 this->m_Points->GetPoint(i, points);
727 *(ptr_f++) = points[0];
728 *(ptr_f++) = points[1];
729 *(ptr_f++) = points[2];
730 }
731 }
732
733 //Vertices
734 if (info.vertices)
735 {
736 igtlUint32 * ptr_i = info.vertices;
737 for (unsigned int i = 0; i < this->m_Vertices->GetNumberOfCells(); i ++)
738 {
739 *ptr_i = this->m_Vertices->GetCellSize(i);
740 ptr_i ++;
741 this->m_Vertices->GetCell(i, ptr_i);
742 ptr_i += this->m_Vertices->GetCellSize(i);
743 }
744 }
745
746 //Lines
747 if (info.lines)
748 {
749 igtlUint32 * ptr_i = info.lines;
750 for (unsigned int i = 0; i < this->m_Lines->GetNumberOfCells(); i ++)
751 {
752 *ptr_i = this->m_Lines->GetCellSize(i);
753 ptr_i ++;
754 this->m_Lines->GetCell(i, ptr_i);
755 ptr_i += this->m_Lines->GetCellSize(i);
756 }
757 }
758
759 //Polygons
760 if (info.polygons)
761 {
762 igtlUint32 * ptr_i = info.polygons;
763 for (unsigned int i = 0; i < this->m_Polygons->GetNumberOfCells(); i ++)
764 {
765 *ptr_i = this->m_Polygons->GetCellSize(i);
766 ptr_i ++;
767 this->m_Polygons->GetCell(i, ptr_i);
768 ptr_i += this->m_Polygons->GetCellSize(i);
769 }
770 }
771
772 //TriangleStrips
773 if (info.triangle_strips)
774 {
775 igtlUint32 * ptr_i = info.triangle_strips;
776 for (unsigned int i = 0; i < this->m_TriangleStrips->GetNumberOfCells(); i ++)
777 {
778 *ptr_i = this->m_TriangleStrips->GetCellSize(i);
779 ptr_i ++;
780 this->m_TriangleStrips->GetCell(i, ptr_i);
781 ptr_i += this->m_TriangleStrips->GetCellSize(i);
782 }
783 }
784
785 SetPolyDataInfoAttribute(&info, this);
786
787 igtl_polydata_pack(&info, this->m_Content, IGTL_TYPE_PREFIX_NONE);
788
789 igtl_polydata_free_info(&info);
790
791 return 1;
792}

795int PolyDataMessage::UnpackContent()
796{
797 igtl_polydata_info info;
798
799 igtl_polydata_init_info(&info);
800
801 int r = 0;
802
803 bool isUnpacked(true);
804 r = igtl_polydata_unpack(IGTL_TYPE_PREFIX_NONE, (void*)this->m_Content, &info, this->CalculateReceiveContentSize(isUnpacked));
805
806 if ( r == 0)
807 {
808 return 0;
809 }
810
811 // Points
812 if (this->m_Points.IsNull())
813 {
814 this->m_Points = igtl::PolyDataPointArray::New();
815 }
816 this->m_Points->Clear();
817 if (info.header.npoints > 0)
818 {
819 this->m_Points->SetNumberOfPoints(info.header.npoints);
820 for (unsigned int i = 0; i < info.header.npoints; i ++)
821 {
822 this->m_Points->SetPoint(i, &(info.points[i*3]));
823 }
824 }
825
826 igtlUint32 * ptr;
827
828 // Vertices
829 if (this->m_Vertices.IsNull())
830 {
831 this->m_Vertices = igtl::PolyDataCellArray::New();
832 }
833 this->m_Vertices->Clear();
834 ptr = info.vertices;
835 for (unsigned int i = 0; i < info.header.nvertices; i ++)
836 {
837 unsigned int n = *ptr;
838 ptr ++;
839 this->m_Vertices->AddCell(n, ptr);
840 ptr += n;
841 }
842
843 // Lines
844 if (this->m_Lines.IsNull())
845 {
846 this->m_Lines = igtl::PolyDataCellArray::New();
847 }
848 this->m_Lines->Clear();
849 ptr = info.lines;
850 for (unsigned int i = 0; i < info.header.nlines; i ++)
851 {
852 unsigned int n = *ptr;
853 ptr ++;
854 this->m_Lines->AddCell(n, ptr);
855 ptr += n;
856 }
857
858 // Polygons
859 if (this->m_Polygons.IsNull())
860 {
861 this->m_Polygons = igtl::PolyDataCellArray::New();
862 }
863 this->m_Polygons->Clear();
864 ptr = info.polygons;
865 for (unsigned int i = 0; i < info.header.npolygons; i ++)
866 {
867 unsigned int n = *ptr;
868 ptr ++;
869 this->m_Polygons->AddCell(n, ptr);
870 ptr += n;
871 }
872
873 // TriangleStrips
874 if (this->m_TriangleStrips.IsNull())
875 {
876 this->m_TriangleStrips = igtl::PolyDataCellArray::New();
877 }
878 this->m_TriangleStrips->Clear();
879 ptr = info.triangle_strips;

885 ptr += n;
886 }
887

889 this->m_Attributes.clear();
890 igtl_polydata_attribute * attr = info.attributes;
891 for (unsigned int i = 0; i < info.header.nattributes; i ++)
892 {
893
894

896 if (pda.IsNotNull())

899 pda->SetType(attr->type, attr->ncomponents);
900 pda->SetSize(attr->n);
901 pda->SetName(attr->name);
902 pda->SetData(attr->data);
903 attr ++;
904 this->m_Attributes.push_back(pda);
905 }
906 }
907
908
909 return 1;
