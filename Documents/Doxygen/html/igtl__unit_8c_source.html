36igtl_unit igtl_export igtl_unit_pack(igtl_unit_data* data)
37{
38 igtl_unit pack;
39 igtl_uint8 exp;
40 int i;
41
42 pack = 0x0;
43
44 /* Prefix */
45 pack |= ((igtl_uint64) data->prefix) << 60;
46
47 /* Units */
48 for (i = 0; i < 6; i ++)
49 {
50 /* Check if exp is within the valid range */
51 if (data->exp[i] < -7 || data->exp[i] > 7)
52 {
53 return 0;
54 }
55 /* Convert signed value from 8-bit to 4-bit */
56 exp = data->exp[i] & 0x0F;
57
58 /* put into the pack */
59 pack |= ((igtl_uint64)data->unit[i]) << (10*(5-i) + 4);
60 pack |= ((igtl_uint64)exp) << (10*(5-i));
61 }
62
63 return pack;
64}

67int igtl_export igtl_unit_unpack(igtl_unit pack, igtl_unit_data* data)
68{
69 int i;
70
71 /* Prefix */
72 data->prefix = (igtl_uint8) (pack >> 60);
73
74 /* Units */
75 for (i = 0; i < 6; i ++)
76 {
77 data->unit[i] = (igtl_uint8) (pack >> (10*(5-i) + 4)) & 0x3F;
78 data->exp[i] = (igtl_uint8) (pack >> (10*(5-i))) & 0x0F;
79 /* Convert signed value in exponent field from 4-bit to 8-bit */
80 if (data->exp[i] & 0x08)
81 {
82 data->exp[i] |= 0xF0;
83 }
84 }
85 return 1;
86}
