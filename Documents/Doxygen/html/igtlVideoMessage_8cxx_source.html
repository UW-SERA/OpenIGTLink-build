211 int VideoMessage::PackContent()
212 {
213 igtl_frame_header* frame_header = (igtl_frame_header*)this->m_FrameHeader;
214
215 frame_header->header_version = IGTL_VIDEO_HEADER_VERSION;
216 strncpy(frame_header->encoding, this->m_CodecType.c_str(), IGTL_VIDEO_CODEC_NAME_SIZE);
217 frame_header->endian = this->endian;
218 frame_header->frameType = this->videoFrameType;
219 frame_header->coord = this->coordinate;
220
221 float origin[3];
222 float norm_i[3];
223 float norm_j[3];
224 float norm_k[3];
225
226 for (int i = 0; i < 3; i++)
227 {
228 norm_i[i] = matrix[i][0];
229 norm_j[i] = matrix[i][1];
230 norm_k[i] = matrix[i][2];
231 origin[i] = matrix[i][3];
232
233 frame_header->size[i] = this->dimensions[i];
234 frame_header->subvol_size[i] = this->subDimensions[i];
235 frame_header->subvol_offset[i] = this->subOffset[i];
236 }
237
238 igtl_frame_set_matrix(this->spacing, origin,
239 norm_i, norm_j, norm_k,
240 frame_header);
241
242 igtl_frame_convert_byte_order(frame_header);
243
244 #if OpenIGTLink_HEADER_VERSION >= 2
245 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
246 {
247 m_MetaDataHeader = &m_Body[CalculateContentBufferSize()+IGTL_EXTENDED_HEADER_SIZE];
248 }
249 #endif
250 return 1;
251 }

254 int VideoMessage::UnpackContent()
255 {
256 #if OpenIGTLink_HEADER_VERSION >= 2
257 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
258 {
259 this->m_FrameHeader = &m_Body[IGTL_EXTENDED_HEADER_SIZE];
260 }
261 else
262 {
263 this->m_FrameHeader = m_Body;
264 }
265 this->m_Frame = &m_FrameHeader[IGTL_VIDEO_HEADER_SIZE];
266 #else
267 this->m_FrameHeader = m_Body;
268 this->m_Frame = m_Body;
269 #endif
270
271 igtl_frame_header* frame_header = (igtl_frame_header*)m_FrameHeader;
272 igtl_frame_convert_byte_order(frame_header);
273
274 if (frame_header->header_version == IGTL_VIDEO_HEADER_VERSION)
275 {
276 // Video format version 1
277 this->m_CodecType.clear();
278 this->m_CodecType.append(frame_header->encoding, IGTL_VIDEO_CODEC_NAME_SIZE);
279 this->endian = frame_header->endian;
280 this->videoFrameType = frame_header->frameType;
281
282 // Set image orientation
283 float rspacing[3];
284 float origin[3];
285 float norm_i[3];
286 float norm_j[3];
287 float norm_k[3];
288
289 igtl_frame_get_matrix(rspacing, origin,
290 norm_i, norm_j, norm_k,
291 frame_header);
292
293 for (int i = 0; i < 3; i++)
294 {
295 this->spacing[i] = rspacing[i];
296 matrix[i][0] = norm_i[i];
297 matrix[i][1] = norm_j[i];
298 matrix[i][2] = norm_k[i];
299 matrix[i][3] = origin[i];
300
301 this->dimensions[i] = frame_header->size[i];
302 this->subDimensions[i] = frame_header->subvol_size[i];
303 this->subOffset[i] = frame_header->subvol_offset[i];
304 }
305
306 matrix[3][0] = 0.0;
307 matrix[3][1] = 0.0;
308 matrix[3][2] = 0.0;
309 matrix[3][3] = 1.0;
310
311 this->m_Frame = this->m_FrameHeader;
312 this->m_Frame += IGTL_VIDEO_HEADER_SIZE;
313
314 return 1;
315 }
316 else
317 {
318 // Incompatible version.
319 return 0;
320 }
321 }
