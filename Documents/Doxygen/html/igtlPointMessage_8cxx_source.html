213int PointMessage::PackContent()
214{
215 // Allocate buffer
216 AllocateBuffer();
217
218 igtl_point_element* element;
219 element = (igtl_point_element*)(this->m_Content);
220
221 igtl_point_element* elementHolder = element;
222 std::vector<PointElement::Pointer>::iterator iter;
223 for (iter = this->m_PointList.begin(); iter != this->m_PointList.end(); iter ++)
224 {
225 strncpy((char*)element->name, (*iter)->GetName(), IGTL_POINT_LEN_NAME);
226 strncpy((char*)element->group_name, (*iter)->GetGroupName(), IGTL_POINT_LEN_GROUP_NAME);
227
228 igtlUint8 rgba[4];
229 (*iter)->GetRGBA(rgba);
230 element->rgba[0] = rgba[0];
231 element->rgba[1] = rgba[1];
232 element->rgba[2] = rgba[2];
233 element->rgba[3] = rgba[3];
234
235 igtlFloat32 position[3];
236 (*iter)->GetPosition(position);
237 element->position[0] = position[0];
238 element->position[1] = position[1];
239 element->position[2] = position[2];
240
241 element->radius = (*iter)->GetRadius();
242
243 strncpy((char*)element->owner, (*iter)->GetOwner(), IGTL_POINT_LEN_OWNER);
244
245 element ++;
246 }
247
248 igtl_point_convert_byte_order(elementHolder, this->m_PointList.size());
249
250 return 1;
251}

254int PointMessage::UnpackContent()
255{
256 this->m_PointList.clear();
257 igtl_point_element* element = NULL;
258 int nElement = 0;
259#if OpenIGTLink_HEADER_VERSION >= 2
260 element = (igtl_point_element*) (this->m_Content);
261 bool isUnpacked(true);
262 nElement = igtl_point_get_data_n(CalculateReceiveContentSize(isUnpacked));
263#elif OpenIGTLink_PROTOCOL_VERSION <=2
264 element = (igtl_point_element*) this->m_Body;
265 nElement = igtl_point_get_data_n(this->m_BodySizeToRead);
266#endif
267
268 igtl_point_convert_byte_order(element, nElement);
269
270 char strbuf[128];
271 for (int i = 0; i < nElement; i ++)
272 {
273 PointElement::Pointer elemClass = PointElement::New();
274
275 // Add '\n' at the end of each string
276 // (necessary for a case, where a string reaches the maximum length.)
277 strbuf[IGTL_POINT_LEN_NAME] = '\n';
278 strncpy(strbuf, (char*)element->name, IGTL_POINT_LEN_NAME);
279 elemClass->SetName((const char*)strbuf);
280
281 strbuf[IGTL_POINT_LEN_GROUP_NAME] = '\n';
282 strncpy(strbuf, (char*)element->group_name, IGTL_POINT_LEN_GROUP_NAME);
283 elemClass->SetGroupName(strbuf);
284
285 elemClass->SetRGBA(element->rgba);
286 elemClass->SetPosition(element->position);
287 elemClass->SetRadius(element->radius);
288
289 strbuf[IGTL_POINT_LEN_OWNER] = '\n';
290 strncpy(strbuf, (char*)element->owner, IGTL_POINT_LEN_OWNER);
291 elemClass->SetOwner(strbuf);
292
293 this->m_PointList.push_back(elemClass);
294
295 element ++;
296 }
297
298 return 1;
299}
