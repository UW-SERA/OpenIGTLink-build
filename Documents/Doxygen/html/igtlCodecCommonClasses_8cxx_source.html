51long ReadConfigFile::ReadLine (std::string* var, const int valSize/* = 4*/) {
52 if (configFile == NULL || var == NULL || valSize <= 1)
53 return 0;
54
55 std::string* strTags = &var[0];
56 int nTagNum = 0, n = 0;
57 bool bCommentFlag = false;
58
59 while (n < valSize) {
60 var[n] = "";
61 ++ n;
62 }
63 do {
64 const char singleChar = (char)fgetc (configFile);
65
66 if (singleChar == '\n' || feof (configFile)) {
67 ++ lineNums;
68 break;
69 }
70 if (singleChar == '#')
71 bCommentFlag = true;
72 if (!bCommentFlag) {
73 if (singleChar == '\t' || singleChar == ' ') {
74 if (nTagNum >= valSize-1)
75 break;
76 if (! (*strTags).empty()) {
77 ++ nTagNum;
78 strTags = &var[nTagNum];
79 }
80 } else
81 *strTags += singleChar;
82 }
83
84 } while (true);
85
86 return 1 + nTagNum;
87}

134int GenericDecoder::UnpackUncompressedData(igtl::VideoMessage* videoMessage, SourcePicture* decodedPic)
135{
136 igtl_int32 iWidth = videoMessage->GetWidth();
137 igtl_int32 iHeight = videoMessage->GetHeight();
138 if ((iWidth*iHeight*3>>1) != videoMessage->GetBitStreamSize())
139 {
140 return -1;
141 }
142 igtl_uint16 frameType = videoMessage->GetFrameType();
143 isGrayImage = false;
144 if (frameType>0X00FF)
145 {
146 frameType= frameType>>8;
147 isGrayImage = true;
148 }
149 decodedPic->picWidth = iWidth;
150 decodedPic->picHeight = iHeight;
151 decodedPic->data[1]= decodedPic->data[0] + iWidth*iHeight;
152 decodedPic->data[2]= decodedPic->data[1] + iWidth*iHeight/4;
153 decodedPic->stride[0] = iWidth;
154 decodedPic->stride[1] = decodedPic->stride[2] = iWidth>>1;
155 decodedPic->stride[3] = 0;
156 memcpy(decodedPic->data[0], videoMessage->GetPackFragmentPointer(2), iWidth*iHeight*3>>1);
157 return 1;
158}

173int GenericDecoder::ConvertYUVToRGB(igtl_uint8 *YUVFrame, igtl_uint8* RGBFrame, int iHeight, int iWidth)
174{
175 int componentLength = iHeight*iWidth;
176 const igtl_uint8 *srcY = YUVFrame;
177 const igtl_uint8 *srcU = YUVFrame+componentLength;
178 const igtl_uint8 *srcV = YUVFrame+componentLength*5/4;
179 igtl_uint8 * YUV444 = new igtl_uint8[componentLength * 3];
180 igtl_uint8 *dstY = YUV444;
181 igtl_uint8 *dstU = dstY + componentLength;
182 igtl_uint8 *dstV = dstU + componentLength;
183
184 memcpy(dstY, srcY, componentLength);
185 const int halfHeight = iHeight/2;
186 const int halfWidth = iWidth/2;
187
188#pragma omp parallel for default(none) shared(dstV,dstU,srcV,srcU,iWidth)
189 for (int y = 0; y < halfHeight; y++)
190 {
191 for (int x = 0; x < halfWidth; x++)
192 {
193 dstU[2 * x + 2 * y*iWidth] = dstU[2 * x + 1 + 2 * y*iWidth] = srcU[x + y*iWidth/2];
194 dstV[2 * x + 2 * y*iWidth] = dstV[2 * x + 1 + 2 * y*iWidth] = srcV[x + y*iWidth/2];
195 }
196 memcpy(&dstU[(2 * y + 1)*iWidth], &dstU[(2 * y)*iWidth], iWidth);
197 memcpy(&dstV[(2 * y + 1)*iWidth], &dstV[(2 * y)*iWidth], iWidth);
198 }
199
200
201 const int yOffset = 16;
202 const int cZero = 128;
203 int yMult, rvMult, guMult, gvMult, buMult;
204 yMult = 298;
205 rvMult = 409;
206 guMult = -100;
207 gvMult = -208;
208 buMult = 517;
209
210 static unsigned char clp_buf[384+256+384];
211 static unsigned char *clip_buf = clp_buf+384;
212
213 // initialize clipping table
214 memset(clp_buf, 0, 384);
215
216 for (int i = 0; i < 256; i++)
217 {
218 clp_buf[384+i] = i;
219 }
220 memset(clp_buf+384+256, 255, 384);
221
222
223#pragma omp parallel for default(none) shared(dstY,dstU,dstV,RGBFrame,yMult,rvMult,guMult,gvMult,buMult,clip_buf,componentLength)// num_threads(2)
224 for (int i = 0; i < componentLength; ++i)
225 {
226 const int Y_tmp = ((int)dstY[i] - yOffset) * yMult;
227 const int U_tmp = (int)dstU[i] - cZero;
228 const int V_tmp = (int)dstV[i] - cZero;
229
230 const int R_tmp = (Y_tmp + V_tmp * rvMult ) >> 8;//32 to 16 bit conversion by left shifting
231 const int G_tmp = (Y_tmp + U_tmp * guMult + V_tmp * gvMult ) >> 8;
232 const int B_tmp = (Y_tmp + U_tmp * buMult ) >> 8;
233
234 RGBFrame[3*i] = clip_buf[R_tmp];
235 RGBFrame[3*i+1] = clip_buf[G_tmp];
236 RGBFrame[3*i+2] = clip_buf[B_tmp];
237 }
238 delete [] YUV444;
239 YUV444 = NULL;
240 dstY = NULL;
241 dstU = NULL;
242 dstV = NULL;
243 return 1;
244}

257void GenericEncoder::ConvertRGBToYUV(igtlUint8 *rgb, igtlUint8 *destination, unsigned int width, unsigned int height)
258{
259 size_t image_size = width * height;
260 size_t upos = image_size;
261 size_t vpos = upos + upos / 4;
262 size_t i = 0;
263
264 for (size_t line = 0; line < height; ++line)
265 {
266 if (!(line % 2))
267 {
268 for (size_t x = 0; x < width; x += 2)
269 {
270 igtlUint8 r = rgb[3 * i];
271 igtlUint8 g = rgb[3 * i + 1];
272 igtlUint8 b = rgb[3 * i + 2];
273
274 destination[i++] = ((66 * r + 129 * g + 25 * b) >> 8) + 16;
275
276 destination[upos++] = ((-38 * r - 74 * g + 112 * b) >> 8) + 128;
277 destination[vpos++] = ((112 * r - 94 * g -18 * b) >> 8) + 128;
278
279 r = rgb[3 * i];
280 g = rgb[3 * i + 1];
281 b = rgb[3 * i + 2];
282
283 destination[i++] = ((66 * r + 129 * g + 25 * b) >> 8) + 16;
284 }
285 }
286 else
287 {
288 for (size_t x = 0; x < width; x += 1)
289 {
290 igtlUint8 r = rgb[3 * i];
291 igtlUint8 g = rgb[3 * i + 1];
292 igtlUint8 b = rgb[3 * i + 2];
293
294 destination[i++] = ((66 * r + 129 * g + 25 * b) >> 8) + 16;
295 }
296 }
297 }

300int GenericEncoder::PackUncompressedData(SourcePicture* pSrcPic, igtl::VideoMessage* videoMessage, bool isGrayImage)
301{
302 int iSourceWidth = pSrcPic->picWidth;
303 int iSourceHeight = pSrcPic->picHeight;
304 long kiPicResSize = iSourceWidth*iSourceHeight * 3 >> 1;
305 videoMessage->SetBitStreamSize(kiPicResSize);
306 videoMessage->AllocateScalars();
307 int endian = (igtl_is_little_endian() == 1 ? IGTL_VIDEO_ENDIAN_LITTLE : IGTL_VIDEO_ENDIAN_BIG);
308 videoMessage->SetEndian(endian); //little endian is 2 big endian is 1
309 videoMessage->SetWidth(pSrcPic->picWidth);
310 videoMessage->SetHeight(pSrcPic->picHeight);
311 encodedFrameType = FrameTypeKey;
312 if (isGrayImage)
313 {
314 encodedFrameType = FrameTypeKey << 8;
315 }
316 videoMessage->SetFrameType(encodedFrameType);
317 memcpy(videoMessage->GetPackFragmentPointer(2), pSrcPic->data[0], kiPicResSize);
318 videoMessage->Pack();
319 return 0;
320}
