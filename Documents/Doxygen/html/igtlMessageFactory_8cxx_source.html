155//-----------------------------------------------------------------------------
156igtl::MessageBase::Pointer MessageFactory::GetMessage(igtl::MessageHeader::Pointer headerMsg)
157{
158 if (headerMsg.IsNull())
159 {
160 return NULL;
161 }
162
163#if OpenIGTLink_HEADER_VERSION >= 2
164 std::string messageType(headerMsg->GetMessageType());
165#else
166 std::string messageType(headerMsg->GetDeviceType());
167#endif
168
169 if (!this->IsValid(headerMsg))
170 {
171 return NULL;
172 }
173
174 std::string messageTypeUpper(messageType);
175 std::transform(messageTypeUpper.begin(), messageTypeUpper.end(), messageTypeUpper.begin(), ::toupper);
176
177 igtl::MessageBase::Pointer result = NULL;
178 if( GetMessageTypeNewPointer(messageTypeUpper) == NULL )
179 {
180 return NULL;
181 }
182 result = GetMessageTypeNewPointer(messageTypeUpper)();
183
184 // Must have a valid message at this point, otherwise its a programming bug.
185 assert(result.IsNotNull());
186
187 result->SetMessageHeader(headerMsg);
188 result->InitBuffer();
189
190 return result;
191}
