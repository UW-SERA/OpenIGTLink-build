21igtl_uint64 igtl_export igtl_image_get_data_size(igtl_image_header * header)
22{
23 igtl_uint64 si;
24 igtl_uint64 sj;
25 igtl_uint64 sk;
26 igtl_uint64 sp;
27 igtl_uint64 data_size;
28
29 si = header->subvol_size[0];
30 sj = header->subvol_size[1];
31 sk = header->subvol_size[2];
32
33 switch (header->scalar_type)
34 {
35 case IGTL_IMAGE_STYPE_TYPE_INT8:
36 case IGTL_IMAGE_STYPE_TYPE_UINT8:
37 sp = 1;
38 break;
39 case IGTL_IMAGE_STYPE_TYPE_INT16:
40 case IGTL_IMAGE_STYPE_TYPE_UINT16:
41 sp = 2;
42 break;
43 case IGTL_IMAGE_STYPE_TYPE_INT32:
44 case IGTL_IMAGE_STYPE_TYPE_UINT32:
45 sp = 4;
46 break;
47 default:
48 sp = 0;
49 break;
50 }
51
52 data_size = si*sj*sk*sp;
53 return data_size;
54}

91void igtl_export igtl_image_get_matrix(float spacing[3], float origin[3],
92 float norm_i[3], float norm_j[3], float norm_k[3],
93 igtl_image_header * header)
94{
95 float tx;
96 float ty;
97 float tz;
98 float sx;
99 float sy;
100 float sz;
101 float nx;
102 float ny;
103 float nz;
104 float px;
105 float py;
106 float pz;
107
108 tx = (float) header->matrix[0];
109 ty = (float) header->matrix[1];
110 tz = (float) header->matrix[2];
111 sx = (float) header->matrix[3];
112 sy = (float) header->matrix[4];
113 sz = (float) header->matrix[5];
114 nx = (float) header->matrix[6];
115 ny = (float) header->matrix[7];
116 nz = (float) header->matrix[8];
117 px = (float) header->matrix[9];
118 py = (float) header->matrix[10];
119 pz = (float) header->matrix[11];
120
121 spacing[0] = sqrtf(tx*tx + ty*ty + tz*tz);
122 spacing[1] = sqrtf(sx*sx + sy*sy + sz*sz);
123 spacing[2] = sqrtf(nx*nx + ny*ny + nz*nz);
124
125 if (spacing[0] != 0)
126 {
127 norm_i[0] = header->matrix[0] / spacing[0];
128 norm_i[1] = header->matrix[1] / spacing[0];
129 norm_i[2] = header->matrix[2] / spacing[0];
130 }
131 if (spacing[1] != 0)
132 {
133 norm_j[0] = header->matrix[3] / spacing[1];
134 norm_j[1] = header->matrix[4] / spacing[1];
135 norm_j[2] = header->matrix[5] / spacing[1];
136 }
137 if (spacing[2] != 0)
138 {
139 norm_k[0] = header->matrix[6] / spacing[2];
140 norm_k[1] = header->matrix[7] / spacing[2];
141 norm_k[2] = header->matrix[8] / spacing[2];
142 }
143
144 origin[0] = header->matrix[9];
145 origin[1] = header->matrix[10];
146 origin[2] = header->matrix[11];
147
148}

174void igtl_export igtl_image_convert_byte_order(igtl_image_header * header)
175{
176 int i;
177 igtl_uint32 tmp[12];
178
179 if (igtl_is_little_endian())
180 {
181
182 header->header_version = BYTE_SWAP_INT16(header->header_version);
183
184 for (i = 0; i < 3; i ++)
185 {
186 header->size[i] = BYTE_SWAP_INT16(header->size[i]);
187 header->subvol_size[i] = BYTE_SWAP_INT16(header->subvol_size[i]);
188 header->subvol_offset[i] = BYTE_SWAP_INT16(header->subvol_offset[i]);
189 }
190
191 memcpy(tmp, header->matrix, sizeof(igtl_uint32)*12);
192
193 /*
194 * TODO: The following loop may cause segmentation fault, when it is compiled
195 * with '-ftree-vectorize' optimization option on 64-bit Linux.
196 * ('-ftree-vectorize' becomes active, when '-O3' optimization is specified.)
197 * -- code has been updated on June 24 -- needs test
198 */
199 for (i = 0; i < 12; i ++)
200 {
201 tmp[i] = BYTE_SWAP_INT32(tmp[i]);
202 }
203 memcpy(header->matrix, tmp, sizeof(igtl_uint32)*12);
204
205 }
206}

209igtl_uint64 igtl_export igtl_image_get_crc(igtl_image_header * header, void* image)
210{
211 igtl_uint64 crc;
212 igtl_uint64 img_size;
213
214 /* calculate image size (we do not call igtl_image_get_data_size()
215 * because header has already been serialized.
216 */
217 igtl_uint64 si;
218 igtl_uint64 sj;
219 igtl_uint64 sk;
220 igtl_uint64 sp;
221
222 if (igtl_is_little_endian())
223 {
224 si = BYTE_SWAP_INT16(header->subvol_size[0]);
225 sj = BYTE_SWAP_INT16(header->subvol_size[1]);
226 sk = BYTE_SWAP_INT16(header->subvol_size[2]);
227 }
228 else
229 {
230 si = header->subvol_size[0];
231 sj = header->subvol_size[1];
232 sk = header->subvol_size[2];
233 }
234 switch (header->scalar_type) {
235 case IGTL_IMAGE_STYPE_TYPE_INT8:
236 case IGTL_IMAGE_STYPE_TYPE_UINT8:
237 sp = 1;
238 break;
239 case IGTL_IMAGE_STYPE_TYPE_INT16:
240 case IGTL_IMAGE_STYPE_TYPE_UINT16:
241 sp = 2;
242 break;
243 case IGTL_IMAGE_STYPE_TYPE_INT32:
244 case IGTL_IMAGE_STYPE_TYPE_UINT32:
245 sp = 4;
246 break;
247 default:
248 sp = 0;
249 break;
250 }
251
252 img_size = si*sj*sk*sp;
253 crc = crc64(0, 0, 0);
254 crc = crc64((unsigned char*) header, IGTL_IMAGE_HEADER_SIZE, crc);
255 crc = crc64((unsigned char*) image, (int)img_size, crc);
256
257 return crc;
258}
