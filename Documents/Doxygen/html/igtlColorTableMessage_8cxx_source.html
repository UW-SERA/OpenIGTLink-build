89int ColorTableMessage::UnpackContent()
90{
91
92 this->m_ColorTableHeader = this->m_Content;
93 this->m_ColorTable = &(this->m_Content[IGTL_COLORTABLE_HEADER_SIZE]);
94
95 igtl_colortable_header* colortable_header = (igtl_colortable_header*)this->m_ColorTableHeader;
96 igtl_colortable_convert_byte_order(colortable_header, (void*)this->m_ColorTable);
97
98 this->indexType = colortable_header->indexType;
99 this->mapType = colortable_header->mapType;
100
101 return 1;
