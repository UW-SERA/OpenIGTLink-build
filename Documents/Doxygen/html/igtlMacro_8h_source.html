638//----------------------------------------------------------------------------
639// Setup legacy code policy.
640//
641// CMake options IGTL_LEGACY_REMOVE and IGTL_LEGACY_SILENT are converted
642// to definitions (or non-defs) in igtlConfigure.h and tested below.
643// They may be used to completely remove legacy code or silence the
644// warnings. The default is to warn about their use.
645//
646// Source files that test the legacy code may define IGTL_LEGACY_TEST
647// like this:
648//
649// #define IGTL_LEGACY_TEST
650// #include "igtlClassWithDeprecatedMethod.h"
651//
652// in order to silence the warnings for calling deprecated methods.
653// No other source files in IGTL should call the methods since they are
654// provided only for compatibility with older user code.
655
656// Define igtlLegacyMacro to mark legacy methods where they are
657// declared in their class. Example usage:
658//
659// // @deprecated Replaced by MyOtherMethod() as of IGTL 2.0.
660// igtlLegacyMacro(void MyMethod());
661#if defined(IGTL_LEGACY_REMOVE)
662// Remove legacy methods completely. Put a bogus declaration in
663// place to avoid stray semicolons because this is an error for some
664// compilers. Using a class forward declaration allows any number
665// of repeats in any context without generating unique names.
666# define igtlLegacyMacro(method) class igtlLegacyMethodRemoved /* no ';' */
667#elif defined(IGTL_LEGACY_SILENT) || defined(IGTL_LEGACY_TEST) || defined(CSWIG)
668 // Provide legacy methods with no warnings.
669# define igtlLegacyMacro(method) method
670#else
671 // Setup compile-time warnings for uses of deprecated methods if
672 // possible on this compiler.
673# if defined(__GNUC__) && !defined(__INTEL_COMPILER) && (__GNUC__ > 3 || (__GNUC__ == 3 && __GNUC_MINOR__ >= 1))
674# define igtlLegacyMacro(method) method __attribute__((deprecated))
675# elif defined(_MSC_VER) && _MSC_VER >= 1300
676# define igtlLegacyMacro(method) __declspec(deprecated) method
677# else
678# define igtlLegacyMacro(method) method
679# endif
680#endif

735/* Define a macro to explicitly instantiate a template.
736 - IGTL_TEMPLATE_EXPORT(X) =
737 Explicitly instantiate X, where X is of the form N(a1[,a2...,aN]).
738 examples: IGTL_TEMPLATE_EXPORT(1(class Foo<int>))
739 IGTL_TEMPLATE_EXPORT(2(class Bar<int, char>))
740 Use one level of expansion delay to allow user code to have
741 a macro determining the number of arguments. */
742#define IGTL_TEMPLATE_EXPORT(x) IGTL_TEMPLATE_EXPORT_DELAY(x)
743#define IGTL_TEMPLATE_EXPORT_DELAY(x) template IGTL_TEMPLATE_##x;
744
745/* Define a macro to prevent template instantiations.
746 - IGTL_TEMPLATE_IMPORT(X) =
747 Prevent instantiation of X, where X is of the form N(a1[,a2...,aN]).
748 examples: IGTL_TEMPLATE_IMPORT(1(class Foo<int>))
749 IGTL_TEMPLATE_IMPORT(2(class Bar<int, char>))
750 Use one level of expansion delay to allow user code to have
751 a macro determining the number of arguments.
752*/
753#if IGTL_TEMPLATE_EXTERN
754# define IGTL_TEMPLATE_IMPORT_DELAY(x) extern template IGTL_TEMPLATE_##x;
755# if defined(_MSC_VER)
756# pragma warning (disable: 4231) /* extern template extension */
757# endif
758#elif IGTL_TEMPLATE_DO_NOT_INSTANTIATE
759# define IGTL_TEMPLATE_IMPORT_DELAY(x) \
760 IGTL_TEMPLATE_IMPORT_IMPL(do_not_instantiate IGTL_TEMPLATE_##x)
761# define IGTL_TEMPLATE_IMPORT_IMPL(x) _Pragma(#x)
762#endif
763#if defined(IGTL_TEMPLATE_IMPORT_DELAY)
764# define IGTL_TEMPLATE_IMPORT(x) IGTL_TEMPLATE_IMPORT_DELAY(x)
765# define IGTL_TEMPLATE_IMPORT_WORKS 1

771/* Define macros to export and import template instantiations. These
772 depend on each class providing a macro defining the instantiations
773 given template arguments in X. The argument X is of the form
774 N(a1[,a2...,aN]). The argument Y is a valid preprocessing token
775 unique to the template arguments given in X. Typical usage is
776
777 IGTL_EXPORT_TEMPLATE(igtlfoo_EXPORT, Foo, (int), I)
778 IGTL_EXPORT_TEMPLATE(igtlfoo_EXPORT, Bar, (int, char), IC)
779
780 The IGTL_TEMPLATE_<name> macro should be defined in igtl<name>.h and
781 is of the following form:
782
783 #define IGTL_TEMPLATE_<name>(_, EXPORT, x, y) namespace igtl { \
784 _(<n>(class EXPORT <name>< IGTL_TEMPLATE_<n> x >)) \
785 namespace Templates { typedef <name>< IGTL_TEMPLATE_<n> x > <name>##y; }\
786 }
787
788 The argument "_" will be replaced by another macro such as
789 IGTL_TEMPLATE_EXPORT or IGTL_TEMPLATE_IMPORT, so it should be used as
790 if calling one of these macros. The argument "EXPORT" will be
791 replaced by a dllexport/dllimport macro such as IGTLCommon_EXPORT.
792 The argument "x" is a paren-enclosed list of template arguments.
793 The argument "y" is a preprocessing token corresponding to the
794 given template arguments and should be used to construct typedef
795 names for the instantiations.
796
797 Note the use of IGTL_TEMPLATE_<n>, where <n> is the number of
798 template arguments for the class template. Note also that the
799 number of template arguments is usually the length of the list
800 nested within the inner parentheses, so the instantiation is listed
801 with the form <n>(...). Example definitions:
802
803 #define IGTL_TEMPLATE_Foo(_, EXPORT, x, y) namespace igtl { \
804 _(1(class EXPORT Foo< IGTL_TEMPLATE_1 x >)) \
805 _(1(EXPORT std::ostream& operator<<(std::ostream&, \
806 const Foo< IGTL_TEMPLATE_1 x >&))) \
807 namespace Templates { typedef Foo< IGTL_TEMPLATE_1 x > Foo##y; }\
808 }
809
810 #define IGTL_TEMPLATE_Bar(_, EXPORT, x, y) namespace igtl { \
811 _(2(class EXPORT Bar< IGTL_TEMPLATE_2 x >)) \
812 _(1(EXPORT std::ostream& operator<<(std::ostream&, \
813 const Bar< IGTL_TEMPLATE_2 x >&))) \
814 namespace Templates { typedef Bar< IGTL_TEMPLATE_2 x > Bar##y; }\
815 }
816
817 Note that in the stream operator for template Bar there is a "1" at
818 the beginning even though two arguments are taken. This is because
819 the expression "IGTL_TEMPLATE_2 x" is contained inside the
820 parentheses of the function signature which protects the resulting
821 comma from separating macro arguments. Therefore the nested
822 parentheses contain a list of only one macro argument.
823
824 The IGTL_EMPTY macro used in these definitions is a hack to work
825 around a VS 6.0 preprocessor bug when EXPORT is empty.
