305int VideoMetaMessage::PackContent()
306{
307 // Allocate buffer
308 AllocateBuffer();
309
310 igtl_videometa_element* element = (igtl_videometa_element*)this->m_Content;
311
312 std::vector<VideoMetaElement::Pointer>::iterator iter;
313 for (iter = this->m_VideoMetaList.begin(); iter != this->m_VideoMetaList.end(); iter ++)
314 {
315 strncpy((char*)element->name, (*iter)->GetName(), IGTL_VIDEOMETA_LEN_NAME);
316 strncpy((char*)element->device_name, (*iter)->GetDeviceName(), IGTL_VIDEOMETA_LEN_DEVICE_NAME);
317 strncpy((char*)element->patient_name, (*iter)->GetPatientName(), IGTL_VIDEOMETA_LEN_PATIENT_NAME);
318 strncpy((char*)element->patient_id, (*iter)->GetPatientID(), IGTL_VIDEOMETA_LEN_PATIENT_ID);
319
320 element->zoom_level = (*iter)->GetZoomLevel();
321 element->focal_length = (*iter)->GetFocalLength();
322
323 igtlUint16 size[3];
324 (*iter)->GetSize(size[0], size[1], size[2]);
325 element->size[0] = size[0];
326 element->size[1] = size[1];
327 element->size[2] = size[2];
328
329 float origin[3];
330 float norm_i[3];
331 float norm_j[3];
332 float norm_k[3];
333 float spacing[3];
334 Matrix4x4 mat;
335 (*iter)->GetMatrix(mat);
336 (*iter)->GetSpacing(spacing[0], spacing[1], spacing[2]);
337 for (int i = 0; i < 3; i ++)
338 {
339 norm_i[i] = mat[i][0];
340 norm_j[i] = mat[i][1];
341 norm_k[i] = mat[i][2];
342 origin[i] = mat[i][3];
343 }
344
345 igtl_videometa_set_matrix(spacing, origin,
346 norm_i, norm_j, norm_k,
347 element);
348
349 element->scalar_type = (*iter)->GetScalarType();
350 element->reserved = 0;
351 element ++;
352 }
353
354 igtl_videometa_convert_byte_order((igtl_videometa_element*)this->m_Content, this->m_VideoMetaList.size());
355
356 return 1;
357}

360int VideoMetaMessage::UnpackContent()
361{
362
363 this->m_VideoMetaList.clear();
364
365 igtl_videometa_element* element = (igtl_videometa_element*) this->m_Content;
366 int elementsDataLength = this->m_BodySizeToRead;
367#if OpenIGTLink_HEADER_VERSION >= 2
368 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
369 {
370 elementsDataLength = this->m_BodySizeToRead-IGTL_EXTENDED_HEADER_SIZE-this->GetMetaDataHeaderSize()-this->GetMetaDataSize();
371 }
372#endif
373 int nElement = igtl_videometa_get_data_n(elementsDataLength);
374
375 igtl_videometa_convert_byte_order(element, nElement);
376
377 char strbuf[128];
378 for (int i = 0; i < nElement; i ++)
379 {
380 VideoMetaElement::Pointer elemClass = VideoMetaElement::New();
381
382 // Add '\n' at the end of each string
383 // (necessary for a case, where a string reaches the maximum length.)
384 strbuf[IGTL_VIDEOMETA_LEN_NAME] = '\n';
385 strncpy(strbuf, (char*)element->name, IGTL_VIDEOMETA_LEN_NAME);
386 elemClass->SetName((const char*)strbuf);
387
388 strbuf[IGTL_VIDEOMETA_LEN_DEVICE_NAME] = '\n';
389 strncpy(strbuf, (char*)element->device_name, IGTL_VIDEOMETA_LEN_DEVICE_NAME);
390 elemClass->SetDeviceName(strbuf);
391
392 strbuf[IGTL_VIDEOMETA_LEN_PATIENT_NAME] = '\n';
393 strncpy(strbuf, (char*)element->patient_name, IGTL_VIDEOMETA_LEN_PATIENT_NAME);
394 elemClass->SetPatientName(strbuf);
395
396 strbuf[IGTL_VIDEOMETA_LEN_PATIENT_ID] = '\n';
397 strncpy(strbuf, (char*)element->patient_id, IGTL_VIDEOMETA_LEN_PATIENT_ID);
398 elemClass->SetPatientID(strbuf);
399
400 elemClass->SetZoomLevel(element->zoom_level);
401 elemClass->SetFocalLength(element->focal_length);
402 elemClass->SetSize(element->size[0],element->size[1],element->size[2]);
403
404 // Set image orientation
405 float rspacing[3];
406 float origin[3];
407 float norm_i[3];
408 float norm_j[3];
409 float norm_k[3];
410
411 igtl_videometa_get_matrix(rspacing, origin,
412 norm_i, norm_j, norm_k,
413 element);
414
415 elemClass->SetSpacing(rspacing[0], rspacing[1], rspacing[2]);
416 elemClass->SetNormals(norm_i, norm_j, norm_k);
417 elemClass->SetOrigin(origin[0], origin[1], origin[2]);
418
419 Matrix4x4 tmpMatrix;
420 for (int i = 0; i < 3; i ++)
421 {
422 tmpMatrix[i][0] = norm_i[i];
423 tmpMatrix[i][1] = norm_j[i];
424 tmpMatrix[i][2] = norm_k[i];
425 tmpMatrix[i][3] = origin[i];
426 }
427 tmpMatrix[3][0] = 0.0;
428 tmpMatrix[3][1] = 0.0;
429 tmpMatrix[3][2] = 0.0;
430 tmpMatrix[3][3] = 1.0;
431 elemClass->SetMatrix(tmpMatrix);
432
433 elemClass->SetScalarType(element->scalar_type);
434
435 this->m_VideoMetaList.push_back(elemClass);
436
437 element ++;
438 }
439
440 return 1;
441}
