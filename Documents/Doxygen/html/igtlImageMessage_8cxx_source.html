114int ImageMessage::SetSubVolume(int dim[3], int off[3])
115{
116 // make sure that sub-volume fits in the dimensions
117 if (off[0] + dim[0] <= dimensions[0] &&
118 off[1] + dim[1] <= dimensions[1] &&
119 off[2] + dim[2] <= dimensions[2])
120 {
121 m_IsBodyPacked = false;
122 subDimensions[0] = dim[0];
123 subDimensions[1] = dim[1];
124 subDimensions[2] = dim[2];
125 subOffset[0] = off[0];
126 subOffset[1] = off[1];
127 subOffset[2] = off[2];
128 return 1;
129 }
130 else
131 {
132 return 0;
133 }

366int ImageMessage::PackContent()
367{
368 AllocateScalars();
369
370 igtl_image_header* image_header = (igtl_image_header*)m_ImageHeader;
371
372 image_header->header_version = IGTL_IMAGE_HEADER_VERSION;
373 image_header->num_components = this->numComponents;
374 image_header->scalar_type = this->scalarType;
375 image_header->endian = this->endian;
376 image_header->coord = this->coordinate;
377 image_header->size[0] = this->dimensions[0];
378 image_header->size[1] = this->dimensions[1];
379 image_header->size[2] = this->dimensions[2];
380 image_header->subvol_offset[0] = this->subOffset[0];
381 image_header->subvol_offset[1] = this->subOffset[1];
382 image_header->subvol_offset[2] = this->subOffset[2];
383 image_header->subvol_size[0] = this->subDimensions[0];
384 image_header->subvol_size[1] = this->subDimensions[1];
385 image_header->subvol_size[2] = this->subDimensions[2];
386
387 float origin[3];
388 float norm_i[3];
389 float norm_j[3];
390 float norm_k[3];
391
392 for (int i = 0; i < 3; i ++)
393 {
394 norm_i[i] = matrix[i][0];
395 norm_j[i] = matrix[i][1];
396 norm_k[i] = matrix[i][2];
397 origin[i] = matrix[i][3];
398 }
399
400 igtl_image_set_matrix(this->spacing, origin,
401 norm_i, norm_j, norm_k,
402 image_header);
403
404 igtl_image_convert_byte_order(image_header);
405
406 return 1;
407
408}

411int ImageMessage::UnpackContent()
412{
413#if OpenIGTLink_HEADER_VERSION >= 2
414 m_ImageHeader = m_Content;
415#elif OpenIGTLink_PROTOCOL_VERSION <=2
416 m_ImageHeader = m_Body;
417#endif
418 igtl_image_header* image_header = (igtl_image_header*)m_ImageHeader;
419 igtl_image_convert_byte_order(image_header);
420
421 if (image_header->header_version == IGTL_IMAGE_HEADER_VERSION)
422 {
423 // Image format version 1
424 this->scalarType = image_header->scalar_type;
425 this->numComponents = image_header->num_components;
426 this->endian = image_header->endian;
427 this->coordinate = image_header->coord;
428 this->dimensions[0] = image_header->size[0];
429 this->dimensions[1] = image_header->size[1];
430 this->dimensions[2] = image_header->size[2];
431 this->subOffset[0] = image_header->subvol_offset[0];
432 this->subOffset[1] = image_header->subvol_offset[1];
433 this->subOffset[2] = image_header->subvol_offset[2];
434 this->subDimensions[0] = image_header->subvol_size[0];
435 this->subDimensions[1] = image_header->subvol_size[1];
436 this->subDimensions[2] = image_header->subvol_size[2];
437
438 // Set image orientation
439 float rspacing[3];
440 float origin[3];
441 float norm_i[3];
442 float norm_j[3];
443 float norm_k[3];
444
445 igtl_image_get_matrix(rspacing, origin,
446 norm_i, norm_j, norm_k,
447 image_header);
448
449 for (int i = 0; i < 3; i ++)
450 {
451 this->spacing[i] = rspacing[i];
452 matrix[i][0] = norm_i[i];
453 matrix[i][1] = norm_j[i];
454 matrix[i][2] = norm_k[i];
455 matrix[i][3] = origin[i];
456 }
457
458 matrix[3][0] = 0.0;
459 matrix[3][1] = 0.0;
460 matrix[3][2] = 0.0;
461 matrix[3][3] = 1.0;
462
463 m_Image = &m_ImageHeader[IGTL_IMAGE_HEADER_SIZE];
464
465 return 1;
466 }
467 else
468 {
469 // Incompatible version.
470 return 0;
471 }
472}
