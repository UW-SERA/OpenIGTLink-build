41void H265Decoder::ComposeByteSteam(igtl_uint8** inputData, int dimension[2], int iStride[2], igtl_uint8 *outputFrame)
42{
43 int iWidth = dimension[0];
44 int iHeight = dimension[1];
45#pragma omp parallel for default(none) shared(outputByteStream,inputData, iStride, iHeight, iWidth)
46 for (int i = 0; i < iHeight; i++)
47 {
48 igtl_uint8* pPtr = inputData[0]+i*iStride[0];
49 for (int j = 0; j < iWidth; j++)
50 {
51 outputFrame[i*iWidth + j] = pPtr[j];
52 }
53 }
54#pragma omp parallel for default(none) shared(outputByteStream,inputData, iStride, iHeight, iWidth)
55 for (int i = 0; i < iHeight/2; i++)
56 {
57 igtl_uint8* pPtr = inputData[1]+i*iStride[1];
58 for (int j = 0; j < iWidth/2; j++)
59 {
60 outputFrame[i*iWidth/2 + j + iHeight*iWidth] = pPtr[j];
61 }
62 }
63#pragma omp parallel for default(none) shared(outputByteStream, inputData, iStride, iHeight, iWidth)
64 for (int i = 0; i < iHeight/2; i++)
65 {
66 igtl_uint8* pPtr = inputData[2]+i*iStride[1];
67 for (int j = 0; j < iWidth/2; j++)
68 {
69 outputFrame[i*iWidth/2 + j + iHeight*iWidth*5/4] = pPtr[j];
70 }
71 }

75int H265Decoder::DecodeVideoMSGIntoSingleFrame(igtl::VideoMessage* videoMessage, SourcePicture* pDecodedPic)
76{
77 if(videoMessage->GetBitStreamSize())
78 {
79 igtl_int32 iWidth = videoMessage->GetWidth();
80 igtl_int32 iHeight = videoMessage->GetHeight();
81 igtl_uint64 iStreamSize = videoMessage->GetBitStreamSize();
82 igtl_uint16 frameType = videoMessage->GetFrameType();
83 isGrayImage = false;
84 if (frameType>0X00FF)
85 {
86 frameType= frameType>>8;
87 isGrayImage = true;
88 }
89 pDecodedPic->picWidth = iWidth;
90 pDecodedPic->picHeight = iHeight;
91 pDecodedPic->data[1]= pDecodedPic->data[0] + iWidth*iHeight;
92 pDecodedPic->data[2]= pDecodedPic->data[1] + iWidth*iHeight/4;
93 pDecodedPic->stride[0] = iWidth;
94 pDecodedPic->stride[1] = pDecodedPic->stride[2] = iWidth>>1;
95 pDecodedPic->stride[3] = 0;
96 igtl_uint32 dimensions[2] = {static_cast<igtl_uint32>(iWidth), static_cast<igtl_uint32>(iHeight)};
97 return this->DecodeBitStreamIntoFrame(videoMessage->GetPackFragmentPointer(2), pDecodedPic->data[0], dimensions, iStreamSize);
98 }
99 return -1;

102int H265Decoder::DecodeBitStreamIntoFrame(unsigned char* kpH265BitStream,igtl_uint8* outputFrame, igtl_uint32 dimensions[2], igtl_uint64& iStreamSize)
103{
104 static int timePts = 0;
105 timePts++;
106 int got_picture = libOpenHevcDecode(openHevcHandle, kpH265BitStream, iStreamSize, timePts);
107 if (got_picture > 0) {
108 libOpenHevcGetPictureInfo(openHevcHandle, &openHevcFrame.frameInfo);
109 if ((dimensions[0] != openHevcFrame.frameInfo.nWidth) || (dimensions[1] != openHevcFrame.frameInfo.nHeight) || openHevcFrame.pvY==NULL) {
110 openHevcFrame.frameInfo.nWidth = dimensions[0];
111 openHevcFrame.frameInfo.nHeight = dimensions[1];
112 if(openHevcFrame.pvY) {
113 free(openHevcFrame.pvY);
114 free(openHevcFrame.pvU);
115 free(openHevcFrame.pvV);
116 }
117 int format = openHevcFrame.frameInfo.chromat_format == YUV420 ? 1 : 0;
118 openHevcFrame.pvY = calloc (openHevcFrame.frameInfo.nYPitch * openHevcFrame.frameInfo.nHeight, sizeof(unsigned char));
119 openHevcFrame.pvU = calloc (openHevcFrame.frameInfo.nUPitch * openHevcFrame.frameInfo.nHeight >> format, sizeof(unsigned char));
120 openHevcFrame.pvV = calloc (openHevcFrame.frameInfo.nVPitch * openHevcFrame.frameInfo.nHeight >> format, sizeof(unsigned char));
121 }
122 libOpenHevcGetOutputCpy(openHevcHandle, 1, &openHevcFrame);
123 ReconstructFrame(&openHevcFrame, outputFrame);
124 return 1;
125 }
126 return -1;
127}

129int H265Decoder::ReconstructFrame(OpenHevc_Frame_cpy *openHevcFrame,igtl_uint8* outputFrame)
130{
131 int y;
132 int y_offset, y_offset2;
133 int height, format, width;
134 int src_stride;
135 int dst_stride;
136 int src_stride_c;
137 int dst_stride_c;
138
139 format = openHevcFrame->frameInfo.chromat_format == YUV420 ? 1 : 0;
140 src_stride = openHevcFrame->frameInfo.nYPitch;
141 src_stride_c = openHevcFrame->frameInfo.nUPitch;
142 height = openHevcFrame->frameInfo.nHeight;
143 width = openHevcFrame->frameInfo.nWidth;;
144 dst_stride = openHevcFrame->frameInfo.nWidth;
145 dst_stride_c = openHevcFrame->frameInfo.nWidth>>1;
146 y_offset = y_offset2 = 0;
147
148 for (y = 0; y < height; y++)
149 {
150 memcpy(outputFrame+y_offset2, ((char*)openHevcFrame->pvY)+y_offset, dst_stride);
151 y_offset += src_stride;
152 y_offset2 += dst_stride;
153 }
154
155 y_offset = y_offset2 = 0;
156
157 for (y = 0; y < height >> format; y++)
158 {
159 memcpy(outputFrame+y_offset2+width*height, ((char*)openHevcFrame->pvU)+y_offset, dst_stride_c);
160 memcpy(outputFrame+y_offset2+width*height*5/4, ((char*)openHevcFrame->pvV)+y_offset, dst_stride_c);
161 y_offset += src_stride_c;
162 y_offset2 += dst_stride_c;
163 }
164 return 1;
165}
