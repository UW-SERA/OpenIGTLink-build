164int CommandMessage::PackContent()
165{
166 // Allocate buffer
167 AllocateBuffer();
168 igtl_command_header * command_header;
169 char * command;
170
171 // Set pointers
172#if OpenIGTLink_HEADER_VERSION >= 2
173 command_header = (igtl_command_header*) this->m_Content;
174 command = (char *) this->m_Content + sizeof(igtl_command_header);
175#else
176 command_header = (igtl_command_header*) this->m_Body;
177 command = (char *) this->m_Body + sizeof(igtl_command_header);
178#endif
179
180 // Copy data
181 command_header->encoding = static_cast<igtlUint16>(this->m_Encoding);
182 command_header->length = static_cast<igtlUint32>(this->m_Command.length());
183 command_header->commandId = this->m_CommandId;
184 memcpy(command_header->commandName, this->m_CommandName, IGTL_COMMAND_NAME_SIZE);
185 strncpy(command, this->m_Command.c_str(), command_header->length);
186
187 // Convert byte order from host to network
188 igtl_command_convert_byte_order(command_header);
189
190 return 1;
191}

193int CommandMessage::UnpackContent()
194{
195 igtl_command_header * command_header;
196 char * command;
197
198#if OpenIGTLink_HEADER_VERSION >= 2
199 command_header = (igtl_command_header*) this->m_Content;
200 command = (char *) this->m_Content + sizeof(igtl_command_header);
201#else
202 command_header = (igtl_command_header*) this->m_Body;
203 command = (char *) this->m_Body + sizeof(igtl_command_header);
204#endif
205
206 // Convert byte order from network to host
207 igtl_command_convert_byte_order(command_header);
208
209 // Copy data
210 this->m_CommandId = command_header->commandId;
211 memcpy(m_CommandName, command_header->commandName, IGTL_COMMAND_NAME_SIZE);
212 this->m_Encoding = command_header->encoding;
213 this->m_Command.clear();
214 this->m_Command.append(command, command_header->length);
215
216 return 1;
217}
