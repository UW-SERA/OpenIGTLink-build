195int TransformMessage::PackContent()
196{
197 AllocateBuffer();
198
199 // The m_Transform is not aligned with m_Body, when allocatePack() is called.
200 // This is to realign the two pointers.
201#if OpenIGTLink_HEADER_VERSION >= 2
202 this->m_Transform = this->m_Content;
203#else
204 this->m_Transform = m_Body;
205#endif
206 //igtl_float32* transform = (igtl_float32*)this->m_Transform; // doesn't work on Solaris
207 igtl_float32 transform[12];
208
209 for (int i = 0; i < 3; i ++) {
210 transform[i] = matrix[i][0];
211 transform[i+3] = matrix[i][1];
212 transform[i+6] = matrix[i][2];
213 transform[i+9] = matrix[i][3];
214 }
215
216 igtl_transform_convert_byte_order(transform);
217#if OpenIGTLink_HEADER_VERSION >= 2
218 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
219 {
220 memcpy((void*)(this->m_Transform), (void*)transform, sizeof(igtl_float32)*12);
221 }
222 else
223 {
224 memcpy((void*)this->m_Transform, (void*)transform, sizeof(igtl_float32)*12);
225 }
226#elif OpenIGTLink_PROTOCOL_VERSION <= 2
227 memcpy((void*)this->m_Transform, (void*)transform, sizeof(igtl_float32)*12);
228#endif
229
230 return 1;
231}

233int TransformMessage::UnpackContent()
234{
235 this->m_Transform = this->m_Content;
236
237 //igtl_float32* transform = (igtl_float32*)this->m_Transform; // doesn't work on Solaris
238 igtl_float32 transform[12];
239#if OpenIGTLink_HEADER_VERSION >= 2
240 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
241 {
242 memcpy((void*)transform, (void*)(this->m_Transform), sizeof(igtl_float32)*12);
243 }
244 else
245 {
246 memcpy((void*)transform, (void*)this->m_Transform, sizeof(igtl_float32)*12);
247 }
248#elif OpenIGTLink_PROTOCOL_VERSION <= 2
249 memcpy((void*)transform, (void*)this->m_Transform, sizeof(igtl_float32)*12);
250#endif
251
252 igtl_transform_convert_byte_order(transform);
253
254 for (int i = 0; i < 3; i ++) {
255 matrix[i][0] = transform[i] ;
256 matrix[i][1] = transform[i+3];
257 matrix[i][2] = transform[i+6];
258 matrix[i][3] = transform[i+9];
259 }
260
261 matrix[3][0] = 0.0;
262 matrix[3][1] = 0.0;
263 matrix[3][2] = 0.0;
264 matrix[3][3] = 1.0;
265
266 return 1;
267}
