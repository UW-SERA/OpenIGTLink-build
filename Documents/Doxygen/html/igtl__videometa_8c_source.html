55void igtl_export igtl_videometa_get_matrix(float spacing[3], float origin[3],
56 float norm_i[3], float norm_j[3], float norm_k[3],
57 igtl_videometa_element * element)
58{
59 float tx;
60 float ty;
61 float tz;
62 float sx;
63 float sy;
64 float sz;
65 float nx;
66 float ny;
67 float nz;
68 float px;
69 float py;
70 float pz;
71
72 tx = (float) element->matrix[0];
73 ty = (float) element->matrix[1];
74 tz = (float) element->matrix[2];
75 sx = (float) element->matrix[3];
76 sy = (float) element->matrix[4];
77 sz = (float) element->matrix[5];
78 nx = (float) element->matrix[6];
79 ny = (float) element->matrix[7];
80 nz = (float) element->matrix[8];
81 px = (float) element->matrix[9];
82 py = (float) element->matrix[10];
83 pz = (float) element->matrix[11];
84
85 spacing[0] = sqrtf(tx*tx + ty*ty + tz*tz);
86 spacing[1] = sqrtf(sx*sx + sy*sy + sz*sz);
87 spacing[2] = sqrtf(nx*nx + ny*ny + nz*nz);
88
89 if (spacing[0] != 0)
90 {
91 norm_i[0] = element->matrix[0] / spacing[0];
92 norm_i[1] = element->matrix[1] / spacing[0];
93 norm_i[2] = element->matrix[2] / spacing[0];
94 }
95 if (spacing[1] != 0)
96 {
97 norm_j[0] = element->matrix[3] / spacing[1];
98 norm_j[1] = element->matrix[4] / spacing[1];
99 norm_j[2] = element->matrix[5] / spacing[1];
100 }
101 if (spacing[2] != 0)
102 {
103 norm_k[0] = element->matrix[6] / spacing[2];
104 norm_k[1] = element->matrix[7] / spacing[2];
105 norm_k[2] = element->matrix[8] / spacing[2];
106 }
107
108 origin[0] = element->matrix[9];
109 origin[1] = element->matrix[10];
110 origin[2] = element->matrix[11];
111
112}

138void igtl_export igtl_videometa_convert_byte_order(igtl_videometa_element * metalist, int nitem)
139{
140 igtl_videometa_element* elem;
141 int item;
142 int i;
143 igtl_uint32 tmp[12];
144 igtl_uint64 tmp_focal;
145 for (item = 0; item < nitem; item ++)
146 {
147 elem = &(metalist[item]);
148
149 if (igtl_is_little_endian())
150 {
151 elem->zoom_level = BYTE_SWAP_INT16(elem->zoom_level);
152 memcpy(&tmp_focal, &elem->focal_length, sizeof(igtl_float64));
153 tmp_focal = BYTE_SWAP_INT64(tmp_focal);
154 memcpy(&elem->focal_length, &tmp_focal, sizeof(igtl_float64));
155
156 for (i = 0; i < 3; i ++)
157 {
158 elem->size[i] = BYTE_SWAP_INT16(elem->size[i]);
159 }
160
161 memcpy(tmp, elem->matrix, sizeof(igtl_uint32)*12);
162
163 /*
164 * TODO: The following loop may cause segmentation fault, when it is compiled
165 * with '-ftree-vectorize' optimization option on 64-bit Linux.
166 * ('-ftree-vectorize' becomes active, when '-O3' optimization is specified.)
167 * -- code has been updated on June 24 -- needs test
168 */
169 for (i = 0; i < 12; i ++)
170 {
171 tmp[i] = BYTE_SWAP_INT32(tmp[i]);
172 }
173 memcpy(elem->matrix, tmp, sizeof(igtl_uint32)*12);
174
175 }
176 }
177}
