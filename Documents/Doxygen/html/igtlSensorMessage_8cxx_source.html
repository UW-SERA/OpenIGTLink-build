125igtlUint64 SensorMessage::CalculateContentBufferSize()
126{
127 // Body pack size is the sum of LARRAY, STATUS, UNIT and DATA

132int SensorMessage::PackContent()
133{
134 // Allocate buffer
135 AllocateBuffer();
136
137 // Set pointers
138 igtl_sensor_header * sensor_header;
139 igtl_float64 * data;
140 sensor_header = (igtl_sensor_header *) (this->m_Content);
141 data = (igtl_float64 *) (this->m_Content + sizeof(igtl_sensor_header));
142
143 // Copy data
144 sensor_header->larray = this->m_Length;
145 sensor_header->status = this->m_Status;
146 sensor_header->unit = this->m_Unit;
147
148 for (int i = 0; i < this->m_Length; i ++)
149 {
150 data[i] = this->m_Array[i];
151 }
152
153 // Convert byte order from local to network
154 igtl_sensor_convert_byte_order(sensor_header, data);
155
156 return 1;
157}

159int SensorMessage::UnpackContent()
160{
161 // Set pointers
162 igtl_sensor_header * sensor_header;
163 igtl_float64 * data;
164 sensor_header = (igtl_sensor_header *) (this->m_Content);
165 data = (igtl_float64 *) (this->m_Content + sizeof(igtl_sensor_header));
166
167 // Convert byte order from local to network
168 igtl_sensor_convert_byte_order(sensor_header, data);
169
170 // Copy data
171 this->m_Length = sensor_header->larray;
172 this->m_Status = sensor_header->status;
173 this->m_Unit = sensor_header->unit;
174
175 this->m_Array.resize(this->m_Length);
176 for (int i = 0; i < this->m_Length; i ++)
177 {
178 this->m_Array[i] = data[i];
179 }
180
181 return 1;
182}
