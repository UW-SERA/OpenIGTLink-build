51 {
52 glock->Lock();
53 std::map<igtl_uint32, igtl::ReorderBuffer*>::iterator itr = this->reorderBufferMap.begin();
54 while (itr != this->reorderBufferMap.end())
55 {
56 // found it - delete it
57 itr->second->~ReorderBuffer();
58 this->reorderBufferMap.erase(itr);
59 if(this->reorderBufferMap.size()==0)
60 {
61 break;
62 }
63 itr++;
64 }
65 std::map<igtl_uint32, igtl::UnWrappedMessage*>::iterator itr2 = this->unWrappedMessages.begin();
66 while (itr2 != this->unWrappedMessages.end())
67 {
68 // found it - delete it
69 itr2->second->~UnWrappedMessage();
70 this->unWrappedMessages.erase(itr2);
71 if(this->unWrappedMessages.size()==0)
72 {
73 break;
74 }
75 itr2++;
76 }
77 glock->Unlock();
78 glock = NULL;

240 igtlUint16 totMsgLen;
241 if(this->FCFS==true)
242 {
243 totMsgLen = this->incommingPackets.pPacketLengthInByte[0];

246 this->incommingPackets.pPacketLengthInByte.erase(this->incommingPackets.pPacketLengthInByte.begin());
247 this->incommingPackets.pBsBuf.erase(this->incommingPackets.pBsBuf.begin(), this->incommingPackets.pBsBuf.begin()+totMsgLen);

256 this->incommingPackets.pPacketLengthInByte.erase(this->incommingPackets.pPacketLengthInByte.end()-1);
257 this->incommingPackets.pBsBuf.erase(this->incommingPackets.pBsBuf.begin() + this->incommingPackets.pBsBuf.size()-totMsgLen, this->incommingPackets.pBsBuf.end());
258 this->incommingPackets.totalLength -= totMsgLen;
259 }
260 this->glock->Unlock();
261 // Set up the RTP header:
262 igtl_uint32 rtpProfileBytes, timeIncrement;
263 rtpProfileBytes = *((igtl_uint32*)UDPPacket);
264 //bool rtpMarkerBit = (rtpHdr&0x00800000) != 0;
265 timeIncrement = *(igtl_uint32*)(UDPPacket+sizeof(rtpProfileBytes));

267 if(igtl_is_little_endian())
268 {
269 rtpProfileBytes = BYTE_SWAP_INT32(rtpProfileBytes);
270 timeIncrement = BYTE_SWAP_INT32(timeIncrement);
271 SSRC = BYTE_SWAP_INT32(SSRC);
272 }

274 status = WaitingForAnotherPacket;
275 while(curPackedMSGLocation<totMsgLen)

277 igtl::MessageHeader::Pointer header = igtl::MessageHeader::New();
278 header->AllocatePack();

280 igtl_uint16 fragmentField;

282 messageID = 0;
283 //fill the lower two bytes of message ID with the upper two byte from packet data.
284 memcpy((char*)(&messageID)+FragmentIndexBytes, (void*)(UDPPacket + RTP_HEADER_LENGTH+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE-sizeof(messageID)), FragmentIndexBytes);
285 // Get rid of the fragment field number.
286 memcpy((void*)(UDPPacket + RTP_HEADER_LENGTH+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE-sizeof(messageID)), &messageID, sizeof(messageID));
287 if(igtl_is_little_endian())
288 {
289 fragmentField = BYTE_SWAP_INT16(fragmentField);
290 }
291 //this->reorderBufferMap.erase(it);
292 if (reorderBufferMap.size()>ReorderBufferMaximumSize) // get rid of the first reoderBuffer when waiting for a long time
293 {
294 if(reorderBufferMap.begin()->first != messageID)
295 {
296 reorderBufferMap.erase(reorderBufferMap.begin());
297 }
298 else
299 {
300 reorderBufferMap.erase(reorderBufferMap.begin()++);
301 }
302 }
303 std::map<igtl_uint32, igtl::ReorderBuffer*>::iterator it = this->reorderBufferMap.find(messageID);
304 if (it == this->reorderBufferMap.end())
305 {
306 this->reorderBufferMap.insert(std::pair<igtl_uint32,igtl::ReorderBuffer*>(messageID,new igtl::ReorderBuffer()));

323 memcpy(message->messagePackPointer, reorderBuffer->firstFragBuffer, reorderBuffer->firstPacketLen);
324 glock->Lock();
325 unWrappedMessages.insert(std::pair<igtl_uint32, igtl::UnWrappedMessage*>(it->first,message));
326 glock->Unlock();
327 delete it->second;
328 it->second = NULL;
329 this->reorderBufferMap.erase(it);
330 status = MessageReady;
331 }
332 curPackedMSGLocation += header->GetBodySizeToRead()+IGTL_HEADER_SIZE;
333 }
334 else
335 {
336 if (strcmp(header->GetDeviceType(),deviceType)==0 && strcmp(header->GetDeviceName(),deviceName)==0)
337 {
338 int bodyMsgLength = (RTPPayloadLength-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE);//this is the length of the body within a full fragment Packet

342 memcpy(reorderBuffer->firstFragBuffer, UDPPacket + curPackedMSGLocation, totMsgLen-curPackedMSGLocation);
343 reorderBuffer->firstPacketLen = totMsgLen-curPackedMSGLocation;

352 reorderBuffer->receivedLastFrag = true;
353 reorderBuffer->lastPacketLen = totMsgLen-(RTP_HEADER_LENGTH+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE);
354 curPackedMSGLocation = totMsgLen;
355 status = WaitingForAnotherPacket;
356 }
357 else if(fragmentField>FragmentBeginIndicator && fragmentField<FragmentEndIndicator)
358 {
359 int curFragNumber = fragmentField - FragmentBeginIndicator;
360 memcpy(reorderBuffer->buffer+(curFragNumber-1)*bodyMsgLength, UDPPacket + RTP_HEADER_LENGTH+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE, totMsgLen-(RTP_HEADER_LENGTH+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE));
361 status = WaitingForAnotherPacket;
362 }
363 reorderBuffer->filledPacketNum++;
364 if(reorderBuffer->receivedFirstFrag == true && reorderBuffer->receivedLastFrag == true && reorderBuffer->filledPacketNum == reorderBuffer->totFragNumber)
365 {
366 igtl::UnWrappedMessage* message = new igtl::UnWrappedMessage();
367 message->messageDataLength = reorderBuffer->lastPacketLen+reorderBuffer->firstPacketLen+ (reorderBuffer->totFragNumber-2)*(RTPPayloadLength-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE);
368 memcpy(message->messagePackPointer, reorderBuffer->firstFragBuffer, reorderBuffer->firstPacketLen);
369 memcpy(message->messagePackPointer+reorderBuffer->firstPacketLen, reorderBuffer->buffer, (RTPPayloadLength-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE)*(reorderBuffer->totFragNumber-2));
370 memcpy(message->messagePackPointer+reorderBuffer->firstPacketLen+(RTPPayloadLength-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE)*(reorderBuffer->totFragNumber-2), reorderBuffer->lastFragBuffer, reorderBuffer->lastPacketLen);
371 glock->Lock();
372 unWrappedMessages.insert(std::pair<igtl_uint32, igtl::UnWrappedMessage*>(it->first,message));
373 glock->Unlock();
374 std::map<igtl_uint32, igtl::ReorderBuffer*>::iterator it_forDelete= this->reorderBufferMap.begin();
375 while(it_forDelete != this->reorderBufferMap.end())
376 {
377 delete it_forDelete->second;
378 it_forDelete->second = NULL;
379 this->reorderBufferMap.erase(it_forDelete);
380 if (this->reorderBufferMap.size())
381 {
382 if (it_forDelete->first == it->first)

384 break;
385 }
386 }
387 else

389 break;
390 }

392 }
393 status = MessageReady;
394 }
395 }
396 else
397 {
398 curPackedMSGLocation = totMsgLen;
399 }

401 }
402 }
403 delete[] UDPPacket;
404 return 1;
405 }
406 return 0;

434 }

436 leftmessageContent = messageContentPointer + this->GetCurMSGLocation();
437 leftMsgLen = MSGContentLength - this->GetCurMSGLocation();
438 }while(leftMsgLen>0 && status!=igtl::MessageRTPWrapper::PacketReady); // to do when bodyMsgLen

442 void MessageRTPWrapper::SleepInNanoSecond(int nanoSecond)
443 {
444#if defined(_WIN32) && !defined(__CYGWIN__)
445
446 // Call Windows Native Sleep() function
447 //::Sleep(nanoSecond/1000);
448
449#else
450
451 struct timespec req;
452 req.tv_sec = (int) nanoSecond/1000000000;
453 req.tv_nsec = nanoSecond % 1000000000;
454
455 while ((nanosleep(&req, &req) == -1) && (errno == EINTR))
456 {
457 continue;
458 }
459
460#endif
461 }
462
463 // SSRC should be set only once before wraping the message in a transmission session.
464 // CSRC should be set when messages from different devices are packed into the same Packet.
465 // this special header is here for the compatiblity with standard RTP protocal.
466 // we have our own defined openigtlink header, which is 58 byte size.
467 // append the openigtlink header as the Extension header after CSRC.
468 int MessageRTPWrapper::WrapMessage(igtl_uint8* messageContent, int bodyMsgLen)
469 {
470 // Set up the RTP header:
471 igtl_uint32 rtpProfileBytes = 0x80000000; // RTP version 2;
472 rtpProfileBytes |= (RTPPayloadType<<16);
473 rtpProfileBytes |= SeqNum; // sequence number, increment the sequence number after sending the data
474 #if defined(WIN32) || defined(_WIN32)
475
476 //LARGE_INTEGER tick;
477 //
478 //::QueryPerformanceCounter( &tick );
479 //
480 //TimeStampType value =
481 // static_cast< TimeStampType >( (__int64)tick.QuadPart ) /
482 // this->m_WinFrequency;
483 //
484 //value += this->m_WinOrigin;
485 //
486 //double second = floor(value);
487
488 clock_t time1 = clock();
489 clock_t winOriginTime = time(NULL);
490 clock_t second = + (time1 - winOriginTime) / CLOCKS_PER_SEC;
491 clock_t microsecond = (time1 - winOriginTime) % CLOCKS_PER_SEC * (1e6 / CLOCKS_PER_SEC);
492 clock_t timeIncrement = appSpecificFreq*second;
493 timeIncrement += igtl_uint32(appSpecificFreq*(microsecond / 1.0e6) + 0.5);
494 #else
495 struct timeval timeNow;
496 gettimeofday(&timeNow, NULL);
497 igtl_uint32 timeIncrement = (appSpecificFreq*timeNow.tv_sec); //need to check the frequency of different application
498 timeIncrement += igtl_uint32(appSpecificFreq*(timeNow.tv_usec/1.0e6)+ 0.5);
499 #endif
500 //igtl_uint32 CSRC = 0x00000000; not used currently
501 if(igtl_is_little_endian())
502 {
503 rtpProfileBytes = BYTE_SWAP_INT32(rtpProfileBytes);
504 timeIncrement = BYTE_SWAP_INT32(timeIncrement);
505 }
506 if (status == PacketReady)
507 {
508 delete packedMsg;
509 packedMsg = new unsigned char[RTPPayloadLength + RTP_HEADER_LENGTH];
510 AvailabeBytesNum = RTPPayloadLength;
511 curMSGLocation = 0;
512 curPackedMSGLocation = 0;
513 fragmentNumber = -1; // -1 = 0XFFFF to indicate the message has no fragments
514 }
515 if (status != ProcessFragment)
516 {
517 this->fragmentTimeIncrement = timeIncrement;
518 memcpy(packedMsg, (void *)(&rtpProfileBytes), sizeof(rtpProfileBytes));
519 memcpy(packedMsg+sizeof(rtpProfileBytes), (void *)(&timeIncrement), sizeof(timeIncrement));
520 memcpy(packedMsg+sizeof(rtpProfileBytes)+sizeof(timeIncrement), (void *)(&SSRC), sizeof(SSRC)); // SSRC needs to set by different devices, collision should be avoided.
521 curPackedMSGLocation += RTP_HEADER_LENGTH;
522 if (bodyMsgLen <= (AvailabeBytesNum-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE))
523 {
524 igtl_uint16 temp = NoFragmentIndicator;
525 memcpy(this->MSGHeader+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE-FragmentIndexBytes, (void*)&temp,FragmentIndexBytes);// no fragmented message here
526 memcpy(packedMsg+curPackedMSGLocation, this->MSGHeader, IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE);
527 curPackedMSGLocation += (IGTL_HEADER_SIZE + IGTL_EXTENDED_HEADER_SIZE);
528 memcpy(packedMsg + curPackedMSGLocation, (void *)(messageContent), bodyMsgLen);
529 AvailabeBytesNum -= (bodyMsgLen+IGTL_EXTENDED_HEADER_SIZE+IGTL_HEADER_SIZE);
530 curPackedMSGLocation += bodyMsgLen;
531 curMSGLocation += bodyMsgLen;
532 status = PacketReady;// now we send the Packet immediately, however, it is possible to send after the packt is full.
533 SeqNum++;
534 }
535 else if(bodyMsgLen > (AvailabeBytesNum-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE))
536 {
537 igtl_uint16 temp = FragmentBeginIndicator;
538 if(igtl_is_little_endian())
539 {
540 temp = BYTE_SWAP_INT16(temp);
541 }
542 memcpy(this->MSGHeader+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE-FragmentIndexBytes, (void*)&temp,FragmentIndexBytes);// no fragmented message here
543 // fragmented message exists, first bit indicate the existance, the second bit indicates if the current section is the ending fragment, the other 14 bits indicates the fragements' sequence number.
544 memcpy(packedMsg+curPackedMSGLocation, this->MSGHeader, IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE);
545 curPackedMSGLocation += (IGTL_HEADER_SIZE + IGTL_EXTENDED_HEADER_SIZE);
546 memcpy(packedMsg + curPackedMSGLocation, (void *)(messageContent), AvailabeBytesNum-IGTL_EXTENDED_HEADER_SIZE-IGTL_HEADER_SIZE);
547 status = ProcessFragment;
548 this->curPackedMSGLocation = RTPPayloadLength+RTP_HEADER_LENGTH;
549 this->curMSGLocation = AvailabeBytesNum-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE;
550 AvailabeBytesNum = RTPPayloadLength;
551 SeqNum++;
552 fragmentNumber++;
553 }
554 }
555 else
556 {
557 memcpy(packedMsg, (void *)(&rtpProfileBytes), sizeof(rtpProfileBytes));
558 memcpy(packedMsg+sizeof(rtpProfileBytes), (void *)(&this->fragmentTimeIncrement), sizeof(this->fragmentTimeIncrement));
559 memcpy(packedMsg+sizeof(rtpProfileBytes)+sizeof(this->fragmentTimeIncrement), (void *)(&SSRC), sizeof(SSRC)); // SSRC needs to set by different devices, collision should be avoided.
560 curPackedMSGLocation = RTP_HEADER_LENGTH;
561 fragmentNumber++;
562 if (bodyMsgLen <= (AvailabeBytesNum-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE))
563 {
564 igtl_uint16 temp = FragmentEndIndicator+fragmentNumber; //set the seconde bit to be 1, indicates the end of fragmented msg.;
565 if(igtl_is_little_endian())
566 {
567 temp = BYTE_SWAP_INT16(temp);
568 }
569 memcpy(this->MSGHeader+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE-FragmentIndexBytes, (void*)&temp,FragmentIndexBytes);
570 // fragmented message exists, first bit indicate the existance, the second bit indicates if the current section is the ending fragment, the other 14 bits indicates the fragements' sequence number.
571 memcpy(packedMsg+curPackedMSGLocation, this->MSGHeader, IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE);
572 curPackedMSGLocation += (IGTL_HEADER_SIZE + IGTL_EXTENDED_HEADER_SIZE);
573 memcpy(packedMsg + curPackedMSGLocation, (void *)(messageContent), bodyMsgLen);
574 this->curPackedMSGLocation += bodyMsgLen;
575 // when it is packing the fragment, we want to sent the data ASAP, otherwize, we will wait for another message
576 this->curMSGLocation += bodyMsgLen;
577 status = PacketReady;
578 }
579 else if(bodyMsgLen > (AvailabeBytesNum-IGTL_HEADER_SIZE-IGTL_EXTENDED_HEADER_SIZE))
580 {
581 igtl_uint16 temp = FragmentBeginIndicator+fragmentNumber;
582 if(igtl_is_little_endian())
583 {
584 temp = BYTE_SWAP_INT16(temp);
585 }
586 memcpy(this->MSGHeader+IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE-FragmentIndexBytes, (void*)&temp,FragmentIndexBytes);//set the second bit to be 0, indicates more fragmented msg are comming.
587 memcpy(packedMsg+curPackedMSGLocation, MSGHeader, IGTL_HEADER_SIZE+IGTL_EXTENDED_HEADER_SIZE);
588 curPackedMSGLocation += (IGTL_HEADER_SIZE + IGTL_EXTENDED_HEADER_SIZE);
589 memcpy(packedMsg + curPackedMSGLocation, (void *)(messageContent), AvailabeBytesNum-IGTL_EXTENDED_HEADER_SIZE-IGTL_HEADER_SIZE);
590 status = ProcessFragment;
591 this->curMSGLocation += (AvailabeBytesNum-IGTL_EXTENDED_HEADER_SIZE-IGTL_HEADER_SIZE);
592 curPackedMSGLocation += (AvailabeBytesNum-IGTL_EXTENDED_HEADER_SIZE-IGTL_HEADER_SIZE);
593 }
594 AvailabeBytesNum = RTPPayloadLength;
595 SeqNum++;
596 }
597 return status;
598 }
