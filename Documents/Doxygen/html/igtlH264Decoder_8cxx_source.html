90void H264Decoder::ComposeByteSteam(igtl_uint8** inputData, int dimension[2], int iStride[2], igtl_uint8 *outputFrame)
91{
92 int iWidth = dimension[0];
93 int iHeight = dimension[1];
94#pragma omp parallel for default(none) shared(outputByteStream,inputData, iStride, iHeight, iWidth)
95 for (int i = 0; i < iHeight; i++)
96 {
97 igtl_uint8* pPtr = inputData[0]+i*iStride[0];
98 for (int j = 0; j < iWidth; j++)
99 {
100 outputFrame[i*iWidth + j] = pPtr[j];
101 }
102 }
103#pragma omp parallel for default(none) shared(outputByteStream,inputData, iStride, iHeight, iWidth)
104 for (int i = 0; i < iHeight/2; i++)
105 {
106 igtl_uint8* pPtr = inputData[1]+i*iStride[1];
107 for (int j = 0; j < iWidth/2; j++)
108 {
109 outputFrame[i*iWidth/2 + j + iHeight*iWidth] = pPtr[j];
110 }
111 }
112#pragma omp parallel for default(none) shared(outputByteStream, inputData, iStride, iHeight, iWidth)
113 for (int i = 0; i < iHeight/2; i++)
114 {
115 igtl_uint8* pPtr = inputData[2]+i*iStride[1];
116 for (int j = 0; j < iWidth/2; j++)
117 {
118 outputFrame[i*iWidth/2 + j + iHeight*iWidth*5/4] = pPtr[j];
119 }
120 }
121}

124int H264Decoder::DecodeVideoMSGIntoSingleFrame(igtl::VideoMessage* videoMessage, SourcePicture* pDecodedPic)
125{
126 if(videoMessage->GetBitStreamSize())
127 {
128 igtl_int32 iWidth = videoMessage->GetWidth();
129 igtl_int32 iHeight = videoMessage->GetHeight();
130 igtl_uint64 iStreamSize = videoMessage->GetBitStreamSize();
131 igtl_uint16 frameType = videoMessage->GetFrameType();
132 isGrayImage = false;
133 if (frameType>0X00FF)
134 {
135 frameType= frameType>>8;
136 isGrayImage = true;
137 }
138 pDecodedPic->picWidth = iWidth;
139 pDecodedPic->picHeight = iHeight;
140 pDecodedPic->data[1]= pDecodedPic->data[0] + iWidth*iHeight;
141 pDecodedPic->data[2]= pDecodedPic->data[1] + iWidth*iHeight/4;
142 pDecodedPic->stride[0] = iWidth;
143 pDecodedPic->stride[1] = pDecodedPic->stride[2] = iWidth>>1;
144 pDecodedPic->stride[3] = 0;
145 igtl_uint32 dimensions[2] = {static_cast<igtl_uint32>(iWidth), static_cast<igtl_uint32>(iHeight)};
146 return this->DecodeBitStreamIntoFrame(videoMessage->GetPackFragmentPointer(2), pDecodedPic->data[0], dimensions, iStreamSize);
147 }
148 return -1;

151int H264Decoder::DecodeBitStreamIntoFrame(unsigned char* kpH264BitStream,igtl_uint8* outputFrame, igtl_uint32 dimensions[2], igtl_uint64& iStreamSize) {
152
153 unsigned long long uiTimeStamp = 0;
154 igtl_int64 iStart = 0, iEnd = 0, iTotal = 0;
155 igtl_int32 iSliceSize;
156 igtl_int32 iSliceIndex = 0;
157 igtl_uint32 iWidth = dimensions[0];
158 igtl_uint32 iHeight = dimensions[1];
159 unsigned char* pBuf = NULL;
160 unsigned char uiStartCode[4] = {0, 0, 0, 1};
161
162 unsigned char* pData[3] = {NULL};
163 SBufferInfo sDstBufInfo;
164
165 igtl_int32 iBufPos = 0;
166 igtl_int32 i = 0;
167 igtl_int32 iFrameCount = 0;
168 igtl_int32 iEndOfStreamFlag = 0;
169 //for coverage test purpose
170 igtl_int32 iErrorConMethod = (igtl_int32) ERROR_CON_SLICE_MV_COPY_CROSS_IDR_FREEZE_RES_CHANGE;
171 pDecoder->SetOption (DECODER_OPTION_ERROR_CON_IDC, &iErrorConMethod);
172 //~end for
173 double dElapsed = 0;
174
175 if (iStreamSize <= 0)
176 {
177 //fprintf (stderr, "Current Bit Stream File is too small, read error!!!!\n");
178 goto label_exit;
179 }
180 pBuf = new unsigned char[iStreamSize + 5];
181 if (pBuf == NULL)
182 {
183 //fprintf (stderr, "new buffer failed!\n");
184 goto label_exit;
185 }
186 memcpy (pBuf, kpH264BitStream, iStreamSize);
187 memcpy (pBuf + iStreamSize, &uiStartCode[0], 4); //confirmed_safe_unsafe_usage
188
189 while (true)
190 {
191 if (iBufPos >= iStreamSize)
192 {
193 iEndOfStreamFlag = true;
194 if (iEndOfStreamFlag)
195 pDecoder->SetOption (DECODER_OPTION_END_OF_STREAM, (void*)&iEndOfStreamFlag);
196 break;
197 }
198 for (i = 0; i < iStreamSize; i++)
199 {
200 if ((pBuf[iBufPos + i] == 0 && pBuf[iBufPos + i + 1] == 0 && pBuf[iBufPos + i + 2] == 0 && pBuf[iBufPos + i + 3] == 1
201 && i > 0) || (pBuf[iBufPos + i] == 0 && pBuf[iBufPos + i + 1] == 0 && pBuf[iBufPos + i + 2] == 1 && i > 0))
202 {
203 break;
204 }
205 }
206 iSliceSize = i;
207 if (iSliceSize < 4)
208 { //too small size, no effective data, ignore
209 iBufPos += iSliceSize;
210 continue;
211 }
212
213 iStart = getCurrentTime();
214 pData[0] = NULL;
215 pData[1] = NULL;
216 pData[2] = NULL;
217 uiTimeStamp ++;
218 memset (&sDstBufInfo, 0, sizeof (SBufferInfo));
219 sDstBufInfo.uiInBsTimeStamp = uiTimeStamp;
220 sDstBufInfo.UsrData.sSystemBuffer.iWidth =
221#ifndef NO_DELAY_DECODING
222 pDecoder->DecodeFrameNoDelay (pBuf + iBufPos, iSliceSize, pData, &sDstBufInfo);
223#else
224 pDecoder->DecodeFrame2 (pBuf + iBufPos, iSliceSize, pData, &sDstBufInfo);
225#endif
226
227 iEnd = getCurrentTime();
228 iTotal = iEnd - iStart;
229 if (sDstBufInfo.iBufferStatus == 1)
230 {
231 Process ((void**)pData, &sDstBufInfo, NULL);
232 iWidth = sDstBufInfo.UsrData.sSystemBuffer.iWidth;
233 iHeight = sDstBufInfo.UsrData.sSystemBuffer.iHeight;
234 ++ iFrameCount;
235 }
236
237#ifdef NO_DELAY_DECODING
238 iStart = getCurrentTime();
239 pData[0] = NULL;
240 pData[1] = NULL;
241 pData[2] = NULL;
242 memset (&sDstBufInfo, 0, sizeof (SBufferInfo));
243 sDstBufInfo.uiInBsTimeStamp = uiTimeStamp;
244 pDecoder->DecodeFrame2 (NULL, 0, pData, &sDstBufInfo);
245 iEnd = getCurrentTime();
246 iTotal = iEnd - iStart;
247 if (sDstBufInfo.iBufferStatus == 1)
248 {
249 Process ((void**)pData, &sDstBufInfo, NULL);
250 int dimension[2] = {static_cast<int>(iWidth), static_cast<int>(iHeight)};
251 int stride[2] = {sDstBufInfo.UsrData.sSystemBuffer.iStride[0],sDstBufInfo.UsrData.sSystemBuffer.iStride[1]};
252 ComposeByteSteam(pData, dimension, stride, outputFrame);
253 iWidth = sDstBufInfo.UsrData.sSystemBuffer.iWidth;
254 iHeight = sDstBufInfo.UsrData.sSystemBuffer.iHeight;
255 ++ iFrameCount;
256 }
257#endif
258 if (iFrameCount)
259 {
260 dElapsed = iTotal / 1e6;
261 }
262 iBufPos += iSliceSize;
263 ++ iSliceIndex;
264 }
265 // coverity scan uninitial
266label_exit:
267 if (pBuf) {
268 delete[] pBuf;
269 pBuf = NULL;
270 }
271 if (iFrameCount)
272 {
273 return 2;
274 }
275 return -1;
276}
