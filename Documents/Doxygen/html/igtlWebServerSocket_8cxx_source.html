111void webSocketServer::on_timer() {
112 // Broadcast count to all connections
113 unique_lock<mutex> lock(m_action_lock);
114
115 while(m_actions.empty()) {
116 m_action_cond.wait(lock);
117 }
118 action a = m_actions.front();
119 m_actions.pop();
120 lock.unlock();
121
122 con_list::iterator it;
123 for (it = m_connections.begin(); it != m_connections.end(); ++it) {
124 if (a.msg.length())
125 {
126 m_endpoint.send(*it,a.msg,websocketpp::frame::opcode::binary);
127 }
128 }
129 // set timer for next telemetry check
130 set_timer();
131}

133void webSocketServer::on_http(connection_hdl hdl) {
134 // Upgrade our connection handle to a full connection_ptr
135 server::connection_ptr con = m_endpoint.get_con_from_hdl(hdl);
136
137 std::ifstream file;
138 std::string filename = con->get_resource();
139 std::string response;
140
141 //m_endpoint.get_alog().write(websocketpp::log::alevel::app, "http request1: "+filename);
142
143 if (filename == "/")
144 {
145 filename = m_docroot+"index.html";
146 }
147 else
148 {
149 filename = m_docroot+filename.substr(1);
150 }
151
152 //m_endpoint.get_alog().write(websocketpp::log::alevel::app, "http request2: "+filename);
153
154 file.open(filename.c_str(), std::ios::in);
155 if (!file)
156 {
157 // 404 error
158 std::stringstream ss;
159
160 ss << "<!doctype html><html><head>"
161 << "<title>Error 404 (Resource not found)</title><body>"
162 << "<h1>Error 404</h1>"
163 << "<p>The requested URL " << filename << " was not found on this server.</p>"
164 << "</body></head></html>";
165
166 con->set_body(ss.str());
167 con->set_status(websocketpp::http::status_code::not_found);
168 return;
169 }
170
171 file.seekg(0, std::ios::end);
172 response.reserve(file.tellg());
173 file.seekg(0, std::ios::beg);
174
175 response.assign((std::istreambuf_iterator<char>(file)),
176 std::istreambuf_iterator<char>());
177
178 con->set_body(response);
179 con->set_status(websocketpp::http::status_code::ok);
180}
