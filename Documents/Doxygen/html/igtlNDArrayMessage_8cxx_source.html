46int ArrayBase::SetSize(IndexType size)
47{
48
49 if (this->m_Size == size)
50 {
51 // If the size of the array is same as specified,
52 // do nothing
53 return 1;
54 }
55
56 if (this->m_ByteArray != NULL)
57 {
58 delete (igtlUint8 *) this->m_ByteArray;
59 }
60 this->m_Size = size;
61
62 this->m_ByteArray = (void *) new igtlUint8[GetRawArraySize()];
63
64 if (this->m_ByteArray)
65 {
66 return 1;
67 }
68 else
69 {
70 this->m_Size.clear();
71 return 0;
72 }
73}

108igtlUint32 ArrayBase::Get1DIndex(IndexType index)
109{
110 igtlUint32 p;
111 igtlUint32 step;
112 igtlUint16 dim = this->m_Size.size();
113
114 if (this->m_Size.size() != index.size())
115 {
116 // TODO: how to pass the error to the caller?
117 return 0;
118 }
119
120 p = 0;
121 step = 1;
122 IndexType::iterator iter;
123 for (int i = dim-1; i >= 0; i --)
124 {
125 p += index[i] * step;
126 step *= this->m_Size[i];
127 }
128 igtlUint32 m = GetNumberOfElements();
129 if (p <= m)
130 {
131 return p;
132 }
133 else
134 {
135 // TODO: how to pass the error to the caller?
136 return m;
137 }
138}

178igtlUint64 NDArrayMessage::CalculateContentBufferSize()
179{
180 igtlUint64 dataSize;
181 int dim;
182 if (this->m_Array == NULL)
183 {
184 return 0;
185 }
186 dim = this->m_Array->GetDimension();
187 dataSize = sizeof(igtlUint8) * 2 + sizeof(igtlUint16) * (igtl_uint64) dim
188 + this->m_Array->GetRawArraySize();
189
190 return dataSize;
191}

194int NDArrayMessage::PackContent()
195{
196 // Allocate buffer
197 AllocateBuffer();
198
199 if (this->m_Array == NULL)
200 {
201 return 0;
202 }
203
204 igtl_ndarray_info info;
205
206 igtl_ndarray_init_info(&info);
207 info.dim = this->m_Array->GetDimension();
208 info.type = this->m_Type;
209
210 ArrayBase::IndexType size = this->m_Array->GetSize();
211
212 igtl_uint16 * s = new igtl_uint16[info.dim];
213 for (int i = 0; i < info.dim; i ++)
214 {
215 s[i] = size[i];
216 }
217 int r = igtl_ndarray_alloc_info(&info, s);
218 delete [] s;
219
220 if (r == 0)
221 {
222 return 0;
223 }
224
225// size.data() doesn't work for some environtments (MacOS X 10.5 and Win32, as far as I know)
226// if (igtl_ndarray_alloc_info(&info, size.data()) == 0)
227// {
228// return 0;
229// }
230
231 memcpy(info.array, this->m_Array->GetRawArray(), this->m_Array->GetRawArraySize());
232 igtl_ndarray_pack(&info, this->m_Content, IGTL_TYPE_PREFIX_NONE);
233
234 return 1;
235}

238int NDArrayMessage::UnpackContent()
239{
240 igtl_ndarray_info info;
241 bool isUnpacked(true);
242 igtl_ndarray_unpack(IGTL_TYPE_PREFIX_NONE, this->m_Content, &info, this->CalculateReceiveContentSize(isUnpacked));
243
244 this->m_Type = info.type;
245 ArrayBase::IndexType size;
246 size.resize(info.dim);
247 for (int i = 0; i < info.dim; i ++)
248 {
249 size[i] = info.size[i];
250 }
251
252 switch (this->m_Type)
253 {
254 case TYPE_INT8:
255 this->m_Array = new Array<igtlInt8>;
256 break;
257 case TYPE_UINT8:
258 this->m_Array = new Array<igtlUint8>;
259 break;
260 case TYPE_INT16:
261 this->m_Array = new Array<igtlInt16>;
262 break;
263 case TYPE_UINT16:
264 this->m_Array = new Array<igtlUint16>;
265 break;
266 case TYPE_INT32:
267 this->m_Array = new Array<igtlInt32>;
268 break;
269 case TYPE_UINT32:
270 this->m_Array = new Array<igtlUint32>;
271 break;
272 case TYPE_FLOAT32:
273 this->m_Array = new Array<igtlFloat32>;
274 break;
275 case TYPE_FLOAT64:
276 this->m_Array = new Array<igtlFloat64>;
277 break;
278 case TYPE_COMPLEX:
279 this->m_Array = new Array<igtlComplex>;
280 break;
281 default:
282 return 0;
283 break;
284 }
285
286 this->m_Array->SetSize(size);
287 memcpy(this->m_Array->GetRawArray(), info.array, this->m_Array->GetRawArraySize());
288
289 return 1;
290}
