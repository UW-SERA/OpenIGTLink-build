59bool UDPServerSocket::IsMulticastAddreesValid(const char* add)
60{
61 //224.0.0.0 through 224.0.0.255, are not routable
62 igtl_uint32 address = inet_addr(add); // to do: make sure the endian is correct
63 if(igtl_is_little_endian())
64 {
65 address = BYTE_SWAP_INT32(address);
66 }
67 return address > 0xE00000FF &&
68 address <= 0xEFFFFFFF;
69}

108//-----------------------------------------------------------------------------
109int UDPServerSocket::WriteSocket(unsigned char* buffer, unsigned bufferSize)
110{
111 int numByteSend = 0;
112 for(std::vector<GroupDest>::size_type i = 0; i < this->groups.size(); i++)
113 {
114 this->SetIPAddress((const char*)this->groups[i].address);
115 this->SetPortNumber(this->groups[i].portNum);
116 numByteSend = SendUDP((char*)buffer, bufferSize);
117 }
118 for(std::vector<ClientDest>::size_type i = 0; i < this->clients.size(); i++)
119 {
120 this->SetIPAddress((const char*)this->clients[i].address);
121 this->SetPortNumber(this->clients[i].portNum);
122 numByteSend = SendUDP((char*)buffer, bufferSize);
123 }
124 return numByteSend;
125}

128int UDPServerSocket::CreateUDPServer()
129{
130 if (this->m_SocketDescriptor != -1)
131 {
132 igtlWarningMacro("Server Socket already exists. Closing old socket.");
133 this->CloseSocket(this->m_SocketDescriptor);
134 this->m_SocketDescriptor = -1;

137 if (this->m_SocketDescriptor < 0)

139 return -1;
