240int ImageMetaMessage::PackContent()
241{
242 // Allocate buffer
243 AllocateBuffer();
244
245 igtl_imgmeta_element* element = (igtl_imgmeta_element*)this->m_Content;
246
247 std::vector<ImageMetaElement::Pointer>::iterator iter;
248 for (iter = this->m_ImageMetaList.begin(); iter != this->m_ImageMetaList.end(); iter ++)
249 {
250 strncpy((char*)element->name, (*iter)->GetName(), IGTL_IMGMETA_LEN_NAME);
251 strncpy((char*)element->device_name, (*iter)->GetDeviceName(), IGTL_IMGMETA_LEN_DEVICE_NAME);
252 strncpy((char*)element->modality, (*iter)->GetModality(), IGTL_IMGMETA_LEN_MODALITY);
253 strncpy((char*)element->patient_name, (*iter)->GetPatientName(), IGTL_IMGMETA_LEN_PATIENT_NAME);
254 strncpy((char*)element->patient_id, (*iter)->GetPatientID(), IGTL_IMGMETA_LEN_PATIENT_ID);
255
256 igtl::TimeStamp::Pointer ts;
257 (*iter)->GetTimeStamp(ts);
258 if (ts.IsNotNull())
259 {
260 element->timestamp = ts->GetTimeStampUint64();
261 }
262 else
263 {
264 element->timestamp = 0;
265 }
266 igtlUint16 size[3];
267 (*iter)->GetSize(size);
268 element->size[0] = size[0];
269 element->size[1] = size[1];
270 element->size[2] = size[2];
271 element->scalar_type = (*iter)->GetScalarType();
272 element->reserved = 0;
273 element ++;
274 }
275
276 igtl_imgmeta_convert_byte_order((igtl_imgmeta_element*)this->m_Content, this->m_ImageMetaList.size());
277
278 return 1;
279}

282int ImageMetaMessage::UnpackContent()
283{
284
285 this->m_ImageMetaList.clear();
286
287 igtl_imgmeta_element* element = (igtl_imgmeta_element*) this->m_Content;
288 int nElement = igtl_imgmeta_get_data_n(this->m_BodySizeToRead);
289
290 igtl_imgmeta_convert_byte_order(element, nElement);
291
292 char strbuf[128];
293 for (int i = 0; i < nElement; i ++)
294 {
295 ImageMetaElement::Pointer elemClass = ImageMetaElement::New();
296
297 // Add '\n' at the end of each string
298 // (necessary for a case, where a string reaches the maximum length.)
299 strbuf[IGTL_IMGMETA_LEN_NAME] = '\n';
300 strncpy(strbuf, (char*)element->name, IGTL_IMGMETA_LEN_NAME);
301 elemClass->SetName((const char*)strbuf);
302
303 strbuf[IGTL_IMGMETA_LEN_DEVICE_NAME] = '\n';
304 strncpy(strbuf, (char*)element->device_name, IGTL_IMGMETA_LEN_DEVICE_NAME);
305 elemClass->SetDeviceName(strbuf);
306
307 strbuf[IGTL_IMGMETA_LEN_MODALITY] = '\n';
308 strncpy(strbuf, (char*)element->modality, IGTL_IMGMETA_LEN_MODALITY);
309 elemClass->SetModality(strbuf);
310
311 strbuf[IGTL_IMGMETA_LEN_PATIENT_NAME] = '\n';
312 strncpy(strbuf, (char*)element->patient_name, IGTL_IMGMETA_LEN_PATIENT_NAME);
313 elemClass->SetPatientName(strbuf);
314
315 strbuf[IGTL_IMGMETA_LEN_PATIENT_ID] = '\n';
316 strncpy(strbuf, (char*)element->patient_id, IGTL_IMGMETA_LEN_PATIENT_ID);
317 elemClass->SetPatientID(strbuf);
318
319 TimeStamp::Pointer ts = TimeStamp::New();
320 ts->SetTime(element->timestamp);
321 elemClass->SetTimeStamp(ts);
322
323 elemClass->SetSize(element->size);
324 elemClass->SetScalarType(element->scalar_type);
325
326 this->m_ImageMetaList.push_back(elemClass);
327
328 element ++;
329 }
330
331 return 1;
332}
