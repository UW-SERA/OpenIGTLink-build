21void igtl_export igtl_tdata_convert_byte_order(igtl_tdata_element* tdatalist, int nitem)
22{
23 igtl_tdata_element* elem;
24 int i;
25 int j;
26 igtl_int32* tmp;
27
28 for (i = 0; i < nitem; i ++)
29 {
30 elem = &(tdatalist[i]);
31 if (igtl_is_little_endian())
32 {
33 for (j = 0; j < 12; j ++)
34 {
35 tmp = (igtl_int32*)&(elem->transform[j]);
36 *tmp = BYTE_SWAP_INT32(*tmp);
37 }
38 }
39 }
40}
