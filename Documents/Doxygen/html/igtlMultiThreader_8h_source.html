28// .NAME igtlMultiThreader - A class for performing multithreaded execution
29// .SECTION Description
30// igtlMultithreader is a class that provides support for multithreaded
31// execution using sproc() on an SGI, or pthread_create on any platform
32// supporting POSIX threads. This class can be used to execute a single
33// method on multiple threads, or to specify a method per thread.
34

138 // Description:
139 // This is the structure that is passed to the thread that is
140 // created from the SingleMethodExecute, MultipleMethodExecute or
141 // the SpawnThread method. It is passed in as a void *, and it is
142 // up to the method to cast correctly and extract the information.
143 // The ThreadID is a number between 0 and NumberOfThreads-1 that indicates
144 // the id of this thread. The NumberOfThreads is this->NumberOfThreads for
145 // threads created from SingleMethodExecute or MultipleMethodExecute,
146 // and it is 1 for threads created from SpawnThread.
147 // The UserData is the (void *)arg passed into the SetSingleMethod,
148 // SetMultipleMethod, or SpawnThread method.
149
150 //BTX
151#define ThreadInfoStruct MultiThreader::ThreadInfo
152 class ThreadInfo
153 {
154 public:
155 int ThreadID;
156 int NumberOfThreads;
157 int *ActiveFlag;
158 MutexLock::Pointer ActiveFlagLock;
159 void *UserData;
160 };
161 //ETX
162
163 // Description:
164 // Get/Set the number of threads to create. It will be clamped to the range
165 // 1 - IGTL_MAX_THREADS, so the caller of this method should check that the
166 // requested number of threads was accepted.
167 igtlSetClampMacro( NumberOfThreads, int, 1, IGTL_MAX_THREADS );
168 virtual int GetNumberOfThreads();
169
170 // Description:
171 // Set/Get the maximum number of threads to use when multithreading.
172 // This limits and overrides any other settings for multithreading.
173 // A value of zero indicates no limit.
174 static void SetGlobalMaximumNumberOfThreads(int val);
175 static int GetGlobalMaximumNumberOfThreads();
176
177 // Description:
178 // Set/Get the value which is used to initialize the NumberOfThreads
179 // in the constructor. Initially this default is set to the number of
180 // processors or IGTL_MAX_THREADS (which ever is less).
181 static void SetGlobalDefaultNumberOfThreads(int val);
182 static int GetGlobalDefaultNumberOfThreads();
183
184 // These methods are excluded from Tcl wrapping 1) because the
185 // wrapper gives up on them and 2) because they really shouldn't be
186 // called from a script anyway.
187 //BTX
188
189 // Description:
190 // Execute the SingleMethod (as define by SetSingleMethod) using
191 // this->NumberOfThreads threads.
192 void SingleMethodExecute();
193
194 // Description:
195 // Execute the MultipleMethods (as define by calling SetMultipleMethod
196 // for each of the required this->NumberOfThreads methods) using
197 // this->NumberOfThreads threads.
198 void MultipleMethodExecute();
199
200 // Description:
201 // Set the SingleMethod to f() and the UserData field of the
202 // ThreadInfo that is passed to it will be data.
203 // This method (and all the methods passed to SetMultipleMethod)
204 // must be of type ThreadFunctionType and must take a single argument of
205 // type void *.
206 void SetSingleMethod(ThreadFunctionType, void *data );
207
208 // Description:
209 // Set the MultipleMethod at the given index to f() and the UserData
210 // field of the ThreadInfo that is passed to it will be data.
211 void SetMultipleMethod( int index, ThreadFunctionType, void *data );
212
213 // Description:
214 // Create a new thread for the given function. Return a thread id
215 // which is a number between 0 and IGTL_MAX_THREADS - 1. This id should
216 // be used to kill the thread at a later time.
217 int SpawnThread( ThreadFunctionType, void *data );
218
219 // Description:
220 // Terminate the thread that was created with a SpawnThreadExecute()
221 void TerminateThread( int thread_id );
222
223 // Description:
224 // Get the thread identifier of the calling thread.
225 static MultiThreaderIDType GetCurrentThreadID();
226
227 // Description:
228 // Check whether two thread identifiers refer to the same thread.
229 static int ThreadsEqual(MultiThreaderIDType t1,
230 MultiThreaderIDType t2);
231
232protected:
233 MultiThreader();
234 ~MultiThreader();
235
236 void PrintSelf(std::ostream& os) const override;

238 // The number of threads to use
239 int m_NumberOfThreads;
240
241 // An array of thread info containing a thread id
242 // (0, 1, 2, .. IGTL_MAX_THREADS-1), the thread count, and a pointer
243 // to void so that user data can be passed to each thread
244 ThreadInfo m_ThreadInfoArray[IGTL_MAX_THREADS];
245
246 // The methods
247 ThreadFunctionType m_SingleMethod;

249
250 // Storage of MutexFunctions and ints used to control spawned

254 ThreadProcessIDType m_SpawnedThreadProcessID[IGTL_MAX_THREADS];
255 ThreadInfo m_SpawnedThreadInfoArray[IGTL_MAX_THREADS];
256
257//ETX
258
259 // Internal storage of the data
260 void *m_SingleData;
261 void *m_MultipleData[IGTL_MAX_THREADS];
262
263private:
264 MultiThreader(const MultiThreader&); // Not implemented.
265 void operator=(const MultiThreader&); // Not implemented.
266};
