50int igtlAV1Decoder::DecodeVideoMSGIntoSingleFrame(igtl::VideoMessage* videoMessage, SourcePicture* pDecodedPic)
51{

53 {
54 igtl_int32 iWidth = videoMessage->GetWidth();
55 igtl_int32 iHeight = videoMessage->GetHeight();
56 igtl_uint64 iStreamSize = videoMessage->GetBitStreamSize();
57 pDecodedPic->picWidth = iWidth;
58 pDecodedPic->picHeight = iHeight;
59 pDecodedPic->data[1]= pDecodedPic->data[0] + iWidth*iHeight;
60 pDecodedPic->data[2]= pDecodedPic->data[1] + iWidth*iHeight/4;
61 pDecodedPic->stride[0] = iWidth;
62 pDecodedPic->stride[1] = pDecodedPic->stride[2] = iWidth>>1;
63 pDecodedPic->stride[3] = 0;
64 igtl_uint32 dimensions[2] = {static_cast<igtl_uint32>(iWidth), static_cast<igtl_uint32>(iHeight)};
65 return this->DecodeBitStreamIntoFrame(videoMessage->GetPackFragmentPointer(2), pDecodedPic->data[0], dimensions, iStreamSize);
66 }
67 return -1;

70void igtlAV1Decoder::ComposeByteSteam(igtl_uint8** inputData, int dimension[2], int iStride[3], igtl_uint8 *outputFrame)
71{
72 int plane;
73 int dimensionW [3] = {dimension[0],dimension[0]/2,dimension[0]/2};
74 int dimensionH [3] = {dimension[1],dimension[1]/2,dimension[1]/2};
75 int shift = 0;
76 for (plane = 0; plane < 3; ++plane)
77 {
78 const unsigned char *buf = inputData[plane];
79 const int stride = iStride[plane];
80 int w = dimensionW[plane];
81 int h = dimensionH[plane];
82 int y;
83 for (y = 0; y < h; ++y)
84 {
85 memcpy(outputFrame+shift + w*y, buf, w);
86 buf += stride;
87 }
88 shift += w*h;
89 }
90}

93int igtlAV1Decoder::DecodeBitStreamIntoFrame(unsigned char* bitstream, igtl_uint8* outputFrame, igtl_uint32 dimensions[2], igtl_uint64& iStreamSize)
94{
95 if (!aom_codec_decode(&codec, bitstream, (unsigned int)iStreamSize, NULL))
96 {
97 iter = NULL;
98 if ((outputImage = aom_codec_get_frame(&codec, &iter)) != NULL)
99 {
100 int stride[3] = { outputImage->stride[0], outputImage->stride[1], outputImage->stride[2] };
101 int convertedDimensions[2] = { aom_img_plane_width(outputImage, 0) *
102 ((outputImage->fmt & AOM_IMG_FMT_HIGHBITDEPTH) ? 2 : 1), aom_img_plane_height(outputImage, 0) };
103 ComposeByteSteam(outputImage->planes, convertedDimensions, stride, outputFrame);
104 return 2;
105 }
106 }
107 else
108 {
109 aom_codec_dec_init(&codec, decoder->codec_interface(), NULL, 0);
110 std::cerr << "decode failed" << std::endl;
111 }
112 return -1;
113}
