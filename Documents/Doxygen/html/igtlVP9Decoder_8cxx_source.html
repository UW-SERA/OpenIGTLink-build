50int VP9Decoder::DecodeVideoMSGIntoSingleFrame(igtl::VideoMessage* videoMessage, SourcePicture* pDecodedPic)
51{
52 if(videoMessage->GetBitStreamSize())
53 {
54 igtl_int32 iWidth = videoMessage->GetWidth();
55 igtl_int32 iHeight = videoMessage->GetHeight();
56 igtl_uint64 iStreamSize = videoMessage->GetBitStreamSize();
57 pDecodedPic->picWidth = iWidth;
58 pDecodedPic->picHeight = iHeight;
59 if (!vpx_codec_decode(&codec, videoMessage->GetPackFragmentPointer(2), (unsigned int)iStreamSize, NULL, 0))
60 {
61 iter = NULL;
62 if ((outputImage = vpx_codec_get_frame(&codec, &iter)) != NULL)
63 {
64 int stride[3] = { outputImage->stride[0], outputImage->stride[1], outputImage->stride[2] };
65 int convertedDimensions[2] = { vpx_img_plane_width(outputImage, 0) *
66 ((outputImage->fmt & VPX_IMG_FMT_HIGHBITDEPTH) ? 2 : 1), vpx_img_plane_height(outputImage, 0) };
67 if (outputImage->fmt == VPX_IMG_FMT_I420)
68 {
69 pDecodedPic->data[1]= pDecodedPic->data[0] + iWidth*iHeight;
70 pDecodedPic->data[2]= pDecodedPic->data[1] + iWidth*iHeight/4;
71 pDecodedPic->stride[0] = iWidth;
72 pDecodedPic->stride[1] = pDecodedPic->stride[2] = iWidth>>1;
73 pDecodedPic->stride[3] = 0;
74 ComposeByteSteam(outputImage->planes, convertedDimensions, stride, pDecodedPic->data[0], FormatI420);
75 }
76 else if (outputImage->fmt == VPX_IMG_FMT_I444)
77 {
78 pDecodedPic->data[1]= pDecodedPic->data[0] + iWidth*iHeight;
79 pDecodedPic->data[2]= pDecodedPic->data[1] + iWidth*iHeight;
80 pDecodedPic->stride[0] = iWidth;
81 pDecodedPic->stride[1] = pDecodedPic->stride[2] = iWidth;
82 pDecodedPic->stride[3] = 0;
83 ComposeByteSteam(outputImage->planes, convertedDimensions, stride, pDecodedPic->data[0], FormatI444);
84 }
85 else
86 {
87 return -1;
88 }
89 return 2;
90 }
91 }
92 else
93 {
94 vpx_codec_dec_init(&codec, decoder->codec_interface(), NULL, 0);
95 std::cerr << "decode failed" << std::endl;
96 }
97 return -1;
98 }
99 return -1;
100}

102void VP9Decoder::ComposeByteSteam(igtl_uint8** inputData, int dimension[2], int iStride[3], igtl_uint8 *outputFrame, VideoFormatType format)
103{
104 int plane;
105 int dimensionW [3];
106 int dimensionH [3];
107 if(format==FormatI420)
108 {
109 dimensionW [0] = dimension[0];
110 dimensionW [1] = dimension[0]/2;
111 dimensionW [2] = dimension[0]/2;
112 dimensionH [0] = dimension[1];
113 dimensionH [1] = dimension[1]/2;
114 dimensionH [2] = dimension[1]/2;
115 }
116 else if(format==FormatI444)
117 {
118 dimensionW [0] = dimension[0];
119 dimensionW [1] = dimension[0];
120 dimensionW [2] = dimension[0];
121 dimensionH [0] = dimension[0];
122 dimensionH [1] = dimension[1];
123 dimensionH [2] = dimension[1];
124 }
125 else
126 {
127 return;
128 }
129 int shift = 0;
130 for (plane = 0; plane < 3; ++plane)
131 {
132 const unsigned char *buf = inputData[plane];
133 const int stride = iStride[plane];
134 int w = dimensionW[plane];
135 int h = dimensionH[plane];
136 int y;
137 for (y = 0; y < h; ++y)
138 {
139 memcpy(outputFrame+shift + w*y, buf, w);
140 buf += stride;
141 }
142 shift += w*h;
143 }
144}

157 if (outputImage->fmt == VPX_IMG_FMT_I420)
