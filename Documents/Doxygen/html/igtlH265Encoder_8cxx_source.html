132int H265Encoder::SetSpeed(int speed)
133{
134 speed = speed>=0?speed:0;
135 speed = speed<=9?speed:9;
136 this->codecSpeed = speed;
137 x265_param* previousSetting = x265_param_alloc();
138
139 this->CopySettingToAnother(this->sSvcParam, previousSetting);
140 x265_param_default_preset(this->sSvcParam,ToString(9-speed).c_str(),"zerolatency"); // In OpenIGTLink, lower speed value corresponding to slower coding speed. In x265, the speed setting is reversed.
141 this->CopySettingToAnother(previousSetting, this->sSvcParam);
142 x265_param_free(previousSetting);
143 if (x265_encoder_reconfig(this->pSVCEncoder, sSvcParam)<0)
144 {
145 fprintf (stderr, "Set speed mode failed.\n");
146 return -1;
147 }
148 return 0;
149}

197int H265Encoder::ConvertToLocalImageFormat(SourcePicture* pSrcPic)
198{
199 if(pSrcPic->colorFormat==FormatI420)
200 {
201 if((pSrcPic->picWidth*pSrcPic->picHeight*3/2 != this->H265SrcPicture->framesize || pSrcPic->picHeight != this->H265SrcPicture->height))
202 {
203 this->SetPicWidthAndHeight(pSrcPic->picWidth, pSrcPic->picHeight);
204 x265_picture_init(this->sSvcParam, H265SrcPicture);
205 }
206 for(int i = 0; i < 3; i++)
207 {
208 H265SrcPicture->stride[i] = pSrcPic->stride[i];
209 H265SrcPicture->planes[i] = pSrcPic->data[i];
210 }
211 return 1;
212 }
213 return -1;// image format not supported
214}

217int H265Encoder::EncodeSingleFrameIntoVideoMSG(SourcePicture* pSrcPic, igtl::VideoMessage* videoMessage, bool isGrayImage)
218{
219 int encodeRet = -1;
220 int iSourceWidth = pSrcPic->picWidth;
221 int iSourceHeight = pSrcPic->picHeight;
222 if (iSourceWidth != this->sSvcParam->sourceWidth || iSourceHeight != this->sSvcParam->sourceHeight)
223 {
224 this->SetPicWidthAndHeight(iSourceWidth, iSourceHeight);
225 this->InitializeEncoder();
226 }
227 if (this->initializationDone == true)
228 {
229 pSrcPic->stride[0] = iSourceWidth;
230 pSrcPic->stride[1] = pSrcPic->stride[2] = pSrcPic->stride[0] >> 1;
231 this->ConvertToLocalImageFormat(pSrcPic);
232
233 igtl_uint32 iNal = 0;
234 encodeRet =x265_encoder_encode(this->pSVCEncoder,&pNals,&iNal,this->H265SrcPicture,NULL);
235 //encodeRet =x265_encoder_encode(this->pSVCEncoder,&pNals,&iNal,NULL,NULL);
236 if (encodeRet>=1)
237 {
238 igtl_uint64 totalBitStreamSize = 0;
239 for(int j=0;j<iNal;j++){
240 totalBitStreamSize += pNals[j].sizeBytes;
241 }
242 if((pNals[0].type & X265_TYPE_IDR) != 0)
243 {
244 encodedFrameType = FrameTypeKey;
245 }
246 else
247 {
248 // To do, assign other frame type too.
249 encodedFrameType = FrameTypeUnKnown;
250 }
251 videoMessage->SetFrameType(encodedFrameType);
252 videoMessage->SetBitStreamSize(totalBitStreamSize);
253 videoMessage->AllocateScalars();
254 videoMessage->SetCodecType(IGTL_VIDEO_CODEC_NAME_X265);
255 videoMessage->SetEndian(igtl_is_little_endian()==true?2:1); //little endian is 2 big endian is 1
256 videoMessage->SetWidth(pSrcPic->picWidth);
257 videoMessage->SetHeight(pSrcPic->picHeight);
258 encodedFrameType = pNals[iNal-1].type;
259 if (isGrayImage)
260 {
261 encodedFrameType = pNals[iNal-1].type<<8;
262 }
263 videoMessage->SetFrameType(encodedFrameType);
264 int nalSize = 0;
265 FILE* testFile = fopen("Test.265", "a");
266 for(int j=0;j<iNal;j++){
267 memcpy(&(videoMessage->GetPackFragmentPointer(2)[nalSize]), pNals[j].payload, pNals[j].sizeBytes);
268 nalSize += pNals[j].sizeBytes;
269 //fwrite(pNals[j].payload, 1 , pNals[j].sizeBytes, testFile);
270 }
271 fclose(testFile);
272 videoMessage->Pack();
273 return 0;
274 }
275 }
276 return -1;
277}
