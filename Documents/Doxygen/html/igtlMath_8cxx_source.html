52void IGTLCommon_EXPORT QuaternionToMatrix(float* q, Matrix4x4& m)
53{
54
55 // normalize
56 float mod = sqrt(q[0]*q[0]+q[1]*q[1]+q[2]*q[2]+q[3]*q[3]);
57
58 // convert to the matrix
59 const float x = q[0] / mod;
60 const float y = q[1] / mod;
61 const float z = q[2] / mod;
62 const float w = q[3] / mod;
63
64 const float xx = x * x * 2.0;
65 const float xy = x * y * 2.0;
66 const float xz = x * z * 2.0;
67 const float xw = x * w * 2.0;
68 const float yy = y * y * 2.0;
69 const float yz = y * z * 2.0;
70 const float yw = y * w * 2.0;
71 const float zz = z * z * 2.0;
72 const float zw = z * w * 2.0;
73
74 m[0][0] = 1.0 - (yy + zz);
75 m[1][0] = xy + zw;
76 m[2][0] = xz - yw;
77
78 m[0][1] = xy - zw;
79 m[1][1] = 1.0 - (xx + zz);
80 m[2][1] = yz + xw;
81
82 m[0][2] = xz + yw;
83 m[1][2] = yz - xw;
84 m[2][2] = 1.0 - (xx + yy);
85
86 m[3][0] = 0.0;
87 m[3][1] = 0.0;
88 m[3][2] = 0.0;
89 m[3][3] = 1.0;
90
91 m[0][3] = 0.0;
92 m[1][3] = 0.0;
93 m[2][3] = 0.0;
94
95}

98void IGTLCommon_EXPORT MatrixToQuaternion(Matrix4x4& m, float* q)
99{
100 float trace = m[0][0] + m[1][1] + m[2][2];
101
102 if ( trace > 0.0 )
103 {
104
105 float s = 0.5f / sqrt(trace + 1.0f);
106
107 q[3] = 0.25f / s;
108 q[0] = ( m[2][1] - m[1][2] ) * s;
109 q[1] = ( m[0][2] - m[2][0] ) * s;
110 q[2] = ( m[1][0] - m[0][1] ) * s;
111
112 }
113 else
114 {
115
116 if ( m[0][0] > m[1][1] && m[0][0] > m[2][2] )
117 {
118
119 float s = 2.0f * sqrt( 1.0f + m[0][0] - m[1][1] - m[2][2]);
120
121 q[3] = (m[2][1] - m[1][2] ) / s;
122 q[0] = 0.25f * s;
123 q[1] = (m[0][1] + m[1][0] ) / s;
124 q[2] = (m[0][2] + m[2][0] ) / s;
125
126 }
127 else if (m[1][1] > m[2][2])
128 {
129
130 float s = 2.0f * sqrt( 1.0f + m[1][1] - m[0][0] - m[2][2]);
131
132 q[3] = (m[0][2] - m[2][0] ) / s;
133 q[0] = (m[0][1] + m[1][0] ) / s;
134 q[1] = 0.25f * s;
135 q[2] = (m[1][2] + m[2][1] ) / s;
136
137 }
138 else
139 {
140
141 float s = 2.0f * sqrt( 1.0f + m[2][2] - m[0][0] - m[1][1] );
142
143 q[3] = (m[1][0] - m[0][1] ) / s;
144 q[0] = (m[0][2] + m[2][0] ) / s;
145 q[1] = (m[1][2] + m[2][1] ) / s;
146 q[2] = 0.25f * s;
147
148 }
149 }
150}

154void IGTLCommon_EXPORT Cross(float *a, float *b, float *c)
155{
156 a[0] = b[1]*c[2] - c[1]*b[2];
157 a[1] = c[0]*b[2] - b[0]*c[2];
158 a[2] = b[0]*c[1] - c[0]*b[1];
159}
