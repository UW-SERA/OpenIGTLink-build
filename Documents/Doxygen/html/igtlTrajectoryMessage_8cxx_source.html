271int TrajectoryMessage::PackContent()
272{
273 // Allocate buffer
274 AllocateBuffer();
275
276 igtl_trajectory_element* element = NULL;
277 element = (igtl_trajectory_element*)(this->m_Content);
278
279 igtl_trajectory_element * elementHolder = element;
280 std::vector<TrajectoryElement::Pointer>::iterator iter;
281 for (iter = this->m_TrajectoryList.begin(); iter != this->m_TrajectoryList.end(); iter ++)
282 {
283 strncpy((char*)element->name, (*iter)->GetName(), IGTL_TRAJECTORY_LEN_NAME);
284 strncpy((char*)element->group_name, (*iter)->GetGroupName(), IGTL_TRAJECTORY_LEN_GROUP_NAME);
285
286 element->type = (*iter)->GetType();
287 element->reserved = 0;
288 igtlUint8 rgba[4];
289 (*iter)->GetRGBA(rgba);
290 element->rgba[0] = rgba[0];
291 element->rgba[1] = rgba[1];
292 element->rgba[2] = rgba[2];
293 element->rgba[3] = rgba[3];
294
295 igtlFloat32 position[3];
296 (*iter)->GetEntryPosition(position);
297 element->entry_pos[0] = position[0];
298 element->entry_pos[1] = position[1];
299 element->entry_pos[2] = position[2];
300
301 (*iter)->GetTargetPosition(position);
302 element->target_pos[0] = position[0];
303 element->target_pos[1] = position[1];
304 element->target_pos[2] = position[2];
305
306 element->radius = (*iter)->GetRadius();
307
308 strncpy((char*)element->owner_name, (*iter)->GetOwner(), IGTL_TRAJECTORY_LEN_OWNER);
309
310 element ++;
311 }
312
313 igtl_trajectory_convert_byte_order(elementHolder, this->m_TrajectoryList.size());
314
315 return 1;
316}

319int TrajectoryMessage::UnpackContent()
320{
321 this->m_TrajectoryList.clear();
322
323 igtl_trajectory_element* element = NULL;
324 int nElement = 0;
325#if OpenIGTLink_HEADER_VERSION >= 2
326 element = (igtl_trajectory_element*)(this->m_Content);
327 bool isUnpacked(true);
328 nElement = igtl_trajectory_get_data_n(CalculateReceiveContentSize(isUnpacked));
329#elif OpenIGTLink_PROTOCOL_VERSION <=2
330 element = (igtl_trajectory_element*)this->m_Body;
331 nElement = igtl_trajectory_get_data_n(this->m_BodySizeToRead);
332#endif
333
334 igtl_trajectory_convert_byte_order(element, nElement);
335
336 char strbuf[128];
337 for (int i = 0; i < nElement; i ++)
338 {
339 TrajectoryElement::Pointer elemClass = TrajectoryElement::New();
340
341 // Add '\n' at the end of each string
342 // (necessary for a case, where a string reaches the maximum length.)
343 strbuf[IGTL_TRAJECTORY_LEN_NAME] = '\n';
344 strncpy(strbuf, (char*)element->name, IGTL_TRAJECTORY_LEN_NAME);
345 elemClass->SetName((const char*)strbuf);
346
347 strbuf[IGTL_TRAJECTORY_LEN_GROUP_NAME] = '\n';
348 strncpy(strbuf, (char*)element->group_name, IGTL_TRAJECTORY_LEN_GROUP_NAME);
349 elemClass->SetGroupName(strbuf);
350
351 elemClass->SetType(element->type);
352
353 elemClass->SetRGBA(element->rgba);
354 elemClass->SetEntryPosition(element->entry_pos);
355 elemClass->SetTargetPosition(element->target_pos);
356 elemClass->SetRadius(element->radius);
357
358 strbuf[IGTL_TRAJECTORY_LEN_OWNER] = '\n';
359 strncpy(strbuf, (char*)element->owner_name, IGTL_TRAJECTORY_LEN_OWNER);
360 elemClass->SetOwner(strbuf);
361
362 this->m_TrajectoryList.push_back(elemClass);
363
364 element ++;
365 }
366
367 return 1;
368}
