31/* Calculate number of bytes / element */
32igtl_uint32 igtl_ndarray_get_nbyte(int type)
33{
34 igtl_uint32 bytes;
35
36 switch (type)
37 {
38 case IGTL_NDARRAY_STYPE_TYPE_INT8:
39 case IGTL_NDARRAY_STYPE_TYPE_UINT8:
40 bytes = 1;
41 break;
42 case IGTL_NDARRAY_STYPE_TYPE_INT16:
43 case IGTL_NDARRAY_STYPE_TYPE_UINT16:
44 bytes = 2;
45 break;
46 case IGTL_NDARRAY_STYPE_TYPE_INT32:
47 case IGTL_NDARRAY_STYPE_TYPE_UINT32:
48 case IGTL_NDARRAY_STYPE_TYPE_FLOAT32:
49 bytes = 4;
50 break;
51 case IGTL_NDARRAY_STYPE_TYPE_FLOAT64:
52 bytes = 8;
53 break;
54 case IGTL_NDARRAY_STYPE_TYPE_COMPLEX:
55 bytes = 16;
56 break;
57 default:
58 bytes = 0;
59 break;
60 }
61
62 return bytes;
63}

66int igtl_export igtl_ndarray_alloc_info(igtl_ndarray_info * info, const igtl_uint16 * size)
67{
68 int i;
69 igtl_uint64 len;
70
71 if (info->size == NULL && info->array == NULL)
72 {
73 info->size = malloc(sizeof(igtl_uint16) * (igtl_uint16) info->dim);
74 if (info->size == NULL)
75 {
76 return 0;
77 }
78 len = 1;
79
80 for (i = 0; i < info->dim; i ++)
81 {
82 info->size[i] = size[i];
83 len *= size[i];
84 }
85
86 info->array = malloc((size_t)(igtl_ndarray_get_nbyte(info->type) * len));
87
88 if (info->array == NULL)
89 {
90 free(info->size);
91 return 0;
92 }
93
94 return 1;
95 }
96 else
97 {
98 return 0;
99 }
100}

118int igtl_export igtl_ndarray_unpack(int type, void * byte_array, igtl_ndarray_info * info, igtl_uint64 pack_size)
119{
120 char * ptr;
121 igtl_uint16 dim;
122 igtl_uint16 * size;
123 igtl_uint16 i;
124 igtl_uint64 len;
125
126 igtl_uint16 * ptr16_src;
127 igtl_uint16 * ptr16_src_end;
128 igtl_uint16 * ptr16_dst;
129
130 igtl_uint32 * ptr32_src;
131 igtl_uint32 * ptr32_src_end;
132 igtl_uint32 * ptr32_dst;
133
134 igtl_uint64 * ptr64_src;
135 igtl_uint64 * ptr64_src_end;
136 igtl_uint64 * ptr64_dst;
137
138
139 if (byte_array == NULL || info == NULL)
140 {
141 return 0;
142 }
143
144 igtl_ndarray_init_info(info);
145 ptr = (char *) byte_array;
146
147 /*** Type field ***/
148 info->type = * (igtl_uint8 *) ptr;
149 ptr ++;
150
151 /*** Dimension field ***/
152 info->dim = * (igtl_uint8 *) ptr;
153 ptr ++;
154
155 /*** Size array field ***/
156 size = (igtl_uint16 *) ptr;
157 dim = info->dim;
158 if (igtl_is_little_endian())
159 {
160 /* Change byte order -- this overwrites memory area for the pack !!*/
161 for (i = 0; i < dim; i ++)
162 {
163 size[i] = BYTE_SWAP_INT16(size[i]);
164 }
165 }
166
167 igtl_ndarray_alloc_info(info, size);
168
169 memcpy(info->size, size, sizeof(igtl_uint16) * dim);
170 if (igtl_is_little_endian())
171 {
172 /* Resotore the overwritten memory area */
173 /* Don't use size[] array after this !! */
174 for (i = 0; i < dim; i ++)
175 {
176 size[i] = BYTE_SWAP_INT16(size[i]);
177 }
178 }
179 ptr += sizeof(igtl_uint16) * dim;
180
181 /*** N-D array field ***/
182 /* Calculate number of elements in N-D array */
183 len = 1;
184 for (i = 0; i < dim; i ++)
185 {
186 len *= info->size[i];
187 }
188
189 /* Copy array */
190 if (igtl_ndarray_get_nbyte(info->type) == 1 || !igtl_is_little_endian())
191 {
192 /* If single-byte data type is used or the program runs on a big-endian machine,
193 just copy the array from the pack to the strucutre */
194 memcpy(info->array, ptr, (size_t)(len * igtl_ndarray_get_nbyte(info->type)));
195 }
196 else if (igtl_ndarray_get_nbyte(info->type) == 2) /* 16-bit */
197 {
198 ptr16_src = (igtl_uint16 *) ptr;
199 ptr16_src_end = ptr16_src + len;
200 ptr16_dst = (igtl_uint16 *) info->array;
201 while (ptr16_src < ptr16_src_end)
202 {
203 *ptr16_dst = BYTE_SWAP_INT16(*ptr16_src);
204 ptr16_dst ++;
205 ptr16_src ++;
206 }
207 }
208 else if (igtl_ndarray_get_nbyte(info->type) == 4) /* 32-bit */
209 {
210 ptr32_src = (igtl_uint32 *) ptr;
211 ptr32_src_end = ptr32_src + len;
212 ptr32_dst = (igtl_uint32 *) info->array;
213 while (ptr32_src < ptr32_src_end)
214 {
215 *ptr32_dst = BYTE_SWAP_INT32(*ptr32_src);
216 ptr32_dst ++;
217 ptr32_src ++;
218 }
219 }
220 else /* 64-bit or Complex type */
221 {
222 ptr64_src = (igtl_uint64 *) ptr;
223 /* Adding number of elements to the pointer -- 64-bit: len * 1; Complex: len * 2*/
224 ptr64_src_end = ptr64_src + len * igtl_ndarray_get_nbyte(info->type)/8;
225 ptr64_dst = (igtl_uint64 *) info->array;
226 while (ptr64_src < ptr64_src_end)
227 {
228 *ptr64_dst = BYTE_SWAP_INT64(*ptr64_src);
229 ptr64_dst ++;
230 ptr64_src ++;
231 }
232 }
233
234 /* TODO: check if the pack size is valid */
235
236 return 1;
237
238}

352igtl_uint64 igtl_export igtl_ndarray_get_size(igtl_ndarray_info * info, int type)
353{
354 igtl_uint64 len;
355 igtl_uint64 data_size;
356 igtl_uint16 i;
357 igtl_uint16 dim;
358
359 dim = info->dim;
360 len = 1;
361 for (i = 0; i < dim; i ++)
362 {
363 len *= (igtl_uint64)info->size[i];
364 }
365
366 data_size = sizeof(igtl_uint8) * 2 + sizeof(igtl_uint16) * (igtl_uint64) dim
367 + len * igtl_ndarray_get_nbyte(info->type);
368
369 return data_size;
370
371}
