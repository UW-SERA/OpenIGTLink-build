223static void* ThreadFunctionServer(void* ptr)
224{
225 //------------------------------------------------------------
226 // Get thread information
227 igtl::MultiThreader::ThreadInfo* info =
228 static_cast<igtl::MultiThreader::ThreadInfo*>(ptr);
229
230 VideoStreamIGTLinkServer* parentObj = static_cast<VideoStreamIGTLinkServer*>(info->UserData);
231 int r = -1;
232 if (parentObj->serverSocket.IsNotNull())
233 {
234 parentObj->serverSocket->CloseSocket();
235 }
236 r = parentObj->serverSocket->CreateServer(parentObj->serverPortNumber);
237 if (r < 0)
238 {
239 std::cerr << "Cannot create a server socket." << std::endl;
240 exit(0);
241 }
242 parentObj->socket = igtl::Socket::New();
243
244 while (1)
245 {
246 //------------------------------------------------------------
247 // Waiting for Connection
248 parentObj->serverConnected = false;
249 parentObj->socket = parentObj->serverSocket->WaitForConnection(1000);
250
251 if (parentObj->socket.IsNotNull()) // if client connected
252 {
253 std::cerr << "A client is connected." << std::endl;
254 igtl::MessageHeader::Pointer headerMsg;
255 headerMsg = igtl::MessageHeader::New();
256 if (!parentObj->waitSTTCommand)
257 {
258 // Create a message buffer to receive header
259 strncpy(parentObj->codecName, "H264", IGTL_VIDEO_CODEC_NAME_SIZE);
260 parentObj->serverIsSet = false;
261 parentObj->serverConnected = true;
262 parentObj->conditionVar->Signal();
263 while (parentObj->serverConnected)
264 {
265 headerMsg->InitPack();
266 bool timeout(false);
267 int rs = parentObj->socket->Receive(headerMsg->GetPackPointer(), headerMsg->GetPackSize(), timeout);
268 if (rs == 0)
269 {
270 std::cerr << "Disconnecting the client." << std::endl;
271 break;
272 }
273 else
274 {
275 igtl::Sleep(10);
276 }
277 }
278 }
279 else if (parentObj->waitSTTCommand)
280 {
281 //------------------------------------------------------------
282 // loop
283 for (;;)
284 {
285 // Initialize receive buffer
286 headerMsg->InitPack();
287 bool timeout(false);
288 // Receive generic header from the socket
289 int rs = parentObj->socket->Receive(headerMsg->GetPackPointer(), headerMsg->GetPackSize(), timeout);
290 if (rs == 0)
291 {
292 std::cerr << "Disconnecting the client." << std::endl;
293 break;
294 }
295 if (rs != headerMsg->GetPackSize())
296 {
297 continue;
298 }
299
300 // Deserialize the header
301 headerMsg->Unpack();
302
303 // Check data type and receive data body
304 if (strcmp(headerMsg->GetDeviceType(), "STP_VIDEO") == 0)
305 {
306 parentObj->glock->Lock();
307 parentObj->socket->Skip(headerMsg->GetBodySizeToRead(), 0);
308 std::cerr << "Received a STP_VIDEO message." << std::endl;
309 std::cerr << "Disconnecting the client." << std::endl;
310 parentObj->serverIsSet = false;
311 parentObj->serverConnected = false;
312 if (parentObj->socket.IsNotNull())
313 {
314 parentObj->socket->CloseSocket();
315 }
316 parentObj->glock->Unlock();
317 parentObj->socket = NULL; // VERY IMPORTANT. Completely remove the instance.
318 break;
319 }
320 else if (strcmp(headerMsg->GetDeviceType(), "STT_VIDEO") == 0)
321 {
322 std::cerr << "Received a STT_VIDEO message." << std::endl;
323
324 igtl::StartVideoMessage::Pointer startVideoMsg;
325 startVideoMsg = igtl::StartVideoMessage::New();
326 startVideoMsg->SetMessageHeader(headerMsg);
327 startVideoMsg->AllocatePack();
328 parentObj->glock->Lock();
329 bool timeout(false);
330 parentObj->socket->Receive(startVideoMsg->GetPackBodyPointer(), startVideoMsg->GetPackBodySize(), timeout);
331 parentObj->glock->Unlock();
332 int c = startVideoMsg->Unpack(1);
333 if (c & igtl::MessageHeader::UNPACK_BODY && strcmp(startVideoMsg->GetCodecType().c_str(), "H264")) // if CRC check is OK
334 {
335 parentObj->interval = startVideoMsg->GetTimeInterval();
336 strncpy(parentObj->codecName, startVideoMsg->GetCodecType().c_str(), IGTL_VIDEO_CODEC_NAME_SIZE);
337 parentObj->serverConnected = true;
338 parentObj->conditionVar->Signal();
339 }
340 }
341 else
342 {
343 std::cerr << "Receiving : " << headerMsg->GetDeviceType() << std::endl;
344 parentObj->glock->Lock();
345 parentObj->socket->Skip(headerMsg->GetBodySizeToRead(), 0);
346 parentObj->glock->Unlock();
347 }
348 igtl::Sleep(100);
349 }
350 }
351 }
352 }
353
354 //------------------------------------------------------------
355 // Close connection (The example code never reaches to this section ...)
356 parentObj->serverSocket->CloseSocket();
357 parentObj->glock->Lock();
358 if (parentObj->socket.IsNotNull())
359 {
360 parentObj->socket->CloseSocket();
361 }
362 parentObj->glock->Unlock();
363 parentObj->socket = NULL; // VERY IMPORTANT. Completely remove the instance.
364 parentObj->serverSocket = NULL;
365}

377int VideoStreamIGTLinkServer::SetupServer()
378{
379 //------------------------------------------------------------
380 int iRet = 0;
381 if (this->videoEncoder == NULL)
382 iRet = 1;
383
384 this->iTotalFrameToEncode = 0;
385
386 // Preparing encoding process
387
388 // Inactive with sink with output file handler
389 FILE* pFpBs = NULL;
390 if (pSrcPic == NULL) {
391 iRet = 1;
392 goto INSIDE_MEM_FREE;
393 }
394 //fill default pSrcPic
395 pSrcPic->colorFormat = FormatI420;//videoFormatI420;
396 pSrcPic->timeStamp = 0;
397
398 // if configure file exit, reading configure file firstly
399 cRdCfg.OpenFile(this->augments.c_str());// to do get the first augments from this->augments.
400 if (cRdCfg.ExistFile())
401 {
402 cRdCfg.OpenFile(this->augments.c_str());// reset the file read pointer to the beginning.
403 iRet = ParseConfigForServer();
404 if (iRet) {
405 fprintf (stderr, "parse server parameter config file failed.\n");
406 iRet = 1;
407 goto INSIDE_MEM_FREE;
408 }
409 }
410 else
411 {
412 fprintf (stderr, "Specified file: %s not exist, maybe invalid path or parameter settting.\n",
413 cRdCfg.GetFileName().c_str());
414 iRet = 1;
415 goto INSIDE_MEM_FREE;
416 }
417
418 //finish reading the configurations
419 igtl_uint32 iSourceWidth, iSourceHeight, kiPicResSize;
420 iSourceWidth = pSrcPic->picWidth;
421 iSourceHeight = pSrcPic->picHeight;
422 kiPicResSize = iSourceWidth * iSourceHeight * 3 >> 1;
423
424 //update pSrcPic
425 pSrcPic->stride[0] = iSourceWidth;
426 pSrcPic->stride[1] = pSrcPic->stride[2] = pSrcPic->stride[0] >> 1;
427 pSrcPic->stride[3] = 0;
428
429 this->serverIsSet = true;
430 return true;
431INSIDE_MEM_FREE:
432 if (pFpBs) {
433 fclose (pFpBs);
434 pFpBs = NULL;
435 }
436 if (pSrcPic) {
437 delete pSrcPic;
438 pSrcPic = NULL;
439 }
440 this->serverConnected = false;
441 this->serverIsSet = false;
442 return iRet;
443}

450static void* ThreadFunctionReadFrameFromFile(void* ptr)
451{
452 // Get thread information
453 igtl::MultiThreader::ThreadInfo* info =
454 static_cast<igtl::MultiThreader::ThreadInfo*>(ptr);
455 serverPointer parentObj = *(static_cast<serverPointer*>(info->UserData));
456 int kiPicResSize = parentObj.server->pSrcPic->picWidth*parentObj.server->pSrcPic->picHeight*3>>1;
457 igtl_int32 iFrameNumInFile = -1;
458 FILE* pFileYUV = NULL;
459 pFileYUV = fopen (parentObj.server->strSeqFile.c_str(), "rb");
460 if (pFileYUV != NULL) {
461#if defined(_WIN32) || defined(_WIN64)
462#if _MSC_VER >= 1400
463 if (!_fseeki64 (pFileYUV, 0, SEEK_END)) {
464 igtl_int64 i_size = _ftelli64 (pFileYUV);
465 _fseeki64 (pFileYUV, 0, SEEK_SET);
466 iFrameNumInFile = max ((igtl_int32) (i_size / kiPicResSize), iFrameNumInFile);
467 }
468#else
469 if (!fseek (pFileYUV, 0, SEEK_END)) {
470 igtl_int64 i_size = ftell (pFileYUV);
471 fseek (pFileYUV, 0, SEEK_SET);
472 iFrameNumInFile = max ((igtl_int32) (i_size / kiPicResSize), iFrameNumInFile);
473 }
474#endif
475#else
476 if (!fseeko (pFileYUV, 0, SEEK_END)) {
477 igtl_int64 i_size = ftello (pFileYUV);
478 fseeko (pFileYUV, 0, SEEK_SET);
479 iFrameNumInFile = std::max((igtl_int32) (i_size / kiPicResSize), iFrameNumInFile);
480 }
481#endif
482 igtl_int32 iFrameIdx = 0;
483 while((igtl_int32)parentObj.server->iTotalFrameToEncode > 0)
484 {
485 iFrameIdx = 0;
486#if defined(_WIN32) || defined(_WIN64)
487#if _MSC_VER >= 1400
488 _fseeki64(pFileYUV, 0, SEEK_SET);
489#else
490 fseek(pFileYUV, 0, SEEK_SET);
491#endif
492#else
493 fseeko(pFileYUV, 0, SEEK_SET);
494#endif
495 while (iFrameIdx < iFrameNumInFile && iFrameIdx < parentObj.server->iTotalFrameToEncode) {
496 bool bCanBeRead = false;
497 igtl_uint8* pYUV = new igtl_uint8[kiPicResSize];
498 bCanBeRead = (fread (pYUV, 1, kiPicResSize, pFileYUV) == kiPicResSize);
499 parentObj.server->glockInFrame->Lock();
500 if (parentObj.server->incommingFrames.size()>3)
501 {
502 std::map<igtlUint32, igtl_uint8*>::iterator it = parentObj.server->incommingFrames.begin();
503 delete it->second;
504 it->second = NULL;
505 parentObj.server->incommingFrames.erase(it);
506 }
507 parentObj.server->incommingFrames.insert(std::pair<igtl_uint32, igtl_uint8*>(0, pYUV));
508 parentObj.server->glockInFrame->Unlock();
509 parentObj.server->iTotalFrameToEncode = parentObj.server->iTotalFrameToEncode - 1; // excluding skipped frame time
510 igtl::Sleep(parentObj.server->interval);
511 if (!bCanBeRead)
512 break;
513 }
514 }
515 if (pFileYUV)
516 {
517 fclose(pFileYUV);
518 pFileYUV = NULL;
519 }
520 } else {
521 fprintf (stderr, "Unable to open source sequence file (%s), check corresponding path!\n",
522 parentObj.server->strSeqFile.c_str());
523 }
524
525 return NULL;
526}

529static void* ThreadFunctionSendPacket(void* ptr)
530{
531 // Get thread information
532 igtl::MultiThreader::ThreadInfo* info =
533 static_cast<igtl::MultiThreader::ThreadInfo*>(ptr);
534 serverPointer parentObj = *(static_cast<serverPointer*>(info->UserData));
535 while(1)
536 {
537 parentObj.server->glock->Lock();
538 unsigned int frameNum = parentObj.server->encodedFrames.size();
539 parentObj.server->glock->Unlock();
540 if(frameNum)
541 {
542 parentObj.server->glock->Lock();
543 VideoStreamIGTLinkServer::encodedFrame* frameCopy = new VideoStreamIGTLinkServer::encodedFrame();
544 std::map<igtlUint32, VideoStreamIGTLinkServer::encodedFrame*>::iterator it = parentObj.server->encodedFrames.begin();
545 frameCopy->messageDataLength = it->second->messageDataLength;
546 memcpy(frameCopy->messagePackPointer,it->second->messagePackPointer,it->second->messageDataLength);
547 delete it->second;
548 it->second = NULL;
549 parentObj.server->encodedFrames.erase(it);
550 if (parentObj.server->transportMethod == VideoStreamIGTLinkServer::UseUDP)
551 {
552 parentObj.server->rtpWrapper->WrapMessageAndSend(parentObj.server->serverUDPSocket, frameCopy->messagePackPointer, frameCopy->messageDataLength);
553 }
554 else if(parentObj.server->transportMethod == VideoStreamIGTLinkServer::UseTCP)
555 {
556 if(parentObj.server->socket)
557 {
558 parentObj.server->socket->Send(frameCopy->messagePackPointer, frameCopy->messageDataLength);
559 }
560 }
561 parentObj.server->glock->Unlock();
562 delete frameCopy;
563 }
564 igtl::Sleep(1);
565 }
566}

598void VideoStreamIGTLinkServer::SendOriginalData()
599{
600 int kiPicResSize = pSrcPic->picWidth*pSrcPic->picHeight*3>>1;
601 igtl_uint8* pYUV = new igtl_uint8[kiPicResSize];
602 static int messageID = -1;
603 while(this->iTotalFrameToEncode)
604 {
605 this->glockInFrame->Lock();
606 int frameNum = this->incommingFrames.size();
607 this->glockInFrame->Unlock();
608 while(frameNum)
609 {
610 this->glockInFrame->Lock();
611 std::map<igtlUint32, igtl_uint8*>::iterator it = this->incommingFrames.begin();
612 memcpy(pYUV,it->second,kiPicResSize);
613 delete it->second;
614 it->second = NULL;
615 this->incommingFrames.erase(it);
616 frameNum = this->incommingFrames.size();
617 this->glockInFrame->Unlock();
618 messageID ++;
619 igtl::VideoMessage::Pointer videoMsg;
620 videoMsg = igtl::VideoMessage::New();
621 videoMsg->SetHeaderVersion(IGTL_HEADER_VERSION_2);
622 videoMsg->SetDeviceName(this->deviceName.c_str());
623 videoMsg->SetBitStreamSize(kiPicResSize);
624 videoMsg->AllocateScalars();
625 int endian = (igtl_is_little_endian() == 1 ? IGTL_VIDEO_ENDIAN_LITTLE : IGTL_VIDEO_ENDIAN_BIG);
626 videoMsg->SetEndian(endian); //little endian is 2 big endian is 1
627 videoMsg->SetWidth(pSrcPic->picWidth);
628 videoMsg->SetHeight(pSrcPic->picHeight);
629 memcpy(videoMsg->GetPackFragmentPointer(2), pYUV, kiPicResSize);
630 ServerTimer->GetTime();
631 videoMsg->SetTimeStamp(ServerTimer);
632 videoMsg->Pack();
633 encodedFrame* frame = new encodedFrame();
634 memcpy(frame->messagePackPointer, videoMsg->GetPackPointer(), videoMsg->GetBufferSize());
635 frame->messageDataLength = videoMsg->GetBufferSize();
636 this->glock->Lock();
637 this->CheckEncodedFrameMap();
638 this->encodedFrames.insert(std::pair<igtlUint32, VideoStreamIGTLinkServer::encodedFrame*>(messageID, frame));
639 this->glock->Unlock();
640 }
641 }
642 delete[] pYUV;
643 pYUV = NULL;
644}

646int VideoStreamIGTLinkServer::EncodeFile(void)
647{
648 if(!this->videoEncoder)
649 {
650 return -1;
651 }
652 if(!this->serverIsSet)
653 {
654 this->SetupServer();
655 }
656 igtl_int64 iStart = 0, iTotal = 0;
657
658 int picSize = pSrcPic->picWidth*pSrcPic->picHeight;
659
660 igtl_int32 iFrameIdx = 0;
661 this->totalCompressedDataSize = 0;
662 int iActualFrameEncodedCount = 0;
663 while(this->iTotalFrameToEncode)
664 {
665 this->glockInFrame->Lock();
666 int frameNum = this->incommingFrames.size();
667 this->glockInFrame->Unlock();
668 while(frameNum)
669 {
670 // To encoder this frame
671 this->glockInFrame->Lock();
672 std::map<igtlUint32, igtl_uint8*>::iterator it = this->incommingFrames.begin();
673 memcpy(this->pSrcPic->data[0],it->second,picSize*3/2);
674 delete it->second;
675 it->second = NULL;
676 this->incommingFrames.erase(it);
677 frameNum = this->incommingFrames.size();
678 this->glockInFrame->Unlock();
679 this->ServerTimer->GetTime();
680 this->encodeStartTime = this->ServerTimer->GetTimeStampInNanoseconds();
681 iStart = this->encodeStartTime;
682 igtl::VideoMessage::Pointer videoMsg = igtl::VideoMessage::New();
683 videoMsg->SetHeaderVersion(IGTL_HEADER_VERSION_2);
684 videoMsg->SetDeviceName(this->deviceName.c_str());
685 int iEncFrames = this->videoEncoder->EncodeSingleFrameIntoVideoMSG(this->pSrcPic, videoMsg, false);
686 this->ServerTimer->GetTime();
687 this->encodeEndTime = this->ServerTimer->GetTimeStampInNanoseconds();
688 iTotal += this->encodeEndTime - iStart;
689 ++ iFrameIdx;
690 if (this->videoEncoder->GetVideoFrameType() == FrameTypeSkip) {
691 continue;
692 }
693
694 if (iEncFrames == ResultSuccess ) {
695 static int messageID = -1;
696 messageID++;
697 this->totalCompressedDataSize += videoMsg->GetPackedBitStreamSize();
698 encodedFrame* frame = new encodedFrame();
699 memcpy(frame->messagePackPointer, videoMsg->GetPackPointer(), videoMsg->GetBufferSize());
700 frame->messageDataLength = videoMsg->GetBufferSize();
701 this->glock->Lock();
702 this->CheckEncodedFrameMap();
703 this->encodedFrames.insert(std::pair<igtlUint32, VideoStreamIGTLinkServer::encodedFrame*>(messageID, frame));
704 this->glock->Unlock();
705 igtl::Sleep(5);
706 iActualFrameEncodedCount ++;
707 } else {
708 fprintf (stderr, "EncodeFrame(), ret: %d, frame index: %d.\n", iEncFrames, iFrameIdx);
709 }
710 if (iActualFrameEncodedCount%10 == 0) {
711 double dElapsed = iTotal / 1e9;
712 float totalFrameSize = iActualFrameEncodedCount*3/2.0*this->pSrcPic->picWidth*this->pSrcPic->picHeight;
713 printf ("Width:\t\t%d\nHeight:\t\t%d\nFrames:\t\t%d\nencode time:\t%f sec\nFPS:\t\t%f fps\nCompressionRate:\t\t%f\n",
714 this->pSrcPic->picWidth, this->pSrcPic->picHeight,
715 iActualFrameEncodedCount, dElapsed, (iActualFrameEncodedCount * 1.0) / dElapsed, totalCompressedDataSize/totalFrameSize);
716 }
717 }
718 }
719 return 0;
720}
