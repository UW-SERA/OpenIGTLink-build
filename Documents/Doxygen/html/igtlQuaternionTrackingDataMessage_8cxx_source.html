314int QuaternionTrackingDataMessage::PackContent()
315{
316 // Allocate buffer
317 AllocateBuffer();
318
319 igtl_qtdata_element* element = NULL;
320 element = (igtl_qtdata_element*)(this->m_Content);
321
322 igtl_qtdata_element * elementHolder = element;
323 std::vector<QuaternionTrackingDataElement::Pointer>::iterator iter;
324 for (iter = this->m_QuaternionTrackingDataList.begin(); iter != this->m_QuaternionTrackingDataList.end(); iter ++)
325 {
326 strncpy((char*)element->name, (*iter)->GetName(), IGTL_QTDATA_LEN_NAME);
327 element->type = (*iter)->GetType();
328 element->reserved = 0;
329 float p[3];
330 (*iter)->GetPosition(p);
331 float q[4];
332 (*iter)->GetQuaternion(q);
333
334 for (int i = 0; i < 3; i ++)
335 {
336 element->position[i] = p[i];
337 }
338 for (int j = 0; j < 4; j ++)
339 {
340 element->quaternion[j] = q[j];
341 }
342
343 element ++;
344 }
345
346 igtl_qtdata_convert_byte_order(elementHolder, this->m_QuaternionTrackingDataList.size());
347
348 return 1;
349}

352int QuaternionTrackingDataMessage::UnpackContent()
353{
354 this->m_QuaternionTrackingDataList.clear();
355
356 igtl_qtdata_element* element = NULL;
357 int nElement = 0;
358#if OpenIGTLink_HEADER_VERSION >= 2
359 element = (igtl_qtdata_element*) (this->m_Content);
360 bool isUnpacked(true);
361 nElement = igtl_qtdata_get_data_n(CalculateReceiveContentSize(isUnpacked));
362#elif OpenIGTLink_PROTOCOL_VERSION <=2
363 element = (igtl_qtdata_element*) this->m_Body;
364 nElement = igtl_qtdata_get_data_n(this->m_BodySizeToRead);
365#endif
366 igtl_qtdata_convert_byte_order(element, nElement);
367
368 char strbuf[128];
369 for (int i = 0; i < nElement; i ++)
370 {
371 QuaternionTrackingDataElement::Pointer elemClass = QuaternionTrackingDataElement::New();
372
373 // Add '\n' at the end of each string
374 // (necessary for a case, where a string reaches the maximum length.)
375 strbuf[IGTL_QTDATA_LEN_NAME] = '\n';
376 strncpy(strbuf, (char*)element->name, IGTL_QTDATA_LEN_NAME);
377 elemClass->SetName((const char*)strbuf);
378 elemClass->SetType(element->type);
379
380 float p[3] = {0, 0, 0};
381 float q[4] = {0, 0, 0, 1};
382
383 for (int i = 0; i < 3; i ++)
384 {
385 p[i] = element->position[i];
386 }
387 for (int j = 0; j < 4; j ++)
388 {
389 q[j] = element->quaternion[j];
390 }
391
392 elemClass->SetPosition(p);
393 elemClass->SetQuaternion(q);
394
395 this->m_QuaternionTrackingDataList.push_back(elemClass);
396
397 element ++;
398 }
399
400 return 1;
401}
