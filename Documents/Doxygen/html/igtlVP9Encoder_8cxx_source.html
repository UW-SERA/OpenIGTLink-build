197int VP9Encoder::InitializeEncoder()
198{
199 cfg.g_lag_in_frames = 0;
200 cfg.g_w = this->GetPicWidth();
201 cfg.g_h = this->GetPicHeight();
202 if (inputImage->fmt == VPX_IMG_FMT_NONE)
203 {
204 return -1;
205 }
206 vpx_img_alloc(inputImage, inputImage->fmt, cfg.g_w,
207 cfg.g_h, 1);
208 if (vpx_codec_enc_init(codec, encoder->codec_interface(), &cfg, 0))
209 {
210 error_output(codec, "Failed to initialize encoder");
211 return -1;
212 }
213
214 if (vpx_codec_control_(codec, VP9E_SET_LOSSLESS, this->GetLosslessLink()))
215 {
216 error_output(codec, "Failed to set lossless mode");
217 return -1;
218 }
219
220 if (codecSpeed >= SlowestSpeed && codecSpeed <= FastestSpeed)
221 {
222 if (vpx_codec_control(codec, VP8E_SET_CPUUSED, codecSpeed))
223 {
224 error_output(codec, "Failed to set Speed");
225 return -1;
226 }
227 }
228 this->initializationDone = true;
229 return 0;
230}

253int VP9Encoder::ConvertToLocalImageFormat(SourcePicture* pSrcPic)
254{
255 if (pSrcPic->picWidth != this->cfg.g_w || pSrcPic->picHeight != this->cfg.g_h)
256 {
257 this->SetPicWidthAndHeight(pSrcPic->picWidth,pSrcPic->picHeight);
258 this->InitializeEncoder();
259 }
260 int plane;
261 for (plane = 0; plane < 3; ++plane)
262 {
263 unsigned char *buf ;
264 const int w = vpx_img_plane_width(inputImage, plane) *
265 ((inputImage->fmt & VPX_IMG_FMT_HIGHBITDEPTH) ? 2 : 1);
266 const int h = vpx_img_plane_height(inputImage, plane);
267 buf = inputImage->planes[plane];
268 memcpy(buf, pSrcPic->data[plane], w*h);
269 }
270
271 return 1;
272}

274int VP9Encoder::EncodeSingleFrameIntoVideoMSG(SourcePicture* pSrcPic, igtl::VideoMessage* videoMessage, bool isGrayImage)
275{
276 int iSourceWidth = pSrcPic->picWidth;
277 int iSourceHeight = pSrcPic->picHeight;
278 if (iSourceWidth != this->cfg.g_w || iSourceHeight != this->cfg.g_h)
279 {
280 this->SetPicWidthAndHeight(iSourceWidth,iSourceHeight);
281 }
282 if (this->initializationDone == true)
283 {
284 static igtl_uint32 messageID = 6;
285 messageID ++;
286 this->ConvertToLocalImageFormat(pSrcPic);
287 const vpx_codec_err_t res2 = vpx_codec_encode(codec, inputImage, messageID, 1, 0, this->deadlineMode);
288 if (res2 != VPX_CODEC_OK)
289 {
290 error_output(codec, "Failed to encode frame");
291 return -1;
292 }
293 iter = NULL;
294 while ((pkt = vpx_codec_get_cx_data(codec, &iter)) != NULL)
295 {
296 if (pkt->kind == VPX_CODEC_CX_FRAME_PKT)
297 {
298 if((pkt->data.frame.flags & VPX_FRAME_IS_KEY) != 0)
299 {
300 encodedFrameType = FrameTypeKey;
301 }
302 else
303 {
304 // To do, assign other frame type too.
305 encodedFrameType = FrameTypeUnKnown;
306 }
307 videoMessage->SetFrameType(encodedFrameType);
308 videoMessage->SetBitStreamSize(pkt->data.frame.sz);
309 videoMessage->AllocateScalars();
310 videoMessage->SetCodecType(IGTL_VIDEO_CODEC_NAME_VP9);
311 int endian = (igtl_is_little_endian() == 1 ? IGTL_VIDEO_ENDIAN_LITTLE : IGTL_VIDEO_ENDIAN_BIG);
312 videoMessage->SetEndian(endian); //little endian is 2 big endian is 1
313 videoMessage->SetWidth(pSrcPic->picWidth);
314 videoMessage->SetHeight(pSrcPic->picHeight);
315 if (isGrayImage)
316 {
317 encodedFrameType = encodedFrameType << 8;
318 }
319 videoMessage->SetFrameType(encodedFrameType);
320 memcpy(videoMessage->GetPackFragmentPointer(2), pkt->data.frame.buf, pkt->data.frame.sz);
321 videoMessage->Pack();
322 }
323 }
324 if (videoMessage->GetBitStreamSize())
325 {
326 return 0;
327 }
328 }
329 return -1;
330}
