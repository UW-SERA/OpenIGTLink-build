90int QueryMessage::PackContent()
91{
92 // Allocate buffer
93 AllocateBuffer();
94 igtl_query_header * query_header;
95 char * deviceName;
96
97 // Set pointers
98#if OpenIGTLink_HEADER_VERSION >= 2
99 query_header = (igtl_query_header*) this->m_Content;
100 deviceName = (char *) this->m_Content + sizeof(igtl_query_header);
101#else
102 query_header = (igtl_query_header*) this->m_Body;
103 deviceName = (char *) this->m_Body + sizeof(igtl_query_header);
104#endif
105
106 // Copy data
107 query_header->deviceUIDLength = static_cast<igtlUint16>(this->m_DeviceUID.length());
108 query_header->queryID = this->m_QueryID;
109 memcpy(query_header->queryDataType, this->m_DataType, IGTL_QUERY_DATE_TYPE_SIZE);
110 strncpy(deviceName, this->m_DeviceUID.c_str(), query_header->deviceUIDLength);
111
112 // Convert byte order from host to network
113 igtl_query_convert_byte_order(query_header);
114
115 return 1;
116}

118int QueryMessage::UnpackContent()
119{
120 igtl_query_header * query_header;
121 char * deviceName;
122
123#if OpenIGTLink_HEADER_VERSION >= 2
124 query_header = (igtl_query_header*) this->m_Content;
125 deviceName = (char *) this->m_Content + sizeof(igtl_query_header);
126#else
127 query_header = (igtl_query_header*) this->m_Body;
128 deviceName = (char *) this->m_Body + sizeof(igtl_query_header);
129#endif
130
131 // Convert byte order from network to host
132 igtl_query_convert_byte_order(query_header);
133
134 // Copy data
135 this->m_QueryID = query_header->queryID;
136 memcpy(m_DataType, query_header->queryDataType, IGTL_QUERY_DATE_TYPE_SIZE);
137 this->m_DeviceUID.clear();
138 this->m_DeviceUID.append(deviceName, query_header->deviceUIDLength);
139
140 return 1;
141}
