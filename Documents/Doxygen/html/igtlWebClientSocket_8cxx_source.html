31int webSocketClient::ConnectToServer(const char* hostName, int port)
32{
33 websocketpp::lib::error_code ec;
34 std::string uri;
35 uri.append("http://");
36 uri.append(hostName);
37 uri.append(":");
38 uri.append(std::to_string(port));
39 client::connection_ptr con = m_client.get_connection(uri, ec);
40 if (ec)
41 {
42 m_client.get_alog().write(websocketpp::log::alevel::app,
43 "Get Connection Error: "+ec.message());
44 return -1;
45 }
46
47 // Grab a handle for this connection so we can talk to it in a thread
48 // safe manor after the event loop starts.
49 m_hdl = con->get_handle();
50
51 // Queue the connection. No DNS queries or network connections will be
52 // made until the io_service event loop is run.
53 m_client.connect(con);
54
55 m_client.run();
56
57 return 0;
58}

97int webSocketClient::Receive(void* data, int& length)
98{
99 unique_lock<mutex> lock(m_action_lock);
100
101 while(m_messages.empty())
102 {
103 m_action_cond.wait(lock);
104 }
105 std::string a = m_messages.front();
106 int messageLen = a.length();
107 if(length && messageLen>=length)
108 {
109 memcpy(data, a.c_str(), length);
110 }
111 if (messageLen == length)
112 {
113 m_messages.pop();
114 }
115 else
116 {
117 a.erase(a.begin(), a.begin()+length);
118 messageLen = a.length();
119 m_messages.front()= a;
120 }
121 lock.unlock();
122 return length;
123}

125int webSocketClient::Send(const void* data, int length)
126{
127 std::stringstream val;
128 websocketpp::lib::error_code ec;
129 m_client.send(m_hdl,val.str(),websocketpp::frame::opcode::binary,ec);
130
131 // The most likely error that we will get is that the connection is
132 // not in the right state. Usually this means we tried to send a
133 // message to a connection that was closed or in the process of
134 // closing. While many errors here can be easily recovered from,
135 // in this simple example, we'll stop the telemetry loop.
136 if (ec)
137 {
138 m_client.get_alog().write(websocketpp::log::alevel::app,
139 "Send Error: "+ec.message());
140 return 0;
141 }
142 return 1;
143}
