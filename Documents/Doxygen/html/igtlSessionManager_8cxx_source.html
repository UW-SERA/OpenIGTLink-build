50int SessionManager::AddMessageHandler(MessageHandler* handler)
51{
52 // Check if there is any handler for the same message type
53 std::vector< MessageHandler* >::iterator iter;
54 for (iter = this->m_MessageHandlerList.begin(); iter != this->m_MessageHandlerList.end(); iter ++)
55 {
56 if ( (*iter)->GetMessageType() == handler->GetMessageType() )
57 {
58 return 0;
59 }
60 }
61
62 // If not, add the handler to the list.
63 this->m_MessageHandlerList.push_back(handler);
64
65 return 1;
66}

85int SessionManager::Connect()
86{
87
88 this->m_Socket = NULL;
89
90 if (this->m_Mode == MODE_CLIENT)
91 {
92 ClientSocket::Pointer clientSocket;
93 clientSocket = ClientSocket::New();
94 //this->DebugOff();
95 if (this->m_Hostname.length() == 0)
96 {
97 return 0;
98 }
99 //this->m_Socket->SetConnectTimeout(1000);
100 int r = clientSocket->ConnectToServer(this->m_Hostname.c_str(), this->m_Port);
101 if (r == 0) // if connected to server
102 {
103 //clientSocket->SetReceiveTimeout(0);
104 this->m_Socket = clientSocket;
105 }
106 else
107 {
108 return 0;
109 }
110 }
111 else
112 {
113 ServerSocket::Pointer serverSocket;
114 serverSocket = ServerSocket::New();
115 int r = serverSocket->CreateServer(this->m_Port);
116 if (r < 0)
117 {
118 return 0;
119 }
120
121 if (serverSocket.IsNotNull())
122 {
123 //this->ServerSocket->CreateServer(this->m_Port);
124 this->m_Socket = serverSocket->WaitForConnection(10000);
125 }
126
127 if (this->m_Socket.IsNotNull() && this->m_Socket->GetConnected()) // if client connected
128 {
129 this->m_Socket->DebugOff();
130 }
131 else
132 {
133 return 0;
134 }
135 }
136
137 this->m_Socket->SetReceiveBlocking(0); // Pseudo non-blocking
138 this->m_CurrentReadIndex = 0;
139 this->m_HeaderDeserialized = 0;
140 return 1;
141}

154int SessionManager::ProcessMessage()
155{
156 // The workflow of this function is as follows:
157 //
158 // HEADER:
159 // IF the message is (a) a new message:
160 // start reading the header;
161 // ELSE IF the message is a message in progress:
162 // if the process is reading the header:
163 // continue to read the header;
164 // ELSE:
165 // GOTO BODY;
166 //
167 // IF the header has not been received:
168 // GOTO BODY;
169 // ELSE
170 // RETURN;
171 //
172 // BODY:
173 // IF the process has not started reading the body:
174 // check the body type;
175 // find an appropriate handler;
176 // start reading the body
177 // ELSE:
178 // continue to read the body
179 //
180
181 //--------------------------------------------------
182 // Header
183 if (this->m_CurrentReadIndex == 0)
184 {
185 // Initialize receive buffer
186 this->m_Header->InitBuffer();
187
188 // Receive generic header from the socket
189 bool timeout(false);
190 igtl_uint64 r = this->m_Socket->Receive(this->m_Header->GetBufferPointer(), this->m_Header->GetBufferSize(), timeout, 0);
191 if (r == 0)
192 {
193 this->m_CurrentReadIndex = 0;
194 this->m_HeaderDeserialized = 0;
195 return 0; // Disconnected
196 }
197 if (r != this->m_Header->GetBufferSize())
198 {
199 // Only a part of header has arrived.
200 if (timeout) // timeout
201 {
202 this->m_CurrentReadIndex = 0;
203 }
204 else
205 {
206 this->m_CurrentReadIndex = r;
207 }
208 return -1;
209 }
210 // The header has been received.
211 this->m_CurrentReadIndex = IGTL_HEADER_SIZE;
212 }
213 else if (this->m_CurrentReadIndex < IGTL_HEADER_SIZE)
214 {
215 // Message transfer was interrupted in the header
216 bool timeout(false);
217 igtl_uint64 r = this->m_Socket->Receive((void*)((char*)this->m_Header->GetBufferPointer()+this->m_CurrentReadIndex),
218 this->m_Header->GetBufferSize()-this->m_CurrentReadIndex, timeout, 0);
219 if (r == 0)
220 {
221 this->m_CurrentReadIndex = 0;
222 this->m_HeaderDeserialized = 0;
223 return 0; // Disconnected
224 }
225 if (r != this->m_Header->GetBufferSize()-this->m_CurrentReadIndex)
226 {
227 // Only a part of header has arrived.
228 if (r > 0) // exclude a case of timeout
229 {
230 this->m_CurrentReadIndex += r;
231 }
232 return -1;
233 }
234 // The header has been received.
235 this->m_CurrentReadIndex = IGTL_HEADER_SIZE;
236 }
237
238
239 //--------------------------------------------------
240 // Body
241 if (this->m_HeaderDeserialized == 0)
242 {
243 // Deserialize the header
244 this->m_Header->Unpack();
245
246 // Get time stamp
247 igtlUint32 sec;
248 igtlUint32 nanosec;
249
250 this->m_Header->GetTimeStamp(this->m_TimeStamp);
251 this->m_TimeStamp->GetTimeStamp(&sec, &nanosec);
252 //std::cerr << "Time stamp: "
253 // << sec << "." << std::setw(9) << std::setfill('0')
254 // << nanosec << std::endl;
255
256 // Look for a message handler that matches to the message type.
257 int found = 0;
258 std::vector< MessageHandler* >::iterator iter;
259 for (iter = this->m_MessageHandlerList.begin(); iter != this->m_MessageHandlerList.end(); iter ++)
260 {
261#if OpenIGTLink_HEADER_VERSION >= 2
262 if ( this->m_Header->GetMessageType() == (*iter)->GetMessageType() )
263#else
264 if (strcmp(this->m_Header->GetDeviceType(), (*iter)->GetMessageType()) == 0)
265#endif
266 {
267 this->m_CurrentMessageHandler = *iter;
268 found = 1;
269 break;
270 }
271 }
272
273 // If there is no message handler, skip the message
274 if (!found)
275 {
276#if OpenIGTLink_HEADER_VERSION >= 2
277 std::cerr << "Receiving: " << this->m_Header->GetMessageType() << std::endl;
278#else
279 std::cerr << "Receiving: " << this->m_Header->GetDeviceType() << std::endl;
280#endif
281 this->m_Socket->Skip(this->m_Header->GetBodySizeToRead(), 0);
282 // Reset the index counter to be ready for the next message
283 this->m_CurrentReadIndex = 0;
284 this->m_HeaderDeserialized = 0;
285 return 1;
286 }
287
288 this->m_HeaderDeserialized = 1;
289 }
290
291 igtl_uint64 r = this->m_CurrentMessageHandler->ReceiveMessage(this->m_Socket, this->m_Header,
292 this->m_CurrentReadIndex-IGTL_HEADER_SIZE);
293 if (r == this->m_Header->GetBodySizeToRead())
294 {
295 this->m_CurrentReadIndex = 0;
296 this->m_HeaderDeserialized = 0;
297 }
298 else
299 {
300 this->m_CurrentReadIndex += IGTL_HEADER_SIZE + r;
301 }
302
303 return 1;
304}
