310int TrackingDataMessage::PackContent()
311{
312 // Allocate buffer
313 AllocateBuffer();
314 igtl_tdata_element* element = NULL;
315 element = (igtl_tdata_element*)(this->m_Content);
316
317 igtl_tdata_element * elementHolder = element;
318 std::vector<TrackingDataElement::Pointer>::iterator iter;
319
320 for (iter = this->m_TrackingDataList.begin(); iter != this->m_TrackingDataList.end(); iter ++)
321 {
322 strncpy((char*)element->name, (*iter)->GetName(), IGTL_TDATA_LEN_NAME);
323 element->type = (*iter)->GetType();
324 element->reserved = 0;
325 Matrix4x4 matrix;
326 (*iter)->GetMatrix(matrix);
327 for (int i = 0; i < 3; i ++)
328 {
329 element->transform[i] = matrix[i][0];
330 element->transform[i+3] = matrix[i][1];
331 element->transform[i+6] = matrix[i][2];
332 element->transform[i+9] = matrix[i][3];
333 }
334 element ++;
335 }
336
337 igtl_tdata_convert_byte_order(elementHolder, this->m_TrackingDataList.size());
338
339 return 1;
340}

343int TrackingDataMessage::UnpackContent()
344{
345
346 this->m_TrackingDataList.clear();
347
348 igtl_tdata_element* element = NULL;
349 int nElement = 0;
350
351 element = (igtl_tdata_element*)(this->m_Content);
352 bool isUnpacked(true);
353 nElement = igtl_tdata_get_data_n(CalculateReceiveContentSize(isUnpacked));
354
355 igtl_tdata_convert_byte_order(element, nElement);
356
357 char strbuf[128];
358 for (int i = 0; i < nElement; i ++)
359 {
360 TrackingDataElement::Pointer elemClass = TrackingDataElement::New();
361
362 // Add '\n' at the end of each string
363 // (neccesary for a case, where a string reaches the maximum length.)
364 strbuf[IGTL_TDATA_LEN_NAME] = '\n';
365 strncpy(strbuf, (char*)element->name, IGTL_TDATA_LEN_NAME);
366 elemClass->SetName((const char*)strbuf);
367 elemClass->SetType(element->type);
368
369 Matrix4x4 matrix;
370 IdentityMatrix(matrix);
371 for (int j = 0; j < 3; j ++)
372 {
373 matrix[j][0] = element->transform[j];
374 matrix[j][1] = element->transform[j+3];
375 matrix[j][2] = element->transform[j+6];
376 matrix[j][3] = element->transform[j+9];
377 }
378 elemClass->SetMatrix(matrix);
379
380 this->m_TrackingDataList.push_back(elemClass);
381
382 element ++;
383 }
384
385 return 1;
386}
