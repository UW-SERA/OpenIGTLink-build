91int StringMessage::PackContent()
92{
93 // Allocate buffer
94 AllocateBuffer();
95 igtl_string_header * string_header;
96 char * string;
97 // Set pointers
98#if OpenIGTLink_HEADER_VERSION >= 2
99 string_header = (igtl_string_header*) this->m_Content;
100 string = (char *) this->m_Content + sizeof(igtlUint16)*2;
101#elif OpenIGTLink_PROTOCOL_VERSION <=2
102 string_header = (igtl_string_header*) this->m_Body;
103 string = (char *) this->m_Body + sizeof(igtlUint16)*2;
104#endif
105
106 // Copy data
107 string_header->encoding = static_cast<igtlUint16>(this->m_Encoding);
108 string_header->length = static_cast<igtlUint16>(this->m_String.length());
109 strncpy(string, this->m_String.c_str(), string_header->length);
110
111 // Convert byte order from host to network
112 igtl_string_convert_byte_order(string_header);
113
114 return 1;
115}

117int StringMessage::UnpackContent()
118{
119 igtl_string_header * string_header;
120 char * string;
121#if OpenIGTLink_HEADER_VERSION >= 2
122 string_header = (igtl_string_header*) (this->m_Content);
123 string = (char *) this->m_Content + sizeof(igtlUint16)*2;
124#elif OpenIGTLink_PROTOCOL_VERSION <=2
125 string_header = (igtl_string_header*) this->m_Body;
126 string = (char *) this->m_Body + sizeof(igtlUint16)*2;
127#endif
128
129 // Convert byte order from network to host
130 igtl_string_convert_byte_order(string_header);
131
132 // Copy data
133 this->m_Encoding = string_header->encoding;
134 this->m_String.clear();
135 this->m_String.append(string, string_header->length);
136
137 return 1;
138}
