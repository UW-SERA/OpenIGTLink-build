42void igtl_export igtl_frame_convert_byte_order(igtl_frame_header * header)
43{
44 int i;
45 igtl_uint32 tmp[12];
46
47 if (igtl_is_little_endian())
48 {
49 header->header_version = BYTE_SWAP_INT16(header->header_version);
50 header->frameType = BYTE_SWAP_INT16(header->frameType);
51
52 for (i = 0; i < 3; i++)
53 {
54 header->size[i] = BYTE_SWAP_INT16(header->size[i]);
55 header->subvol_size[i] = BYTE_SWAP_INT16(header->subvol_size[i]);
56 header->subvol_offset[i] = BYTE_SWAP_INT16(header->subvol_offset[i]);
57 }
58
59 memcpy(tmp, header->matrix, sizeof(igtl_uint32) * 12);
60 for (i = 0; i < 12; i++)
61 {
62 tmp[i] = BYTE_SWAP_INT32(tmp[i]);
63 }
64 memcpy(header->matrix, tmp, sizeof(igtl_uint32) * 12);
65 }
66}

102void igtl_export igtl_frame_get_matrix(float spacing[3], float origin[3],
103 float norm_i[3], float norm_j[3], float norm_k[3],
104 igtl_frame_header * header)
105{
106 float tx;
107 float ty;
108 float tz;
109 float sx;
110 float sy;
111 float sz;
112 float nx;
113 float ny;
114 float nz;
115 float px;
116 float py;
117 float pz;
118
119 tx = (float)header->matrix[0];
120 ty = (float)header->matrix[1];
121 tz = (float)header->matrix[2];
122 sx = (float)header->matrix[3];
123 sy = (float)header->matrix[4];
124 sz = (float)header->matrix[5];
125 nx = (float)header->matrix[6];
126 ny = (float)header->matrix[7];
127 nz = (float)header->matrix[8];
128 px = (float)header->matrix[9];
129 py = (float)header->matrix[10];
130 pz = (float)header->matrix[11];
131
132 spacing[0] = sqrtf(tx*tx + ty*ty + tz*tz);
133 spacing[1] = sqrtf(sx*sx + sy*sy + sz*sz);
134 spacing[2] = sqrtf(nx*nx + ny*ny + nz*nz);
135
136 if (spacing[0] != 0)
137 {
138 norm_i[0] = header->matrix[0] / spacing[0];
139 norm_i[1] = header->matrix[1] / spacing[0];
140 norm_i[2] = header->matrix[2] / spacing[0];
141 }
142 if (spacing[1] != 0)
143 {
144 norm_j[0] = header->matrix[3] / spacing[1];
145 norm_j[1] = header->matrix[4] / spacing[1];
146 norm_j[2] = header->matrix[5] / spacing[1];
147 }
148 if (spacing[2] != 0)
149 {
150 norm_k[0] = header->matrix[6] / spacing[2];
151 norm_k[1] = header->matrix[7] / spacing[2];
152 norm_k[2] = header->matrix[8] / spacing[2];
153 }
154
155 origin[0] = header->matrix[9];
156 origin[1] = header->matrix[10];
157 origin[2] = header->matrix[11];
158
159}
