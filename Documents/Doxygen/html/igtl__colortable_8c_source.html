18igtl_uint64 igtl_export igtl_colortable_get_table_size(igtl_colortable_header* header)
19{
20 igtl_int32 n;
21 igtl_int32 s;
22
23 if (header->indexType == IGTL_COLORTABLE_INDEX_UINT8)
24 {
25 n = 256;
26 }
27 else /* IGTL_COLORTABLE_INDEX_UINT64 */
28 {
29 n = 256*256;
30 }
31
32 if (header->mapType == IGTL_COLORTABLE_MAP_UINT8)
33 {
34 s = 1;
35 }
36 else if (header->mapType == IGTL_COLORTABLE_MAP_UINT16)
37 {
38 s = 2;
39 }
40 else /* IGTL_COLORTABLE_MAP_RGB */
41 {
42 s = 3;
43 }
44
45 return (igtl_uint64) n*s;
46}

49void igtl_export igtl_colortable_convert_byte_order(igtl_colortable_header* header, void* table)
50{
51 int i;
52 int n;
53 igtl_uint16* tmp;
54
55 if (igtl_is_little_endian())
56 {
57 if (header->mapType == IGTL_COLORTABLE_MAP_UINT16)
58 {
59 n = (header->indexType == IGTL_COLORTABLE_INDEX_UINT16)? 256*256 : 256;
60 tmp = (igtl_uint16*) table;
61 for (i = 0; i < n; i ++)
62 {
63 tmp[i] = BYTE_SWAP_INT16(tmp[i]);
64 }
65 }
66 }
67}

70igtl_uint64 igtl_export igtl_colortable_get_crc(igtl_colortable_header* header, void* table)
71{
72 igtl_uint64 crc;
73 igtl_uint64 data_size;
74
75 /* calculate header size using igtl_colortable_get_table_size().
76 * This funciton can be called after byte order conversion, since
77 * both index and map type field are uint8 and not affected by byte conversion.
78 */
79
80 data_size = igtl_colortable_get_table_size(header);
81 crc = crc64(0, 0, 0);
82 crc = crc64((unsigned char*) header, IGTL_COLORTABLE_HEADER_SIZE, crc);
83 crc = crc64((unsigned char*) table, (int)data_size, crc);
84
85 return crc;
86}
