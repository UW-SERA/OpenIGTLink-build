64{
65 // Get thread information

68
69 ReadSocketAndPush parentObj = *(static_cast<ReadSocketAndPush*>(info->UserData));
70 unsigned char UDPPacket[RTP_PAYLOAD_LENGTH+RTP_HEADER_LENGTH];

72 {
73 int totMsgLen = parentObj.clientSocket->ReadSocket(UDPPacket, RTP_PAYLOAD_LENGTH+RTP_HEADER_LENGTH);
74
75 WriteTimeInfo(UDPPacket, totMsgLen, parentObj.receiver);

110int VideoStreamIGTLinkReceiver::RunOnTCPSocket()
111{
112 //------------------------------------------------------------
113 // Establish Connection
114 int r = socket->ConnectToServer(TCPServerIPAddress, TCPServerPort);
115
116 if (r != 0)
117 {
118 std::cerr << "Cannot connect to the server." << std::endl;
119 exit(0);
120 }
121
122 //------------------------------------------------------------
123 // Ask the server to start pushing tracking data
124 std::cerr << "Sending STT_VIDEO message....." << std::endl;
125 igtl::StartVideoMessage::Pointer startVideoMsg;
126 startVideoMsg = igtl::StartVideoMessage::New();
127 startVideoMsg->AllocatePack();
128 startVideoMsg->SetHeaderVersion(IGTL_HEADER_VERSION_2);
129 startVideoMsg->SetDeviceName("MacCamera5");
130 startVideoMsg->SetCodecType(codecType);
131 startVideoMsg->SetTimeInterval(interval);
132 startVideoMsg->Pack();
133 socket->Send(startVideoMsg->GetPackPointer(), startVideoMsg->GetPackSize());
134 while (1)
135 {
136 //------------------------------------------------------------
137 // Wait for a reply
138 igtl::MessageHeader::Pointer headerMsg;
139 headerMsg = igtl::MessageHeader::New();
140 headerMsg->InitPack();
141 bool timeout(false);
142 int rs = socket->Receive(headerMsg->GetPackPointer(), headerMsg->GetPackSize(), timeout);
143 if (rs == 0)
144 {
145 std::cerr << "Connection closed." << std::endl;
146 socket->CloseSocket();
147 exit(0);
148 }
149 if (rs != headerMsg->GetPackSize())
150 {
151 std::cerr << rs <<"Message size information and actual data size don't match." <<headerMsg->GetPackSize()<< std::endl;
152 continue;
153 }
154
155 headerMsg->Unpack();
156 if (strcmp(headerMsg->GetDeviceName(), this->deviceName.c_str()) == 0)
157 {
158 //------------------------------------------------------------
159 // Allocate Video Message Class
160
161 igtl::VideoMessage::Pointer videoMsg;
162 videoMsg = igtl::VideoMessage::New();
163 videoMsg->SetMessageHeader(headerMsg);
164 videoMsg->AllocateScalars();
165 bool timeout(false);
166 // Receive body from the socket
167 socket->Receive(videoMsg->GetPackBodyPointer(), videoMsg->GetPackBodySize(), timeout);
168 // Deserialize the transform data
169 // If you want to skip CRC check, call Unpack() without argument.
170 int c = videoMsg->Unpack(1);
171
172 if (c==0) // if CRC check fails
173 {
174 // TODO: error handling
175 return 0;
176 }
177 this->SetWidth(videoMsg->GetWidth());
178 this->SetHeight(videoMsg->GetHeight());
179 int streamLength = videoMsg->GetBitStreamSize();
180 if (!(this->videoMessageBuffer==NULL))
181 {
182 delete[] this->videoMessageBuffer;
183 }
184 this->videoMessageBuffer = new igtl_uint8[streamLength];
185 memcpy(this->videoMessageBuffer, videoMsg->GetPackFragmentPointer(2), streamLength);
186 static int frameNum = 0;
187 int status = this->ProcessVideoStream(this->videoMessageBuffer,streamLength);
188
189 if (status == 0)
190 {
191 this->SendStopMessage();
192 break;
193 }
194 else if(status == 2)
195 {
196 frameNum ++;
197 }
198 }
199 else
200 {
201 std::cerr << "Receiving : " << headerMsg->GetDeviceType() << std::endl;
202 socket->Skip(headerMsg->GetBodySizeToRead(), 0);
203 }
204 igtl::Sleep(20);
205 }
206 return 1;
207}

223int VideoStreamIGTLinkReceiver::RunOnUDPSocket()
224{
225 igtl::ConditionVariable::Pointer conditionVar = igtl::ConditionVariable::New();
226 igtl::SimpleMutexLock* glock = igtl::SimpleMutexLock::New();
227 UDPSocket->JoinNetwork("127.0.0.1", UDPClientPort);
228 ReadSocketAndPush info;
229 info.wrapper = rtpWrapper;
230 info.clientSocket = UDPSocket;
231 info.receiver = this;
232
233 Wrapper infoWrapper;
234 infoWrapper.wrapper = rtpWrapper;
235 infoWrapper.receiver = this;
236
237 igtl::MultiThreader::Pointer threader = igtl::MultiThreader::New();
238 threader->SpawnThread((igtl::ThreadFunctionType)&ThreadFunctionReadSocket, &info);
239 threader->SpawnThread((igtl::ThreadFunctionType)&ThreadFunctionUnWrap, &infoWrapper);
240 while(1)
241 {
242 glock->Lock();
243 unsigned int messageNum = rtpWrapper->unWrappedMessages.size();
244 glock->Unlock();
245 if(messageNum)// to do: glock this session
246 {
247 igtl::VideoMessage::Pointer videoMultiPKTMSG = igtl::VideoMessage::New();
248 glock->Lock();
249 std::map<igtl_uint32, igtl::UnWrappedMessage*>::iterator it = rtpWrapper->unWrappedMessages.begin();
250 igtlUint8 * message = new igtlUint8[it->second->messageDataLength];
251 static int frameNum = 0;
252 int MSGLength = it->second->messageDataLength;
253 memcpy(message, it->second->messagePackPointer, it->second->messageDataLength);
254 delete it->second;
255 it->second = NULL;
256 rtpWrapper->unWrappedMessages.erase(it);
257 glock->Unlock();
258 igtl::MessageHeader::Pointer header = igtl::MessageHeader::New();
259 header->InitPack();
260 memcpy(header->GetPackPointer(), message, IGTL_HEADER_SIZE);
261 header->Unpack();
262 videoMultiPKTMSG->SetMessageHeader(header);
263 videoMultiPKTMSG->AllocateBuffer();
264 if (MSGLength == videoMultiPKTMSG->GetPackSize())
265 {
266 memcpy(videoMultiPKTMSG->GetPackPointer(), message, MSGLength);
267 videoMultiPKTMSG->Unpack(0);
268 int streamLength = videoMultiPKTMSG->GetBitStreamSize();
269 this->SetWidth(videoMultiPKTMSG->GetWidth());
270 this->SetHeight(videoMultiPKTMSG->GetHeight());
271 if (!(this->videoMessageBuffer==NULL))
272 {
273 delete[] this->videoMessageBuffer;
274 }
275 this->videoMessageBuffer = new igtl_uint8[streamLength];
276 memcpy(this->videoMessageBuffer, videoMultiPKTMSG->GetPackFragmentPointer(2), streamLength);
277
278 int status = this->ProcessVideoStream(this->videoMessageBuffer,streamLength);
279
280 if (status == 0)
281 {
282 break;
283 }
284 else if(status == 2)
285 {
286 frameNum ++;
287 }
288 }
289 delete [] message;
290 }
291 }
292 return 0;

325int VideoStreamIGTLinkReceiver::ParseConfigForClient()
326{
327 std::string strTag[4];
328 while (!cRdCfg.EndOfFile())
329 {
330 strTag->clear();
331 long iRd = cRdCfg.ReadLine (&strTag[0]);
332 if (iRd > 0)
333 {
334 if (strTag[0].empty())
335 continue;
336 if (strTag[0].compare ("TCPServerPortNumber") == 0) {
337 this->TCPServerPort = atoi (strTag[1].c_str());
338 if(this->TCPServerPort<0 || this->TCPServerPort>65535)
339 {
340 fprintf (stderr, "Invalid parameter for server port number should between 0 and 65525.");
341 return 1;
342 }
343 }
344 if (strTag[0].compare ("TCPServerIPAddress") == 0) {
345 this->TCPServerIPAddress = new char[IP4AddressStrLen];
346 memcpy(this->TCPServerIPAddress, strTag[1].c_str(), IP4AddressStrLen);
347 if(!inet_addr(this->TCPServerIPAddress))
348 {
349 fprintf (stderr, "Invalid parameter for IP address");
350 return 1;
351 }
352 }
353 if (strTag[0].compare ("UDPClientPortNumber") == 0) {
354 this->UDPClientPort = atoi (strTag[1].c_str());
355 if(this->UDPClientPort<0 || this->UDPClientPort>65535)
356 {
357 fprintf (stderr, "Invalid parameter for server port number should between 0 and 65525.");
358 return 1;
359 }
360 }
361 if (strTag[0].compare ("DeviceName") == 0)
362 {
363 this->deviceName =strTag[1].c_str();
364 }
365 if (strTag[0].compare ("TransportMethod") == 0)
366 {
367 this->transportMethod = atoi(strTag[1].c_str());
368 }
369
370 if (strTag[0].compare ("DecodedFile") == 0)
371 {
372 this->kpOuputFileName = std::string(strTag[1].c_str());
373 }
374 }
375 }
376 return 0;

399int VideoStreamIGTLinkReceiver::ProcessVideoStream(igtl_uint8* bitStream, int streamLength)
400{
401 //std::cerr << "Receiving Video data type." << std::endl;
402 //this->videoMessageBuffer = new igtl_uint8[StreamLength];
403 //memcpy(this->videoMessageBuffer, bitStream, StreamLength);// copy slow down the process, however, the videoMsg is a smart pointer, it gets deleted unpredictable.
404 if(!decodeInstance)
405 {
406 return 0;
407 }
408
409 this->InitializeDecodedFrame();
410 igtl_uint32 dimensions[2] = {static_cast<igtl_uint32>(Width), static_cast<igtl_uint32>(Height)};
411 igtl_uint64 bitStreamLength = streamLength;
412 int status = decodeInstance->DecodeBitStreamIntoFrame(bitStream, this->decodedFrame, dimensions , bitStreamLength);
413 // return 2 for succefully decode one frame
414 FILE* pYuvFile = NULL;
415 // Lenght input mode support
416 if (kpOuputFileName.c_str())
417 {
418 pYuvFile = fopen (kpOuputFileName.c_str(), "ab");
419 if (pYuvFile == NULL)
420 {
421 fprintf (stderr, "Can not open yuv file to output result of decoding..\n");
422 }
423 }
424 igtl_uint32 stride[2] = {static_cast<igtl_uint32>(Width), static_cast<igtl_uint32>(Width/2)};
425 SourcePicture pic;
426 pic.data[0] = this->decodedFrame;
427 pic.data[1] = this->decodedFrame+Width*Height;
428 pic.data[2] = this->decodedFrame+Width*Height*5/4;
429 //pic.data[0] =
430 if(pYuvFile)
431 {
432 decodeInstance->Write2File (pYuvFile, pic.data, dimensions, stride);
433 fclose (pYuvFile);
434 }
435 return status;
436}
