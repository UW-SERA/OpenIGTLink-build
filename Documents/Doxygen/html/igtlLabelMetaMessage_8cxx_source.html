218int LabelMetaMessage::PackContent()
219{
220 // Allocate buffer
221 AllocateBuffer();
222
223 igtl_lbmeta_element* element;
224 element = (igtl_lbmeta_element*)this->m_Content;
225 std::vector<LabelMetaElement::Pointer>::iterator iter;
226
227 for (iter = this->m_LabelMetaList.begin(); iter != this->m_LabelMetaList.end(); iter ++)
228 {
229 strncpy((char*)element->name, (*iter)->GetName(), IGTL_LBMETA_LEN_NAME);
230 strncpy((char*)element->device_name, (*iter)->GetDeviceName(), IGTL_LBMETA_LEN_DEVICE_NAME);
231
232 element->label = (*iter)->GetLabel();
233 element->reserved = 0;
234 igtlUint8 rgba[4];
235 (*iter)->GetRGBA(rgba);
236 element->rgba[0] = rgba[0];
237 element->rgba[1] = rgba[1];
238 element->rgba[2] = rgba[2];
239 element->rgba[3] = rgba[3];
240
241 igtlUint16 size[3];
242 (*iter)->GetSize(size);
243 element->size[0] = size[0];
244 element->size[1] = size[1];
245 element->size[2] = size[2];
246
247 strncpy((char*)element->owner, (*iter)->GetOwner(), IGTL_LBMETA_LEN_OWNER);
248
249 element ++;
250 }
251
252 igtl_lbmeta_convert_byte_order((igtl_lbmeta_element*)this->m_Content, this->m_LabelMetaList.size());
253
254 return 1;
255}

258int LabelMetaMessage::UnpackContent()
259{
260
261 this->m_LabelMetaList.clear();
262
263 igtl_lbmeta_element* element = (igtl_lbmeta_element*) this->m_Content;
264 int nElement = igtl_lbmeta_get_data_n(this->m_BodySizeToRead);
265
266 igtl_lbmeta_convert_byte_order(element, nElement);
267
268 char strbuf[128];
269 for (int i = 0; i < nElement; i ++)
270 {
271 LabelMetaElement::Pointer elemClass = LabelMetaElement::New();
272
273 // Add '\n' at the end of each string
274 // (neccesary for a case, where a string reaches the maximum length.)
275 strbuf[IGTL_LBMETA_LEN_NAME] = '\n';
276 strncpy(strbuf, (char*)element->name, IGTL_LBMETA_LEN_NAME);
277 elemClass->SetName((const char*)strbuf);
278
279 strbuf[IGTL_LBMETA_LEN_DEVICE_NAME] = '\n';
280 strncpy(strbuf, (char*)element->device_name, IGTL_LBMETA_LEN_DEVICE_NAME);
281 elemClass->SetDeviceName(strbuf);
282
283 elemClass->SetLabel(element->label);
284 elemClass->SetRGBA(element->rgba);
285 elemClass->SetSize(element->size);
286
287 strbuf[IGTL_LBMETA_LEN_OWNER] = '\n';
288 strncpy(strbuf, (char*)element->owner, IGTL_LBMETA_LEN_OWNER);
289 elemClass->SetOwner(strbuf);
290
291 this->m_LabelMetaList.push_back(elemClass);
292
293 element ++;
294 }
295
296 return 1;
297}
