161void
162LightObject
163::UnRegister() const
164{
165 m_ReferenceCountLock.Lock();
166 int tmpReferenceCount = --m_ReferenceCount;
167 m_ReferenceCountLock.Unlock();
168
169 // ReferenceCount in now unlocked. We may have a race condition
170 // to delete the object.
171 if ( tmpReferenceCount <= 0)
172 {
173 delete this;
174 }
175}

181void

184{
185 m_ReferenceCountLock.Lock();
186 m_ReferenceCount = ref;
187 m_ReferenceCountLock.Unlock();

189 if ( ref <= 0)

213void
214LightObject
215::PrintSelf(std::ostream& os) const
216{
217 const char* indent = " ";
218
219#ifdef GCC_USEDEMANGLE
220 char const * mangledName = typeid(*this).name();
221 int status;
222 char* unmangled = abi::__cxa_demangle(mangledName, 0, 0, &status);
223
224 os << indent << "RTTI typeinfo: ";
225
226 if(status == 0)
227 {
228 os << unmangled;
229 free(unmangled);
230 }
231 else
232 {
233 os << mangledName;
234 }
235
236 os << std::endl;
237#else
238 os << indent << "RTTI typeinfo: " << typeid( *this ).name() << std::endl;
239#endif
240 os << indent << "Reference Count: " << m_ReferenceCount << std::endl;
241}
