81igtl::MessageBase::Pointer MessageBase::Clone()
82{
83 igtl::MessageBase::Pointer clone;
84 {
85 igtl::MessageFactory::Pointer factory = igtl::MessageFactory::New();
86 clone = factory->CreateSendMessage(this->GetMessageType(), this->GetHeaderVersion());
87 }
88
89 igtlUint64 bodySize = this->m_MessageSize - IGTL_HEADER_SIZE;
90 clone->InitBuffer();
91 clone->CopyHeader(this);
92 clone->AllocateBuffer(bodySize);
93 if (bodySize > 0)
94 {
95 clone->CopyBody(this);
96 }
97
98#if OpenIGTLink_HEADER_VERSION >= 2
99 clone->m_MetaDataHeader = this->m_MetaDataHeader;
100 clone->m_MetaDataMap = this->m_MetaDataMap;
101 clone->m_IsExtendedHeaderUnpacked = this->m_IsExtendedHeaderUnpacked;
102#endif
103
104 return clone;
105}

326bool MessageBase::PackExtendedHeader()
327{
328 if( m_HeaderVersion == IGTL_HEADER_VERSION_2 )
329 {
330 //igtlUint64 aSize = m_MessageSize - IGTL_HEADER_SIZE - sizeof(igtl_extended_header);
331 if(m_MessageSize < (IGTL_HEADER_SIZE + sizeof(igtl_extended_header)))
332 {
333 // Ensure we have enough space to write the header
334 AllocateBuffer(0);
335 }
336
337 igtl_extended_header* extended_header = (igtl_extended_header*) m_ExtendedHeader;
338 extended_header->extended_header_size = sizeof(igtl_extended_header);
339 extended_header->meta_data_header_size = this->GetMetaDataHeaderSize();
340 extended_header->meta_data_size = this->GetMetaDataSize();
341 extended_header->message_id = this->GetMessageID();
342 igtl_extended_header_convert_byte_order(extended_header);
343
344 return true;
345 }
346
347 return false;

350bool MessageBase::PackMetaData()
351{
352 if( m_HeaderVersion == IGTL_HEADER_VERSION_2 )
353 {
354 if (m_MetaDataMap.size() > std::numeric_limits<igtl_uint16>::max())
355 {
356 return false;
357 }
358 igtl_uint16 index_count = static_cast<igtl_uint16>(m_MetaDataMap.size()); // first two byte are the total number of meta data
359 if(igtl_is_little_endian())
360 {
361 index_count = BYTE_SWAP_INT16(index_count);
362 }
363
364 // Pack meta data header key/encoding/value trios
365 memcpy(m_MetaDataHeader, &index_count, META_DATA_INDEX_COUNT_SIZE);
366
367 if (m_MetaDataMap.size() > 0)
368 {
369 unsigned char* entryPointer = &m_MetaDataHeader[META_DATA_INDEX_COUNT_SIZE];
370 for (MetaDataMap::const_iterator it = m_MetaDataMap.begin(); it != m_MetaDataMap.end();++it)
371 {
372 igtl_metadata_header_entry entry;
373 entry.key_size = it->first.length();
374 entry.value_encoding = it->second.first;
375 entry.value_size = it->second.second.length();
376
377 if(igtl_is_little_endian())
378 {
379 entry.key_size = BYTE_SWAP_INT16(entry.key_size);
380 entry.value_encoding = BYTE_SWAP_INT16(entry.value_encoding);
381 entry.value_size = BYTE_SWAP_INT32(entry.value_size);
382 }
383 memcpy(&entryPointer[0], &entry.key_size, sizeof(igtlUint16));
384 memcpy(&entryPointer[2], &entry.value_encoding, sizeof(igtlUint16));
385 memcpy(&entryPointer[4], &entry.value_size, sizeof(igtlUint32));
386
387 entryPointer += sizeof(igtl_metadata_header_entry);
388 }
389
390 // Pack meta data key/value pairs
391 std::string key("");
392 std::string value("");
393
394 unsigned char* metaDataPointer = m_MetaData;
395 int i(0);
396 for (MetaDataMap::iterator it = m_MetaDataMap.begin(); it != m_MetaDataMap.end(); ++it, ++i)
397 {
398 memcpy(metaDataPointer, it->first.c_str(), it->first.length());
399 metaDataPointer += it->first.length();
400 memcpy(metaDataPointer, it->second.second.c_str(), it->second.second.length());
401 metaDataPointer += it->second.second.length();
402 }
403
404 return true;
405 }
406 else
407 {
408 return true;
409 }
410 }
411
412 return false;
413}

415bool MessageBase::UnpackExtendedHeader()
416{
417 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
418 {
419 igtl_extended_header* extended_header = (igtl_extended_header*)m_ExtendedHeader;
420 igtl_extended_header_convert_byte_order(extended_header);
421 if( extended_header->extended_header_size != sizeof(igtl_extended_header) )
422 {
423 // any extra data will be dropped, if the order of variables is changed, this will be seriously broken
424 // TODO : add error reporting?
425 }
426 this->m_MessageId = extended_header->message_id;
427
428 m_Content = &m_Body[extended_header->extended_header_size];
429 m_MetaDataHeader = &m_Body[m_BodySizeToRead - extended_header->meta_data_header_size - extended_header->meta_data_size];
430 m_MetaData = &m_Body[m_BodySizeToRead - extended_header->meta_data_size];
431
432 m_IsExtendedHeaderUnpacked = true;
433
434 return true;
435 }
436
437 return false;

440bool MessageBase::UnpackMetaData()
441{
442 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
443 {
444 // Parse the header
445 igtl_uint16 index_count = 0; // first two byte are the total number of meta data
446 memcpy(&index_count, m_MetaDataHeader, META_DATA_INDEX_COUNT_SIZE);
447 if(igtl_is_little_endian())
448 {
449 index_count = BYTE_SWAP_INT16(index_count);
450 }
451
452 if( index_count == 0 )
453 {
454 return true;
455 }
456
457 std::vector<igtl_metadata_header_entry> metaDataEntries;
458 igtl_metadata_header_entry entry;
459 unsigned char* entryPointer = &m_MetaDataHeader[META_DATA_INDEX_COUNT_SIZE];
460 const igtl_uint16 metaDataHeaderSize = ((igtl_extended_header*)m_ExtendedHeader)->meta_data_header_size;
461 igtl_uint32 currentMetaDataHeaderRead = 0;
462 for (int i = 0; i < index_count; i++)
463 {
464 currentMetaDataHeaderRead += sizeof(igtl_metadata_header_entry);
465 if (currentMetaDataHeaderRead > metaDataHeaderSize)
466 {
467 return false;
468 }
469
470 memcpy(&entry.key_size, &entryPointer[0], sizeof(igtlUint16));
471 memcpy(&entry.value_encoding, &entryPointer[2], sizeof(igtlUint16));
472 memcpy(&entry.value_size, &entryPointer[4], sizeof(igtlUint32));
473 if(igtl_is_little_endian())
474 {
475 entry.key_size = BYTE_SWAP_INT16(entry.key_size);
476 entry.value_encoding = BYTE_SWAP_INT16(entry.value_encoding);
477 entry.value_size = BYTE_SWAP_INT32(entry.value_size);
478 }
479 metaDataEntries.push_back(entry);
480
481 entryPointer += sizeof(igtl_metadata_header_entry);
482 }
483
484 // Parse the meta data
485 m_MetaDataMap.clear();
486
487 unsigned char* metaDataPointer = m_MetaData;
488 const igtl_uint16 metaDataSize = ((igtl_extended_header*)m_ExtendedHeader)->meta_data_size;
489 igtl_uint32 currentMetaDataRead = 0;
490 for (int i = 0; i < index_count; i++)
491 {
492 currentMetaDataRead += metaDataEntries[i].key_size + metaDataEntries[i].value_size;
493 if (currentMetaDataRead > metaDataSize)
494 {
495 return false;
496 }
497 std::string key;
498 key.assign(metaDataPointer, metaDataPointer + metaDataEntries[i].key_size);
499 metaDataPointer += metaDataEntries[i].key_size;
500 std::string value;
501 value.assign(metaDataPointer, metaDataPointer + metaDataEntries[i].value_size);
502 metaDataPointer += metaDataEntries[i].value_size;
503 m_MetaDataMap[key] = std::pair<IANA_ENCODING_TYPE, std::string>((IANA_ENCODING_TYPE)metaDataEntries[i].value_encoding, value);
504 }
505
506 return true;
507 }
508
509 return false;
510}

541int MessageBase::Pack()
542{
543 if (m_IsBodyPacked)
544 {
545 // Allow for multiple packs
546 return 1;
547 }
548
549 if( m_SendMessageType.empty() )
550 {
551 // We do not allow sending of base class messages, aka igtl::MessageBase
552 // TODO : error reporting?
553 return 0;
554 }
555
556#if OpenIGTLink_HEADER_VERSION >= 2
557 PackExtendedHeader();
558#endif
559
560 // Derived classes will re-call allocate pack with their required content size
561 PackContent();
562
563#if OpenIGTLink_HEADER_VERSION >= 2
564 PackMetaData();
565#endif
566
567 m_IsBodyPacked = true;
568
569 // pack header
570 igtl_header* h = (igtl_header*) m_Header;
571
572 igtl_uint64 crc = crc64(0, 0, 0LL); // initial crc
573 h->header_version = m_HeaderVersion;
574
575 igtl_uint64 ts = m_TimeStampSec & 0xFFFFFFFF;
576 ts = (ts << 32) | (m_TimeStampSecFraction & 0xFFFFFFFF);
577
578 h->timestamp = ts;
579 h->body_size = GetBufferBodySize();
580 h->crc = crc64((unsigned char*)m_Body, h->body_size, crc);
581
582 strncpy(h->name, m_SendMessageType.c_str(), 12);
583
584 strncpy(h->device_name, m_DeviceName.c_str(), 20);
585
586 igtl_header_convert_byte_order(h);
587
588 return 1;
589}

591int MessageBase::Unpack(int crccheck)
592{
593 int r = UNPACK_UNDEF;
594
595 // Check if the pack exists and if it has not been unpacked.
596 if (m_Header != NULL && m_MessageSize >= IGTL_HEADER_SIZE && !m_IsHeaderUnpacked )
597 {
598 InitBuffer();
599 UnpackHeader(r);
600 }
601
602#if OpenIGTLink_HEADER_VERSION >= 2
603 // Check if the body exists and it has not been unpacked
604 // The extended header is technically located inside the body, so we have to check to see if the remaining body size
605 // is > 0, or if full body size > sizeof(igtl_extended_header)
606 if( m_HeaderVersion >= IGTL_HEADER_VERSION_2 )
607 {
608 if (GetBufferBodySize() > static_cast<int>(sizeof(igtl_extended_header)) + META_DATA_INDEX_COUNT_SIZE && !m_IsBodyUnpacked)
609 {
610 UnpackBody(crccheck, r);
611 }
612 }
613 else
614 {
615#endif
616 if(GetBufferBodySize() > 0 && !m_IsBodyUnpacked)
617 {
618 UnpackBody(crccheck, r);
619 }
620#if OpenIGTLink_HEADER_VERSION >= 2
621 }
622#endif
623
624 return r;
625}

699void MessageBase::InitBuffer()
700{
701 m_IsHeaderUnpacked = false;
702 m_IsBodyPacked = false;
703 m_IsBodyUnpacked = false;
704 m_BodySizeToRead = 0;
705
706 m_DeviceName = "";
707 m_ReceiveMessageType = "";
708
709 // Re-allocate header area
710 igtlUint64 message_size = IGTL_HEADER_SIZE;
711
712 if (m_Header == NULL)
713 {
714 // For the first time
715 m_Header = new unsigned char [message_size];
716 m_IsHeaderUnpacked = false;
717 m_IsBodyUnpacked = false;
718 }
719 else if (m_MessageSize != message_size)
720 {
721 // If the pack area exists but needs to be reallocated
722 // m_IsHeaderUnpacked status is not changed in this case.
723 unsigned char* old = m_Header;
724 m_Header = new unsigned char [message_size];
725 memcpy(m_Header, old, std::min<int>(m_MessageSize, message_size));
726 delete [] old;
727 m_IsBodyUnpacked = false;
728 }
729 m_Body = &m_Header[IGTL_HEADER_SIZE];
730#if OpenIGTLink_HEADER_VERSION >= 2
731 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
732 {
733 m_ExtendedHeader = m_Body;
734 // Other members can't be populated until the message is unpacked
735 }
736 else
737 {
738 m_Content = m_Body;
739 }
740#else
741 m_Content = m_Body;
742#endif
743 m_MessageSize = message_size;
744}

746void MessageBase::AllocateBuffer(igtlUint64 contentSize)
747{
748#if OpenIGTLink_HEADER_VERSION >= 2
749 igtlUint64 message_size(0);
750 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
751 {
752 message_size = IGTL_HEADER_SIZE + contentSize + sizeof(igtl_extended_header) + GetMetaDataHeaderSize() + GetMetaDataSize();
753 }
754 else
755 {
756 message_size = IGTL_HEADER_SIZE + contentSize;
757 }
758#else
759 igtlUint64 message_size = IGTL_HEADER_SIZE + contentSize;
760#endif
761
762 if (m_Header == NULL)
763 {
764 // For the first time
765 m_Header = new unsigned char [message_size];
766 m_IsHeaderUnpacked = false;
767 m_IsBodyUnpacked = false;
768 m_IsBodyPacked = false;
769 }
770 else if (m_MessageSize != message_size)
771 {
772 // If the pack area exists but needs to be reallocated
773 // m_IsHeaderUnpacked status is not changed in this case.
774 unsigned char* old = m_Header;
775 m_Header = new unsigned char [message_size];
776 memcpy(m_Header, old, std::min<int>(m_MessageSize, message_size));
777 delete [] old;
778 m_IsBodyUnpacked = false;
779 }
780 m_Body = &m_Header[IGTL_HEADER_SIZE];
781
782#if OpenIGTLink_HEADER_VERSION >= 2
783 if (m_HeaderVersion == IGTL_HEADER_VERSION_2)
784 {
785 m_ExtendedHeader = m_Body;

860void MessageBase::UnpackHeader(int& r)
861{
862 // Unpack (deserialize) the header
863 igtl_header* h = (igtl_header*) m_Header;
864 igtl_header_convert_byte_order(h);
865 m_TimeStampSecFraction = h->timestamp & 0xFFFFFFFF;
866 m_TimeStampSec = (h->timestamp >> 32 ) & 0xFFFFFFFF;
867 m_HeaderVersion = h->header_version;
868 m_BodySizeToRead = h->body_size;
869
870 char bodyType[IGTL_HEADER_TYPE_SIZE+1];
871 char deviceName[IGTL_HEADER_NAME_SIZE+1];
872
873 bodyType[IGTL_HEADER_TYPE_SIZE] = '\0';
874 deviceName[IGTL_HEADER_NAME_SIZE] = '\0';
875 strncpy(bodyType, h->name, IGTL_HEADER_TYPE_SIZE);
876 strncpy(deviceName, h->device_name, IGTL_HEADER_NAME_SIZE);
877
878 m_ReceiveMessageType = bodyType;
879 if( m_ReceiveMessageType.empty() )
880 {
881 // MessageBase class has been sent, this can't be good
882 // TODO : how to report an error
883 }
884 m_DeviceName = deviceName;
885
886 // Mark as unpacked.
887 m_IsHeaderUnpacked = true;
888 r |= UNPACK_HEADER;

891void MessageBase::UnpackBody(int crccheck, int& r)
892{
893 igtl_header* h = (igtl_header*) m_Header;
894 igtl_uint64 crc = crc64(0, 0, 0LL); // initial crc
895
896 if (crccheck)
897 {
898 // Calculate CRC of the body
899 crc = crc64((unsigned char*)m_Body, m_BodySizeToRead, crc);
900 }
901 else
902 {
903 crc = h->crc;
904 }
905
906 if (crc == h->crc)
907 {
908 bool unpackSuccessful = true;
909 // Unpack (deserialize) the Body
910#if OpenIGTLink_HEADER_VERSION >= 2
911 if (h->header_version >= 2)
912 {
913 unpackSuccessful &= UnpackExtendedHeader();
914 }
915#endif
916 unpackSuccessful &= UnpackContent();
917#if OpenIGTLink_HEADER_VERSION >= 2
918 if (h->header_version >= 2)
919 {
920 unpackSuccessful &= UnpackMetaData();
921 }
922#endif
923 m_IsBodyUnpacked = unpackSuccessful;
924 if (unpackSuccessful)
925 {
926 r |= UNPACK_BODY;
927 }
928 }
929 else
930 {
931 m_IsBodyUnpacked = false;
932 }

935void MessageBase::AllocateUnpack(igtl_uint64 bodySizeToRead)
936{
937 if (bodySizeToRead == 0)
938 {
939 bodySizeToRead = 0;
940 m_IsBodyUnpacked = false;
941 }
942
943 igtl_uint64 message_size = IGTL_HEADER_SIZE + bodySizeToRead;
944
945 if (m_Header == NULL)
946 {
947 // For the first time
948 m_Header = new unsigned char [message_size];
949 m_IsHeaderUnpacked = false;
950 m_IsBodyUnpacked = false;
951 m_IsBodyPacked = false;
952 }
953 else if (m_MessageSize != message_size)
954 {
955 // If the pack area exists but needs to be reallocated
956 // m_IsHeaderUnpacked status is not changed in this case.
957 unsigned char* old = m_Header;
958 m_Header = new unsigned char [message_size];
959 memcpy(m_Header, old, std::min<int>(m_MessageSize, message_size));
960 delete [] old;
961 m_IsBodyUnpacked = false;
962 }
963 m_Body = &m_Header[IGTL_HEADER_SIZE];
964
965#if OpenIGTLink_HEADER_VERSION >= 2
966 if (m_HeaderVersion >= IGTL_HEADER_VERSION_2)
967 {
968 m_ExtendedHeader = m_Body;
969 // Other members can't be populated until the message is unpacked
970 }
971 else
972 {
973#endif
974 m_Content = m_Body;
975#if OpenIGTLink_HEADER_VERSION >= 2
976 }
977#endif
978
979 m_MessageSize = message_size;
980}

982int MessageBase::Copy(const MessageBase* mb)
983{
984 if (this == mb)
985 {
986 return 0;
987 }
988
989 // Check if the destination (this class) is MessageBase or
990 // the source and destination class are the same type.
991 // The pack size is also checked if it is larger than the header size.
992 if ((m_SendMessageType.length() == 0 || m_SendMessageType == mb->m_ReceiveMessageType)
993 && mb->m_MessageSize >= IGTL_HEADER_SIZE)
994 {
995 // Set the header version before calling any functions, as it determines later behavior
996 m_HeaderVersion = mb->m_HeaderVersion;
997
998 igtlUint64 bodySize = mb->m_MessageSize - IGTL_HEADER_SIZE;
999 AllocateBuffer(bodySize);
1000 CopyHeader(mb);
1001 if (bodySize > 0)
1002 {
1003 CopyBody(mb);
1004 }
1005 return 1;
1006 }
1007 else
1008 {
1009 return 0;
1010 }
